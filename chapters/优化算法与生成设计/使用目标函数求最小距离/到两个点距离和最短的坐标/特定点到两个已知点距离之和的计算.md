# 特定点到两个已知点距离之和的计算

```python
import matplotlib.pyplot as plt
import math

# ---------------- 输入 ----------------
# 输入点 C 的坐标（手动输入 or 可替换为随机生成）
x = float(input('请输入0到20之间的任意数：\n'))  # x = random.uniform(0,20)
y = float(input('请输入0到15之间的任意数：\n'))  # y = random.uniform(0,15)

# ---------------- 初始点定义 ----------------
# 已存在的两个固定点 A、B（硬编码）
p1 = [2, 3]      # point A
p2 = [14, 12]    # point B
p3 = [x, y]      # point C

# ---------------- 距离计算函数 ----------------
def sum_of_distance(p1, p2, p3):
    # C → A
    dist_of_p3p1 = math.sqrt((p3[0] - p1[0]) ** 2 + (p3[1] - p1[1]) ** 2)
    print("C 到 A 的距离:", dist_of_p3p1)

    # C → B
    dist_of_p3p2 = math.sqrt((p3[0] - p2[0]) ** 2 + (p3[1] - p2[1]) ** 2)
    print("C 到 B 的距离:", dist_of_p3p2)

    # 总距离
    total_distance = dist_of_p3p1 + dist_of_p3p2
    print("距离之和:", total_distance)

    return total_distance

# ---------------- 执行计算 ----------------
sum_of_distance(p1, p2, p3)

# ---------------- 可视化 ----------------
plt.scatter([p1[0], p2[0], p3[0]], [p1[1], p2[1], p3[1]])
plt.show()
```
