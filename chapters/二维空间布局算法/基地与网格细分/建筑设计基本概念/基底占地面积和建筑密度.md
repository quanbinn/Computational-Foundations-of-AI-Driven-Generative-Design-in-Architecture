# 基底占地面积和建筑密度

## 实体实验1

![](/images/二维空间布局算法/基地与网格细分/建筑设计基本概念/基底占地面积和建筑密度/1a1.jpg)

```python
import matplotlib.pyplot as plt

# ===============================
# 设置绘图比例相等（防止变形）
# ===============================
plt.axis('equal')

# ===============================
# 矩形类（基础几何对象）
# ===============================
class Rectangle:
    """通过左下角坐标 + 长宽定义矩形"""
    def __init__(self, x, y, length, width):
        self.x = x              # 左下角 X
        self.y = y              # 左下角 Y
        self.length = length    # 长（X方向）
        self.width = width      # 宽（Y方向）

    # 面积
    def area(self):
        return self.length * self.width

    # 中心点坐标
    def centralCoordinates(self):
        x_center = self.x + self.length / 2
        y_center = self.y + self.width / 2
        return [x_center, y_center]

    # 绘制矩形
    def render(self):
        p1 = [self.x, self.y]
        p2 = [self.x + self.length, self.y]
        p3 = [self.x + self.length, self.y + self.width]
        p4 = [self.x, self.y + self.width]

        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color="green")
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color="green")
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color="green")
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color="green")

# ===============================
# 1. 建设用地
# ===============================
x_of_site = -13
y_of_site = -9
length_of_site = 37
width_of_site = 28

rect_of_site = Rectangle(x_of_site, y_of_site, length_of_site, width_of_site)
rect_of_site.render()

print("建设用地面积是", rect_of_site.area(), "平方米。")

# ===============================
# 2. 建筑红线（内缩）
# ===============================
property_line_offset = 2  # 红线退界

rect_of_property_line_rec = Rectangle(
    x_of_site + property_line_offset,
    y_of_site + property_line_offset,
    length_of_site - 2 * property_line_offset,
    width_of_site - 2 * property_line_offset
)
rect_of_property_line_rec.render()

print("建筑红线面积是", rect_of_property_line_rec.area(), "平方米。")

# ===============================
# 3. 建筑分区（人为布置）
# ===============================
zone1 = Rectangle(-7.8, 0, 3.9, 8.1)
zone2 = Rectangle(-3.9, 0, 3.9, 8.1)
zone3 = Rectangle(0, 0, 3.9, 8.1)
zone4 = Rectangle(3.9, 0, 3.9, 8.1)
zone5 = Rectangle(7.8, 0, 3.9, 8.1)
zone6 = Rectangle(11.7, 0, 3.9, 8.1)

# 放入列表（便于统一处理）
zones = [zone1, zone2, zone3, zone4, zone5, zone6]

# ===============================
# 4. 建筑面积（总建筑面积）
# ===============================
total_floor_area = 0

for zone in zones:
    zone.render()
    total_floor_area += zone.area()

print("建筑面积是", total_floor_area, "平方米。")

# ===============================
# 5. 容积率 = 总建筑面积 / 用地面积
# ===============================
floor_area_ratio = total_floor_area / rect_of_site.area()
print("容积率是", round(floor_area_ratio, 2))

# ===============================
# 6. 建筑占地面积（基底面积）
# ===============================
# ⚠️ 教学重点：
# 如果这些 zone 在同一层 → 可以直接求和
area_of_building_footprint = sum([zone.area() for zone in zones])

print("建筑占地面积是", area_of_building_footprint, "平方米。")

# ===============================
# 7. 建筑密度 = 占地面积 / 用地面积
# ===============================
buildingDensity = area_of_building_footprint / rect_of_site.area()
print("建筑密度是", round(buildingDensity, 2))

# ===============================
# 显示图形
# ===============================
plt.show()
```
