```python
import matplotlib.pyplot as plt

# ===============================
# 设置绘图比例相等，保证矩形比例正确
# ===============================
plt.axis('equal')

# ===============================
# 矩形类
# ===============================
class Rectangle:
    """矩形类：通过左下角坐标和尺寸定义矩形"""
    def __init__(self, x, y, length, width):
        """
        初始化矩形
        参数：
        x, y - 矩形左下角坐标
        length - 矩形在X方向长度
        width  - 矩形在Y方向宽度
        """
        self.x = x
        self.y = y
        self.length = length
        self.width = width        

    # ===========================
    # 获取属性方法
    # ===========================
    def getX(self): return self.x
    def getY(self): return self.y
    def getLength(self): return self.length
    def getWidth(self): return self.width

    # ===========================
    # 计算面积
    # ===========================
    def area(self):
        """返回矩形面积"""
        return self.length * self.width

    # ===========================
    # 获取中心点坐标
    # ===========================
    def centralCoordinates(self):
        """返回矩形中心点坐标 [x_center, y_center]"""
        x_center = self.x + self.length / 2
        y_center = self.y + self.width / 2
        return [x_center, y_center]

    # ===========================
    # 绘制矩形
    # ===========================
    def render(self):
        """在当前Matplotlib图中绘制矩形"""
        p1 = [self.x, self.y]  # 左下角
        p2 = [self.x + self.length, self.y]  # 右下角
        p3 = [self.x + self.length, self.y + self.width]  # 右上角
        p4 = [self.x, self.y + self.width]  # 左上角

        # 绘制四条边
        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color="green")
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color="green")
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color="green")
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color="green")	

# ===============================
# 示例：建设用地矩形
# ===============================
x_of_site = -13 
y_of_site = -9
length_of_site = 37
width_of_site = 28

# 创建建设用地矩形并绘制
rect_of_site = Rectangle(x_of_site, y_of_site, length_of_site, width_of_site)
rect_of_site.render()

# 输出建设用地面积
print("建设用地面积是", rect_of_site.area(), "平方米。")

# ===============================
# 示例：建筑红线矩形（距离边界property_line_offset）
# ===============================
property_line_offset = 2  # 与建设用地边界的距离

# 根据偏移量计算红线矩形位置和尺寸
x_of_property_line_rec = x_of_site + property_line_offset
y_of_property_line_rec = y_of_site + property_line_offset
length_of_property_line_rec = length_of_site - property_line_offset * 2
width_of_property_line_rec = width_of_site - property_line_offset * 2

# 创建建筑红线矩形并绘制
rect_of_property_line_rec = Rectangle(
    x_of_property_line_rec,
    y_of_property_line_rec,
    length_of_property_line_rec,
    width_of_property_line_rec
)
rect_of_property_line_rec.render()

# 输出建筑红线面积
print("建筑红线面积是", rect_of_property_line_rec.area(), "平方米。")

# ===============================
# 显示绘图
# ===============================
plt.show()
```
