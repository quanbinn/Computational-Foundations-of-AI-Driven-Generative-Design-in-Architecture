# 轮廓移动

### 基于参考向量的二维点集整体平移

```python
def move(data, p1, p2):
    """
    将一组二维点整体平移，使 p1 移动到 p2
    data: 点集 [[x1,y1],[x2,y2],...]
    p1: 原参考点
    p2: 目标参考点
    """

    # 点的数量
    length = len(data)

    # 计算平移向量（x方向与y方向的偏移量）
    xdiff = p2[0] - p1[0]
    ydiff = p2[1] - p1[1]

    # 对每一个点进行平移操作
    for i in range(0, length):
        # x坐标平移
        data[i][0] = data[i][0] + xdiff

        # y坐标平移
        data[i][1] = data[i][1] + ydiff

    # 返回平移后的点集
    return data


# ===============================
# 原始数据点（矩形/多边形顶点）
# ===============================
data = [[50, 30], [285, 77], [246, 273], [10, 226]]

# 参考起点（当前基准点）
p1 = [50, 30]

# 目标点（移动到的目标位置）
p2 = [0, 0]

# ===============================
# 执行平移操作
# ===============================
new_data = move(data, p1, p2)

# 输出结果（平移后的坐标）
print(new_data)
```
### 基于参考向量的二维多边形平移与可视化

```python
import matplotlib.pyplot as plt
import random

# 设置坐标轴比例相等（保证图形不变形）
plt.axis('equal')


class Site:
    def __init__(self, data_of_points):
        # data_of_points: [[x1,y1], [x2,y2], ...]
        self.data = data_of_points

    def getData(self):
        # 返回当前点数据
        return self.data

    def edges(self):
        # 返回点的数量（边界点数量）
        return len(self.data)

    def move(self, data, p1, p2):
        """
        将整个图形进行平移：
        p1 -> p2 表示移动向量
        """

        # 通过输入数据重新构造一个 Site（保持你的写法不变）
        site = Site(data).getData()

        # 注意：这里仍然使用 self.data 的长度（保持原语义）
        length = len(self.data)

        # 计算平移向量
        xdiff = p2[0] - p1[0]
        ydiff = p2[1] - p1[1]

        # 对所有点进行平移
        for i in range(0, length):
            site[i][0] = site[i][0] + xdiff
            site[i][1] = site[i][1] + ydiff

        # 将移动后的图形重新绘制
        Site(site).render()

    def render(self):
        """
        绘制多边形（按点顺序连线，并闭合）
        """

        data = self.data
        length = len(data)

        # 逐条边绘制（i -> i+1）
        for i in range(0, length - 1):
            plt.plot(
                [data[i][0], data[i + 1][0]],
                [data[i][1], data[i + 1][1]],
                color="green"
            )

        # 最后一条边：闭合多边形（最后一个点 -> 第一个点）
        plt.plot(
            [data[length - 1][0], data[0][0]],
            [data[length - 1][1], data[0][1]],
            color="green"
        )


# -------------------- 测试数据 --------------------
data = [[50, 30], [285, 77], [246, 273], [10, 226]]

Site(data).render()

# 定义平移向量：p1 -> p2
```

### 基于参考向量的二维多边形平移与双状态对比可视化

```python
import matplotlib.pyplot as plt
import random

# 设置坐标轴比例相等（保证图形不变形）
plt.axis('equal')


class Site:
    def __init__(self, data_of_points):
        # data_of_points: 多边形顶点数据，例如：
        # [[x1,y1], [x2,y2], [x3,y3], ...]
        self.data = data_of_points

    def getData(self):
        # 返回当前图形的所有点
        return self.data

    def edges(self):
        # 返回点的数量（也就是多边形的边数）
        return len(self.data)

    def move(self, data, p1, p2):
        """
        平移操作：
        输入：
            data - 原始点集
            p1   - 起点（参考点）
            p2   - 终点（目标点）
        含义：
            将整个图形按照 (p1 -> p2) 的向量进行平移
        """

        # 用输入数据重新构造 Site（保持你的原写法）
        site = Site(data).getData()

        # 图形点的数量
        length = len(self.data)

        # 计算平移向量（核心数学部分）
        # x方向移动量
        xdiff = p2[0] - p1[0]
        # y方向移动量
        ydiff = p2[1] - p1[1]

        # 对所有点进行平移
        for i in range(0, length):
            site[i][0] = site[i][0] + xdiff
            site[i][1] = site[i][1] + ydiff

        # 绘制移动后的图形（红色版本）
        Site(site).render2()

    def render1(self):
        """
        绘制原始图形（绿色）
        用于对比：初始状态
        """

        data = self.data
        length = len(data)

        # 依次连接相邻点
        for i in range(0, length - 1):
            plt.plot(
                [data[i][0], data[i + 1][0]],
                [data[i][1], data[i + 1][1]],
                color="green"
            )

        # 闭合多边形（最后一个点 -> 第一个点）
        plt.plot(
            [data[length - 1][0], data[0][0]],
            [data[length - 1][1], data[0][1]],
            color="green"
        )

    def render2(self):
        """
        绘制平移后的图形（红色）
        用于对比：移动后状态
        """

        data = self.data
        length = len(data)

        # 依次连接相邻点
        for i in range(0, length - 1):
            plt.plot(
                [data[i][0], data[i + 1][0]],
                [data[i][1], data[i + 1][1]],
                color="red"
            )

        # 闭合多边形
        plt.plot(
            [data[length - 1][0], data[0][0]],
            [data[length - 1][1], data[0][1]],
            color="red"
        )
```
