### 将多个坐标点写入矩阵

```python
import matplotlib.pyplot as plt

# ---------------- 场地尺寸定义 ----------------
length = 240
width = 200

# 用四个点表示矩形场地
site = [[0, 0], [length, 0], [length, width], [0, width]]  # print(site)

# ---------------- 绘制方法（场地边界） ----------------
plt.plot([site[0][0], site[1][0]], [site[0][1], site[1][1]], color="green")
plt.plot([site[1][0], site[2][0]], [site[1][1], site[2][1]], color="green")
plt.plot([site[2][0], site[3][0]], [site[2][1], site[3][1]], color="green")
plt.plot([site[3][0], site[0][0]], [site[3][1], site[0][1]], color="green")

# ---------------- 网格参数设置 ----------------
distance_in_length = 16   # X方向间距
distance_in_width = 16    # Y方向间距

# 计算划分段数（向下取整）
segments_in_length = int(length / distance_in_length)   # print(segments_in_length)
segments_in_width = int(width / distance_in_width)      # print(segments_in_width)

# ---------------- 生成X方向分点（不含边界） ----------------
xArrayInLength = []
for i in range(1, segments_in_length):
    xArrayInLength.append(distance_in_length * i)

# ---------------- 生成Y方向分点（不含边界） ----------------
yArrayInWidth = []
for i in range(1, segments_in_width):
    yArrayInWidth.append(distance_in_width * i)

# ---------------- 初始化交点二维列表 ----------------
# intersections 结构：
# [
#   [第1行所有交点],
#   [第2行所有交点],
#   ...
# ]
intersections = []
for _ in range(segments_in_width - 1):
    intersections.append([])

print(intersections)  # 空二维结构

# ---------------- 生成网格交点（核心） ----------------
# 外层：Y方向（行）
# 内层：X方向（列）
for j in range(segments_in_width - 1):
    for i in range(1, segments_in_length):
        intersections[j].append([
            xArrayInLength[i - 1],
            yArrayInWidth[j]
        ])

print(intersections)
```

### 打印多个点坐标

```python
import matplotlib.pyplot as plt

# ---------------- 场地尺寸定义 ----------------
length = 240
width = 200

# 用四个点表示矩形场地（顺时针）
site = [[0, 0], [length, 0], [length, width], [0, width]]  # print(site)

# ---------------- 绘制方法（场地边界） ----------------
plt.plot([site[0][0], site[1][0]], [site[0][1], site[1][1]], color="green")
plt.plot([site[1][0], site[2][0]], [site[1][1], site[2][1]], color="green")
plt.plot([site[2][0], site[3][0]], [site[2][1], site[3][1]], color="green")
plt.plot([site[3][0], site[0][0]], [site[3][1], site[0][1]], color="green")

# ---------------- 网格参数设置 ----------------
distance_in_length = 20   # X方向间距
distance_in_width = 20    # Y方向间距

# 计算划分段数（向下取整）
segments_in_length = int(length / distance_in_length)   # print(segments_in_length)
segments_in_width = int(width / distance_in_width)      # print(segments_in_width)

# ---------------- 生成X方向分点（不含边界） ----------------
xArrayInLength = []
for i in range(1, segments_in_length):
    # 等距分点：i * 间距
    xArrayInLength.append(distance_in_length * i)

# ---------------- 生成Y方向分点（不含边界） ----------------
yArrayInWidth = []
for i in range(1, segments_in_width):
    yArrayInWidth.append(distance_in_width * i)

# ---------------- 初始化二维交点数组 ----------------
# intersections 结构：
# [
#   [第1行所有点],
#   [第2行所有点],
#   ...
# ]
intersections = []
for _ in range(segments_in_width - 1):
    intersections.append([])

print(intersections)  # 初始为空结构

# ---------------- 生成网格交点（核心） ----------------
# 外层循环：Y方向（行）
# 内层循环：X方向（列）
for j in range(segments_in_width - 1):
    for i in range(1, segments_in_length):
        intersections[j].append([
            xArrayInLength[i - 1],
            yArrayInWidth[j]
        ])

print(intersections)

# 输出行数（Y方向网格层数）
print(len(intersections))

# ---------------- 绘制方法（网格点） ----------------
# 遍历二维数组并绘制所有交点
for j in range(len(intersections)):
    for i in range(len(intersections[j])):
        plt.scatter(
            intersections[j][i][0],
            intersections[j][i][1],
            c="red"
        )

# ---------------- 显示图形 ----------------
plt.axis('equal')  # 保证X/Y比例一致
plt.show()
```
