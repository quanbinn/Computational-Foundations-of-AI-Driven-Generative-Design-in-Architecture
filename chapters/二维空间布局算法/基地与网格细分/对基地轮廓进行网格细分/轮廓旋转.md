# 轮廓旋转

### 基于数量积的向量夹角计算

```python
# 真实的角度要考虑象限和顺时针/逆时针方向对角度的影响，例如：+-degree
# 本函数目前使用的是“点积 + 反余弦”，只能得到 0~180°，不带方向

import math

def angle(v1, v2):
    """
    计算两个二维向量之间的夹角（单位：度）

    注意：
    - 使用点积公式 cosθ = (v1·v2)/(|v1||v2|)
    - 使用 acos 得到的是 0~π（即 0~180°）
    - 不包含方向（无法区分顺/逆时针）
    """

    # =========================
    # 1. 点积（dot product）
    # =========================
    # v1 · v2 = x1*x2 + y1*y2
    dot_product = v1[0] * v2[0] + v1[1] * v2[1]

    # =========================
    # 2. 向量模长（magnitude）
    # =========================
    magnitude_of_v1 = math.sqrt(v1[0] ** 2 + v1[1] ** 2)
    magnitude_of_v2 = math.sqrt(v2[0] ** 2 + v2[1] ** 2)

    # =========================
    # 3. 计算 cosθ
    # =========================
    value_of_cosine = dot_product / (magnitude_of_v1 * magnitude_of_v2)

    # =========================
    # 4. 反余弦得到弧度角
    # =========================
    pi_of_angle = math.acos(value_of_cosine)

    # =========================
    # 5. 转换为角度制
    # =========================
    angle = pi_of_angle * 180 / math.pi

    return angle

# =========================
# 测试
# =========================
print(angle([235, 47], [1, 0]))
```

### 单点旋转的几何变换过程

```python
import math

# =========================
# 旋转角度（单位：度）
# 负值表示顺时针旋转
# 正值表示逆时针旋转
# =========================
angle = -11.31


# =========================
# 2D 旋转矩阵（标准公式）
# =========================
# [ cosθ  -sinθ ]
# [ sinθ   cosθ ]
# =========================
matrix2dRotation = [
    [math.cos(math.radians(angle)), -math.sin(math.radians(angle))],
    [math.sin(math.radians(angle)),  math.cos(math.radians(angle))]
]

print(matrix2dRotation)


def rotate(matrix, angle):
    """
    对二维向量进行旋转变换

    参数：
        matrix: [[x, y]] 形式的二维点
        angle : 当前代码未直接使用（保留结构一致性）

    返回：
        旋转后的新坐标 [x', y']
    """

    # 初始化结果向量
    new_matrix = [0, 0]

    # =========================
    # 向量 × 旋转矩阵（核心计算）
    # =========================

    # x' = x*cosθ + y*sinθ(符号由矩阵结构决定)
    new_matrix[0] = round(
        matrix[0][0] * matrix2dRotation[0][0] +
        matrix[0][1] * matrix2dRotation[0][1],
        2
    )

    # y' = x*sinθ + y*cosθ
    new_matrix[1] = round(
        matrix[0][0] * matrix2dRotation[1][0] +
        matrix[0][1] * matrix2dRotation[1][1],
        2
    )

    # 输出中间结果（教学用）
    print(new_matrix)

    return new_matrix


# =========================
# 测试数据
# =========================
matrix = [[235, 47]]

# 执行旋转
print(rotate(matrix, angle))
```

### 多个点旋转的几何变换过程

```python
import math

# 旋转角度（单位：度）
# =========================
# 注意：
# - 负值 = 顺时针旋转
# - 正值 = 逆时针旋转
angle = -11.309932474020195

# 2D 旋转矩阵（标准形式）
# =========================
# R(θ) =
# [ cosθ  -sinθ ]
# [ sinθ   cosθ ]
# =========================
matrix2dRotation = [
    [math.cos(math.radians(angle)), -math.sin(math.radians(angle))],
    [math.sin(math.radians(angle)),  math.cos(math.radians(angle))]
]

print(matrix2dRotation)


def rotate(matrix, angle):
    """
    对多个二维点进行旋转变换（围绕原点）

    参数：
        matrix: [[x1,y1], [x2,y2], ...]
        angle : 当前版本未直接使用（保留教学结构）

        new_matrix: 旋转后的点集
    """

    # 初始化结果矩阵

    new_matrix = [[0, 0], [0, 0], [0, 0], [0, 0]]

    # 核心：矩阵乘法（逐点计算）

    return new_matrix
```

### 展示旋转变换后的图形效果

```python
import matplotlib.pyplot as plt
import random

# 设置坐标轴比例相等（保证几何形状不变形）
plt.axis('equal')


class Site:
    def __init__(self, data_of_points):
        # data_of_points: 多边形顶点列表，例如：
        # [[x1,y1], [x2,y2], [x3,y3], ...]
        self.data = data_of_points

    def getData(self):
        # 返回当前图形数据
        return self.data

    def edges(self):
        # 返回多边形边数（顶点数量）
        return len(self.data)

    def move(self, data, p1, p2):
        """
        平移操作（核心几何变换）

        含义：
            将整个图形从 p1 平移到 p2
        """

        # 用输入数据重新构造（保持你原始写法）
        site = Site(data).getData()

        # 点的数量
        length = len(self.data)

        # =========================
        # 计算平移向量（核心）
        # =========================
        xdiff = p2[0] - p1[0]
        ydiff = p2[1] - p1[1]

        # =========================
        # 对每个点进行平移
        # =========================
        for i in range(0, length):
            site[i][0] = site[i][0] + xdiff
            site[i][1] = site[i][1] + ydiff

        # 输出平移后的结果（教学用）
        print(site)

        # 绘制移动后的图形（红色）
        Site(site).render2()

    def render1(self):
        """
        绘制图形（绿色版本）
        ——用于表示“原始状态”
        """

        data = self.data
        length = len(data)

        # 逐边连接相邻点
        for i in range(0, length - 1):
            plt.plot(
                [data[i][0], data[i + 1][0]],
                [data[i][1], data[i + 1][1]],
                color="green"
            )

        # 闭合多边形（最后一点 -> 第一点）
        plt.plot(
            [data[length - 1][0], data[0][0]],
            [data[length - 1][1], data[0][1]],
            color="green"
        )

    def render2(self):
        """
        绘制图形（红色版本）
        ——用于表示“变换后状态”
        """

        data = self.data
        length = len(data)

        # 逐边连接相邻点
        for i in range(0, length - 1):
            plt.plot(
                [data[i][0], data[i + 1][0]],
                [data[i][1], data[i + 1][1]],
                color="red"
            )

        # 闭合多边形
        plt.plot(
            [data[length - 1][0], data[0][0]],
            [data[length - 1][1], data[0][1]],
            color="red"
        )


# =========================
# 测试1：原始多边形（绿色）
# =========================
data = [[50, 30], [285, 77], [246, 273], [10, 226]]
Site(data).render1()

# =========================
# 测试2：另一个多边形（红色）
# =========================
data = [[0, 0], [235, 47], [196, 243], [-40, 196]]
Site(data).render2()

# =========================
# 测试3：矩形（绿色）
# =========================
data = [[0, 0], [240, 0], [240, 200], [0, 200]]
Site(data).render1()

plt.show()
```
