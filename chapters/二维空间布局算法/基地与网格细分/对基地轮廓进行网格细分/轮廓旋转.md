### 基于数量积的向量夹角计算

```python
import math

# ---------------- 角度计算说明 ----------------
# 真实的角度要考虑象限和顺时针/逆时针方向对角度的影响，例如：+-degree
# 本函数目前使用的是“点积 + 反余弦”，只能得到 0~180°，不带方向

def angle(v1, v2):
    """
    计算两个二维向量之间的夹角（单位：度）

    注意：
    - 使用点积公式 cosθ = (v1·v2)/(|v1||v2|)
    - 使用 acos 得到的是 0~π（即 0~180°）
    - 不包含方向（无法区分顺/逆时针）
    """

    # ---------------- 1. 点积（dot product） ----------------
    # v1 · v2 = x1*x2 + y1*y2
    dot_product = v1[0] * v2[0] + v1[1] * v2[1]

    # ---------------- 2. 向量模长（magnitude） ----------------
    magnitude_of_v1 = math.sqrt(v1[0] ** 2 + v1[1] ** 2)
    magnitude_of_v2 = math.sqrt(v2[0] ** 2 + v2[1] ** 2)

    # ---------------- 3. 计算 cosθ ----------------
    value_of_cosine = dot_product / (magnitude_of_v1 * magnitude_of_v2)

    # ---------------- 4. 反余弦得到弧度角 ----------------
    pi_of_angle = math.acos(value_of_cosine)

    # ---------------- 5. 转换为角度制 ----------------
    angle_value = pi_of_angle * 180 / math.pi

    return angle_value

# ---------------- 测试 ----------------
print(angle([235, 47], [1, 0]))
```

```python
# 真实的角度要考虑象限和顺时针/逆时针方向对角度的影响，例如：+-degree

def angle(v1, v2):
    """
    计算两个二维向量之间的夹角（单位：度）

    注意：
    - 使用点积公式 cosθ = (v1·v2)/(|v1||v2|)
    - 使用 acos 得到的是 0~π（即 0~180°）
    - 不包含方向（无法区分顺/逆时针）
    """

    # ---------------- 1. 点积（dot product） ----------------
    # v1 · v2 = x1*x2 + y1*y2
    dot_product = v1[0] * v2[0] + v1[1] * v2[1]

    # ---------------- 2. 向量模长（magnitude） ----------------
    magnitude_of_v1 = math.sqrt(v1[0] ** 2 + v1[1] ** 2)
    magnitude_of_v2 = math.sqrt(v2[0] ** 2 + v2[1] ** 2)

    # ---------------- 3. 数值稳定处理（防止浮点误差） ----------------
    value_of_cosine = dot_product / (magnitude_of_v1 * magnitude_of_v2)
    value_of_cosine = max(-1.0, min(1.0, value_of_cosine))

    # ---------------- 4. 反余弦得到弧度角 ----------------
    pi_of_angle = math.acos(value_of_cosine)

    # ---------------- 5. 转换为角度制 ----------------
    angle_value = pi_of_angle * 180 / math.pi

    return angle_value

print(angle([235, 47], [1, 0]))
```

### 多个点旋转的几何变换过程

```python
import math

# ---------------- 旋转角度定义（单位：度） ----------------
# 注意：
# - 负值 = 顺时针旋转
# - 正值 = 逆时针旋转
angle = -11.309932474020195

# ---------------- 2D 旋转矩阵（标准形式） ----------------
# R(θ) =
# [ cosθ  -sinθ ]
# [ sinθ   cosθ ]
matrix2dRotation = [
    [math.cos(math.radians(angle)), -math.sin(math.radians(angle))],
    [math.sin(math.radians(angle)),  math.cos(math.radians(angle))]
]

print(matrix2dRotation)

def rotate(matrix, angle):
    """
    对多个二维点进行旋转变换（围绕原点）

    参数：
        matrix: [[x1,y1], [x2,y2], ...]

    """

    new_matrix = [[0, 0], [0, 0], [0, 0], [0, 0]]

    # 每个点 (x, y) 变换公式：
```

### 展示旋转变换后的图形效果

```python
import matplotlib.pyplot as plt
import random

# ---------------- 坐标轴设置（保证几何比例不变形） ----------------
plt.axis('equal')

# ---------------- 场地类（多边形结构） ----------------
class Site:
    def __init__(self, data_of_points):
        # data_of_points: 多边形顶点列表，例如：
        # [[x1,y1], [x2,y2], [x3,y3], ...]
        self.data = data_of_points

    # ---------------- 数据访问 ----------------
    def getData(self):
        return self.data  # 返回当前图形数据

    def edges(self):
        return len(self.data)  # 返回多边形边数（顶点数）

    # ---------------- 几何变换：平移 ----------------
    def move(self, data, p1, p2):
        """
        平移操作（核心几何变换）

        含义：
            将整个图形从 p1 平移到 p2
        """

        # 用输入数据重新构造（保持你原始写法）
        site = Site(data).getData()

        # 点数量
        length = len(self.data)

        # ---------------- 计算平移向量（核心） ----------------
        xdiff = p2[0] - p1[0]
        ydiff = p2[1] - p1[1]

        # ---------------- 执行平移变换 ----------------
        for i in range(length):
            site[i][0] += xdiff
            site[i][1] += ydiff

        # 输出平移结果（教学观察用）
        print(site)

        # ---------------- 绘制方法 ----------------
        # 红色表示变换后的结果
        Site(site).render2()

    # ---------------- 绘制方法（原始：绿色） ----------------
    def render1(self):
        """
        绘制图形（绿色版本）
        ——用于表示“原始状态”
        """

        data = self.data
        length = len(data)

        # ---------------- 连接相邻顶点 ----------------
        for i in range(length - 1):
            plt.plot(
                [data[i][0], data[i + 1][0]],
                [data[i][1], data[i + 1][1]],
                color="green"
            )

        # ---------------- 闭合多边形 ----------------
        plt.plot(
            [data[length - 1][0], data[0][0]],
            [data[length - 1][1], data[0][1]],
            color="green"
        )

    # ---------------- 绘制方法（变换后：红色） ----------------
    def render2(self):
        """
        绘制图形（红色版本）
        ——用于表示“变换后状态”
        """

        data = self.data
        length = len(data)

        # ---------------- 连接相邻顶点 ----------------
        for i in range(length - 1):
            plt.plot(
                [data[i][0], data[i + 1][0]],
                [data[i][1], data[i + 1][1]],
                color="red"
            )

        # ---------------- 闭合多边形 ----------------
        plt.plot(
            [data[length - 1][0], data[0][0]],
            [data[length - 1][1], data[0][1]],
            color="red"
        )


# ---------------- 测试1：原始多边形（绿色） ----------------
data = [[50, 30], [285, 77], [246, 273], [10, 226]]
Site(data).render1()

# ---------------- 测试2：变换多边形（红色） ----------------
data = [[0, 0], [235, 47], [196, 243], [-40, 196]]
Site(data).render2()

# ---------------- 测试3：矩形（绿色） ----------------
data = [[0, 0], [240, 0], [240, 200], [0, 200]]
Site(data).render1()

# ---------------- 显示结果 ----------------
plt.show()
```
