# 创建基地轮廓类

### 定义带构造函数的类

```python
class Site:
    """场地类（由多个二维顶点构成的多边形）"""

    def __init__(self, points):
        """初始化；points：顶点坐标列表，如 [[x1,y1], ...]"""
        self.points = points

    # ---------------- 获取顶点数据 ----------------
    def getPoints(self):
        """获取所有顶点"""
        return self.points

    # ---------------- 获取边数量（由顶点决定） ----------------
    def getEdges(self):
        """获取边数（顶点数）"""
        return len(self.points)

# ---------------- 示例：创建 Site 实例 ----------------
site = Site([[50, 30], [285, 77], [246, 273], [10, 226]])

# ---------------- 输出结果 ----------------
print(site.getPoints())
print(site.getEdges())
```

### 在类中增加多边形轮廓绘制方法

```python
import matplotlib.pyplot as plt

# ---------------- 坐标轴设置（等比例显示） ----------------
plt.axis('equal')

# ---------------- 场地类（多边形） ----------------
class Site:
    """场地类（由多个二维顶点构成的多边形）"""

    def __init__(self, points):
        """初始化；points：顶点坐标列表，如 [[x1,y1], ...]"""
        self.points = points

    # ---------------- 获取数据 ----------------
    def getPoints(self):
        return self.points  # 获取所有顶点

    def getEdges(self):
        return len(self.points)  # 获取边数（顶点数）

    # ---------------- 绘制方法 ----------------
    def render(self):
        """绘制多边形轮廓"""
        pts = self.points
        n = len(pts)

        # ---------------- 连接相邻顶点 ----------------
        for i in range(n - 1):
            plt.plot([pts[i][0], pts[i + 1][0]],
                     [pts[i][1], pts[i + 1][1]],
                     color="green")

        # ---------------- 闭合多边形 ----------------
        plt.plot([pts[n - 1][0], pts[0][0]],
                 [pts[n - 1][1], pts[0][1]],
                 color="green")

# ---------------- 创建 Site 实例并绘制 ----------------
site = Site([[50, 30], [285, 77], [246, 273], [10, 226]])
site.render()

# ---------------- 显示图形 ----------------
plt.show()
```
