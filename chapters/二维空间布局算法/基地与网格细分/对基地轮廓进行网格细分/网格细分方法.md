### **矩形基地长边与短边的等距细分点生成及可视化**

```python
import matplotlib.pyplot as plt
import matplotlib.pyplot as plt

# ===============================
# 1. 定义场地尺寸（单位：米）
# ===============================
length = 240   # 场地长度（X方向）
width = 200    # 场地宽度（Y方向）

# 用4个点定义矩形场地（按顺时针）
site = [
    [0, 0],                # 左下角
    [length, 0],           # 右下角
    [length, width],       # 右上角
    [0, width]             # 左上角
]

# ===============================
# 2. 绘制场地边界
# ===============================
plt.plot([site[0][0], site[1][0]], [site[0][1], site[1][1]], color="green")
plt.plot([site[1][0], site[2][0]], [site[1][1], site[2][1]], color="green")
plt.plot([site[2][0], site[3][0]], [site[2][1], site[3][1]], color="green")
plt.plot([site[3][0], site[0][0]], [site[3][1], site[0][1]], color="green")

# ===============================
# 3. 网格划分间距
# ===============================
distance_in_length = 16   # X方向间距
distance_in_width = 16    # Y方向间距

# 计算可以划分的段数（取整）
segments_in_length = int(length / distance_in_length)
segments_in_width  = int(width / distance_in_width)

# ===============================
# 4. 计算X方向分点（沿底边）
# ===============================
xArrayInLength = []   # 存储X坐标
yCoordInLength = []   # 对应Y坐标（全部为0）

for i in range(1, segments_in_length):
    xArrayInLength.append(distance_in_length * i)  # 等距分点
    yCoordInLength.append(0)                       # 位于底边

print("X方向分点：", xArrayInLength)

# ===============================
# 5. 计算Y方向分点（沿左边）
# ===============================
xCoordInWidth = []   # 对应X坐标（全部为0）
yArrayInWidth = []   # 存储Y坐标

for i in range(1, segments_in_width):
    xCoordInWidth.append(0)                        # 位于左边
    yArrayInWidth.append(distance_in_width * i)    # 等距分点

print("Y方向分点：", yArrayInWidth)

# ===============================
# 6. 绘制分点
# ===============================
plt.scatter(xArrayInLength, yCoordInLength, c="blue")  # 底边点（蓝色）
plt.scatter(xCoordInWidth, yArrayInWidth, c="red")     # 左边点（红色）

# ===============================
# 7. 显示图形
# ===============================
plt.axis('equal')  # 保证比例一致
plt.show()
```
