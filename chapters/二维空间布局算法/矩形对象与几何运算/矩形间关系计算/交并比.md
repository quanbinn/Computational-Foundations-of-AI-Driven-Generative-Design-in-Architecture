### 在矩形类中添加计算交并比（IoU）的方法
```python
import matplotlib.pyplot as plt
import numpy as np

# ---------------- 坐标设置 ----------------
plt.axis('equal')  # 坐标轴等比例显示

# ---------------- 定义矩形类 ----------------
class Rectangle:
    """二维矩形类：由左下角坐标、长度和宽度定义"""

    # ---------------- 初始化 ----------------
    def __init__(self, x, y, length, width):
        self.x1 = x
        self.y1 = y
        self.x2 = x + length
        self.y2 = y + width
        self.length = length
        self.width = width

    # ---------------- 属性访问 ----------------
    def getx1(self): return self.x1
    def gety1(self): return self.y1
    def getx2(self): return self.x2
    def gety2(self): return self.y2
    def getlength(self): return self.length
    def getwidth(self): return self.width

    # ---------------- 几何计算 ----------------
    def area(self):
        return self.length * self.width

    def centralCoordinates(self):
        return [(self.x1 + self.x2) / 2, (self.y1 + self.y2) / 2]

    # ---------------- 绘制方法 ----------------
    def render(self, color="green"):
        p1 = [self.x1, self.y1]
        p2 = [self.x2, self.y1]
        p3 = [self.x2, self.y2]
        p4 = [self.x1, self.y2]

        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color=color)
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color=color)
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color=color)
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color=color)

# ---------------- 创建矩形并绘制 ----------------
rect1 = Rectangle(0, 0, 4, 8)
rect1.render()

rect2 = Rectangle(2, 2, 8, 12)
rect2.render()

# ---------------- IoU计算函数 ----------------
def box_iou_xyxy(box1, box2):
    """
    计算两个矩形的交并比（IoU）
    box格式：[x1, y1, x2, y2]
    """
    x1min, y1min, x1max, y1max = box1
    x2min, y2min, x2max, y2max = box2

    # 面积计算
    area1 = (x1max - x1min) * (y1max - y1min)
    area2 = (x2max - x2min) * (y2max - y2min)
    print('矩形1面积:', area1)
    print('矩形2面积:', area2)

    # ---------------- 交集计算 ----------------
    xmin = np.maximum(x1min, x2min)
    ymin = np.maximum(y1min, y2min)
    xmax = np.minimum(x1max, x2max)
    ymax = np.minimum(y1max, y2max)

    inter_length = np.maximum(xmax - xmin, 0)
    inter_width = np.maximum(ymax - ymin, 0)
    inter_area = inter_length * inter_width
    print('交集面积:', inter_area)

    # ---------------- 并集计算 ----------------
    union_area = area1 + area2 - inter_area
    print('并集面积:', union_area)

    iou = inter_area / union_area
    return iou

# ---------------- IoU计算调用 ----------------
box1 = [rect1.getx1(), rect1.gety1(), rect1.getx2(), rect1.gety2()]
box2 = [rect2.getx1(), rect2.gety1(), rect2.getx2(), rect2.gety2()]

iou = box_iou_xyxy(box1, box2)
print('两个矩形的交并比IoU:', iou)

# ---------------- 图形显示 ----------------
plt.show()
```

### 矩形中心坐标与尺寸转对角顶点坐标

```python
import matplotlib.pyplot as plt
import numpy as np

# ---------------- 坐标设置 ----------------
plt.axis('equal')  # 设置坐标比例相等

# ---------------- 矩形类（中心点定义） ----------------
class RectangleCenter:
    """以矩形中心点和尺寸定义的矩形"""

    # ---------------- 初始化 ----------------
    def __init__(self, x_center, y_center, length, width):
        self.x_center = x_center
        self.y_center = y_center
        self.length = length
        self.width = width

        # 计算四个边界坐标
        self.x1 = x_center - length / 2
        self.y1 = y_center - width / 2
        self.x2 = x_center + length / 2
        self.y2 = y_center + width / 2

    # ---------------- 几何计算 ----------------
    def area(self):
        return self.length * self.width

    def centralCoordinates(self):
        return [self.x_center, self.y_center]

    # ---------------- 绘制方法 ----------------
    def render(self, color="green"):
        p1 = [self.x1, self.y1]
        p2 = [self.x2, self.y1]
        p3 = [self.x2, self.y2]
        p4 = [self.x1, self.y2]

        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color=color)
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color=color)
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color=color)
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color=color)

# ---------------- 创建矩形并绘制 ----------------
rect1 = RectangleCenter(10, 8, 12, 8)
rect1.render()

rect2 = RectangleCenter(20, 30, 16, 16)
rect2.render()

# ---------------- IoU计算函数 ----------------
def box_iou_xyxy(box1, box2):
    """计算两个矩形的交并比 IoU"""
    x1min, y1min, x1max, y1max = box1
    x2min, y2min, x2max, y2max = box2

    # 面积计算
    area1 = (x1max - x1min) * (y1max - y1min)
    area2 = (x2max - x2min) * (y2max - y2min)
    print('矩形1面积:', area1)
    print('矩形2面积:', area2)

    # ---------------- 交集计算 ----------------
    xmin = np.maximum(x1min, x2min)
    ymin = np.maximum(y1min, y2min)
    xmax = np.minimum(x1max, x2max)
    ymax = np.minimum(y1max, y2max)

    inter_length = np.maximum(xmax - xmin, 0)
    inter_width = np.maximum(ymax - ymin, 0)
    inter_area = inter_length * inter_width
    print('相交面积:', inter_area)

    # ---------------- 并集计算 ----------------
    union_area = area1 + area2 - inter_area
    print('并集面积:', union_area)

    iou = inter_area / union_area
    return iou

# ---------------- IoU计算调用 ----------------
box1 = [rect1.x1, rect1.y1, rect1.x2, rect1.y2]
box2 = [rect2.x1, rect2.y1, rect2.x2, rect2.y2]

iou = box_iou_xyxy(box1, box2)
print('两个矩形的交并比（IoU）:', iou)

# ---------------- 图形显示 ----------------
plt.show()
```
