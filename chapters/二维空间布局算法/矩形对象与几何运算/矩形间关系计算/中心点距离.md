### 在矩形类中添加从左下角坐标及宽高计算对角坐标的方法

### （输入：左下角坐标 `x, y` 和尺寸 `length, width`；输出：左下角与右上角坐标 `x1, y1, x2, y2`）

```python
import math
import matplotlib.pyplot as plt

plt.axis('equal')  # 设置坐标轴比例相等

# ==================== 矩形类 ====================
class Rectangle:
    """以左下角坐标和尺寸定义的矩形"""

    def __init__(self, x, y, length, width):
        # 左下角坐标
        self.x1 = x
        self.y1 = y
        # 右上角坐标
        self.x2 = x + length
        self.y2 = y + width
        # 尺寸
        self.length = length
        self.width = width

    # ========== 获取属性 ==========
    def getx1(self): return self.x1
    def gety1(self): return self.y1
    def getx2(self): return self.x2
    def gety2(self): return self.y2
    def getlength(self): return self.length
    def getwidth(self): return self.width

    # ========== 矩形面积 ==========
    def area(self):
        return self.length * self.width

    # ========== 矩形中心坐标 ==========
    def centralCoordinates(self):
        x_center = (self.x1 + self.x2) / 2
        y_center = (self.y1 + self.y2) / 2
        return [x_center, y_center]

    # ========== 绘制矩形轮廓 ==========
    def render(self, color="green"):
        p1, p2, p3, p4 = [self.x1, self.y1], [self.x2, self.y1], [self.x2, self.y2], [self.x1, self.y2]
        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color=color)
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color=color)
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color=color)
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color=color)

# ==================== 两矩形中心点距离函数 ====================
def distance_of_centers(rect1, rect2):
    """计算两个矩形中心点之间的欧氏距离"""
    x1, y1 = rect1.centralCoordinates()
    x2, y2 = rect2.centralCoordinates()
    distance = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
    return distance

# ==================== 示例使用 ====================
rect1 = Rectangle(10, 8, 11, 8)
rect1.render()  # 绘制矩形1

rect2 = Rectangle(20, 30, 11, 8)
rect2.render()  # 绘制矩形2

# 计算中心点距离
dist = distance_of_centers(rect1, rect2)
print('两个矩形中心点的距离是：', dist)

plt.show()
```

### 在矩形类中添加从中心点及宽高计算对角坐标的方法
### （输入：中心点坐标 x, y 和宽高 length, width；输出：x1, y1, x2, y2）

```python
import math
import matplotlib.pyplot as plt

plt.axis('equal')  # 设置坐标轴比例相等

# ==================== 矩形类（以中心点定义） ====================
class Rectangle_center:
    """以矩形中心点和尺寸定义矩形"""

    def __init__(self, xOfCenter, yOfCenter, length, width):
        # 中心点坐标
        self.xOfCenter = xOfCenter
        self.yOfCenter = yOfCenter
        # 根据中心点计算左下角和右上角坐标
        self.x1 = xOfCenter - length / 2
        self.y1 = yOfCenter - width / 2
        self.x2 = xOfCenter + length / 2
        self.y2 = yOfCenter + width / 2
        # 矩形尺寸
        self.length = length
        self.width = width        

    # ========== 获取属性 ==========
    def getx1(self): return self.x1
    def gety1(self): return self.y1
    def getx2(self): return self.x2
    def gety2(self): return self.y2
    def getlength(self): return self.length
    def getwidth(self): return self.width

    # ========== 矩形面积 ==========
    def area(self):
        return self.length * self.width

    # ========== 矩形中心坐标 ==========
    def centralCoordinates(self):
        return [self.xOfCenter, self.yOfCenter]

    # ========== 绘制矩形轮廓 ==========
    def render(self, color="green"):
        p1, p2, p3, p4 = [self.x1, self.y1], [self.x2, self.y1], [self.x2, self.y2], [self.x1, self.y2]
        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color=color)
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color=color)
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color=color)
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color=color)

# ==================== 两矩形中心点距离函数 ====================
def distance_of_centers(rect1, rect2):
    """计算两个矩形中心点之间的欧氏距离"""
    x1, y1 = rect1.centralCoordinates()
    x2, y2 = rect2.centralCoordinates()
    distance = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
    return distance

# ==================== 示例使用 ====================
rect1 = Rectangle_center(0, -5, 10, 10)
rect1.render()  # 绘制矩形1

rect2 = Rectangle_center(15, 5, 10, 10)
rect2.render()  # 绘制矩形2

# 计算中心点距离
dist = distance_of_centers(rect1, rect2)
print('两个矩形中心点的距离是：', dist)

plt.show()
```
