```python
class Rectangle:
    """矩形类：存储位置和尺寸，并提供面积及中心坐标计算"""

    # ---------------- 初始化 ----------------
    def __init__(self, x, y, length, width):
        # 参数：x,y（左下角坐标）；length（X方向）；width（Y方向）
        self.x = x
        self.y = y
        self.length = length
        self.width = width        

    # ---------------- 属性获取 ----------------
    def getX(self): return self.x
    def getY(self): return self.y
    def getLength(self): return self.length
    def getWidth(self): return self.width

    # ---------------- 面积计算 ----------------
    def area(self):
        return self.length * self.width

    # ---------------- 中心坐标 ----------------
    def centralCoordinates(self):
        x_center = self.x + self.length / 2
        y_center = self.y + self.width / 2
        return [x_center, y_center]

# ---------------- 使用示例 ----------------
rect1 = Rectangle(0, -5, 10, 10)  # 创建矩形对象

# 输出矩形宽度、面积和中心坐标
print(rect1.getWidth())
print(rect1.area())
print(rect1.centralCoordinates())
```

### 在类中添加矩形边框渲染方法

![](/images/二维空间布局算法/矩形对象与几何运算/矩形类的设计与实现/创建矩形类/4a1.jpg)

```python
import matplotlib.pyplot as plt

# ---------------- 坐标设置 ----------------
plt.axis('equal')  # 保持 XY 比例一致

class Rectangle:
    """矩形类：存储位置和尺寸，提供属性、面积、中心及绘制"""

    # ---------------- 初始化 ----------------
    def __init__(self, x, y, length, width):
        # x,y：左下角 | length：X方向 | width：Y方向
        self.x = x
        self.y = y
        self.length = length
        self.width = width        

    # ---------------- 属性访问 ----------------
    def getX(self): return self.x
    def getY(self): return self.y
    def getLength(self): return self.length
    def getWidth(self): return self.width

    # ---------------- 计算方法 ----------------
    def area(self):
        return self.length * self.width

    def centralCoordinates(self):
        x_center = self.x + self.length / 2
        y_center = self.y + self.width / 2
        return [x_center, y_center]

    # ---------------- 绘制方法 ----------------
    def render(self):
        # 四个顶点
        p1 = [self.x, self.y]
        p2 = [self.x + self.length, self.y]
        p3 = [self.x + self.length, self.y + self.width]
        p4 = [self.x, self.y + self.width]

        # 四条边
        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color="green")
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color="green")
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color="green")
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color="green")

# ---------------- 使用示例 ----------------
rect1 = Rectangle(10, 8, 11, 8)
rect2 = Rectangle(20, 30, 11, 8)

rect1.render()
rect2.render()

plt.show()
```

### 根据矩形中心坐标和尺寸计算左下角与右上角坐标
```python
import matplotlib.pyplot as plt

# ---------------- 坐标设置 ----------------
plt.axis('equal')

class RectangleCenter:
    """矩形类（中心点定义）：几何属性 + 计算 + 绘制"""

    # ---------------- 初始化 ----------------
    def __init__(self, x_center, y_center, length, width):
        self.x_center = x_center
        self.y_center = y_center
        self.length = length
        self.width = width
        
        # 边界坐标
        self.x1 = x_center - length / 2
        self.y1 = y_center - width / 2
        self.x2 = x_center + length / 2
        self.y2 = y_center + width / 2

    # ---------------- 属性 ----------------
    def getXCenter(self): return self.x_center
    def getYCenter(self): return self.y_center
    def getX1(self): return self.x1
    def getY1(self): return self.y1
    def getX2(self): return self.x2
    def getY2(self): return self.y2
    def getLength(self): return self.length
    def getWidth(self): return self.width

    # ---------------- 计算 ----------------
    def area(self):
        return self.length * self.width

    def centralCoordinates(self):
        return [self.x_center, self.y_center]
        
    # ---------------- 绘制方法 ----------------
    def render(self):
        p1 = [self.x1, self.y1]
        p2 = [self.x2, self.y1] 
        p3 = [self.x2, self.y2]
        p4 = [self.x1, self.y2]

        plt.plot([p1[0], p2[0]], [p1[1], p2[1]])
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]])
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]])
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]])

# ---------------- 示例 ----------------
rect1 = RectangleCenter(10, 8, 12, 8)
rect2 = RectangleCenter(20, 30, 16, 16)

rect1.render()
rect2.render()

print(rect1.centralCoordinates())
print(rect2.centralCoordinates())

plt.show()
```
