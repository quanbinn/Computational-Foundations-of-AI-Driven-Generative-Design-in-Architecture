### 定义带构造函数的类

```python
class Rectangle:
    """矩形类，存储位置及尺寸信息"""

    def __init__(self, x, y, length, width):
        """
        初始化矩形

        参数：
        x, y     : 左下角坐标
        length   : 矩形长度（X方向）
        width    : 矩形宽度（Y方向）
        """
        self.x = x
        self.y = y
        self.length = length
        self.width = width        

    # 获取矩形属性
    def getX(self): return self.x
    def getY(self): return self.y
    def getLength(self): return self.length
    def getWidth(self): return self.width
    	
# ================= 使用示例 =================
rect1 = Rectangle(0, -5, 10, 10)  # 创建矩形对象

# 输出矩形属性
print(rect1.getX())
print(rect1.getY())
print(rect1.getLength())
print(rect1.getWidth())
```

```python
class Rectangle:
    """矩形类：存储位置和尺寸，并提供面积及中心坐标计算"""

    def __init__(self, x, y, length, width):
        """
        初始化矩形

        参数：
        x, y     : 左下角坐标
        length   : 矩形长度（X方向）
        width    : 矩形宽度（Y方向）
        """
        self.x = x
        self.y = y
        self.length = length
        self.width = width        

    # 获取矩形属性
    def getX(self): return self.x
    def getY(self): return self.y
    def getLength(self): return self.length
    def getWidth(self): return self.width

    # 计算矩形面积
    def area(self):
        return self.length * self.width

    # 计算矩形中心坐标
    def centralCoordinates(self):
        x_center = self.x + self.length / 2
        y_center = self.y + self.width / 2
        return [x_center, y_center]

# ================= 使用示例 =================
rect1 = Rectangle(0, -5, 10, 10)  # 创建矩形对象

# 输出矩形宽度、面积和中心坐标
print(rect1.getWidth())
print(rect1.area())
print(rect1.centralCoordinates())
```

### 根据矩形左下角坐标及尺寸计算右上角坐标
```python
import matplotlib.pyplot as plt

plt.axis('equal')  # 保持 XY 坐标比例相等

class Rectangle:
    """矩形类：存储矩形位置和尺寸，提供面积、中心坐标及绘制功能"""

    def __init__(self, x, y, length, width):
        """
        初始化矩形

        参数：
        x, y     : 左下角坐标
        length   : 矩形长度（X方向）
        width    : 矩形宽度（Y方向）
        """
        self.x1 = x                  # 左下角 X
        self.y1 = y                  # 左下角 Y
        self.x2 = x + length         # 右上角 X
        self.y2 = y + width          # 右上角 Y
        self.length = length
        self.width = width            

    # ---------------- 属性访问方法 ----------------
    def getX1(self): return self.x1
    def getY1(self): return self.y1
    def getX2(self): return self.x2
    def getY2(self): return self.y2
    def getLength(self): return self.length
    def getWidth(self): return self.width

    # ---------------- 计算方法 ----------------
    def area(self):
        """计算矩形面积"""
        return self.length * self.width

    def centralCoordinates(self):
        """计算矩形中心坐标"""
        x_center = (self.x1 + self.x2) / 2
        y_center = (self.y1 + self.y2) / 2
        return [x_center, y_center]

    # ---------------- 绘制方法 ----------------
    def render(self):
        """在 matplotlib 坐标系中绘制矩形边界"""
        # 定义四个顶点
        p1 = [self.x1, self.y1]
        p2 = [self.x2, self.y1]
        p3 = [self.x2, self.y2]
        p4 = [self.x1, self.y2]
        # 绘制四条边
        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color="green")
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color="green")
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color="green")
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color="green")

# ================= 使用示例 =================
rect1 = Rectangle(10, 8, 11, 8)   # 创建第一个矩形
rect1.render()                     # 绘制矩形

rect2 = Rectangle(20, 30, 11, 8)  # 创建第二个矩形
rect2.render()                     # 绘制矩形

plt.show()  # 显示绘图
```

### 根据矩形中心坐标和尺寸计算左下角与右上角坐标
```python
import matplotlib.pyplot as plt

plt.axis('equal')  # 保持 XY 坐标比例相等

class RectangleCenter:
    """矩形类（以中心点为参考），提供面积、中心坐标及绘制功能"""

    def __init__(self, x_center, y_center, length, width):
        """
        初始化矩形（以中心点为参考）

        参数：
        x_center, y_center : 矩形中心坐标
        length             : 矩形长度（X方向）
        width              : 矩形宽度（Y方向）
        """
        self.x_center = x_center
        self.y_center = y_center
        self.length = length
        self.width = width
        
        # 计算左下角和右上角坐标
        self.x1 = x_center - length / 2
        self.y1 = y_center - width / 2
        self.x2 = x_center + length / 2
        self.y2 = y_center + width / 2

    # ---------------- 属性访问方法 ----------------
    def getXCenter(self): return self.x_center
    def getYCenter(self): return self.y_center
    def getX1(self): return self.x1
    def getY1(self): return self.y1
    def getX2(self): return self.x2
    def getY2(self): return self.y2
    def getLength(self): return self.length
    def getWidth(self): return self.width

    # ---------------- 计算方法 ----------------
    def area(self):
        """计算矩形面积"""
        return self.length * self.width

    def centralCoordinates(self):
        """返回中心坐标"""
        return [self.x_center, self.y_center]
        
    # ---------------- 绘制方法 ----------------
    def render(self):
        """在 matplotlib 坐标系中绘制矩形边界"""
        p1 = [self.x1, self.y1]
        p2 = [self.x2, self.y1] 
        p3 = [self.x2, self.y2]
        p4 = [self.x1, self.y2]

        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color="green")
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color="green")
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color="green")
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color="green")   

# ================= 使用示例 =================
rect1 = RectangleCenter(10, 8, 12, 8)  # 创建第一个矩形
rect1.render()                          # 绘制矩形
print(rect1.centralCoordinates())       # 输出中心坐标

rect2 = RectangleCenter(20, 30, 16, 16) # 创建第二个矩形
rect2.render()                           # 绘制矩形
print(rect2.centralCoordinates())        # 输出中心坐标

plt.show()  # 显示绘图
```
