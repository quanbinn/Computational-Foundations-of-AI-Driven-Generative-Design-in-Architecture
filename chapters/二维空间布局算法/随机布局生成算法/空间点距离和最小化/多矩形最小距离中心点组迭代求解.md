# 多矩形最小距离中心点组迭代求解

## 开始做实体实验

![](/images/二维空间布局算法/随机布局生成算法/空间点距离和最小化/基地主次入口坐标表示/1a1.jpg)

```python
import math
import random
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.patches import Rectangle

random.seed(42)

# ------ 定义基地（左下角坐标 + 宽高） ------
site_x, site_y = 0, 0
site_width, site_height = 240, 200

# ------ 定义入口坐标（位于边界） ------
main_entrance = (120, 0)
secondary_entrance = (240, 100)

# ------ 定义矩形尺寸（长50m、宽30m）与数量 ------
rect_w = 50
rect_h = 30
n_rect = 6

# ---------------- IoU计算函数（判断矩形是否重叠） ----------------
def box_iou_xyxy(box1, box2):
    x1min, y1min, x1max, y1max = box1
    x2min, y2min, x2max, y2max = box2

    xmin = np.maximum(x1min, x2min)
    ymin = np.maximum(y1min, y2min)
    xmax = np.minimum(x1max, x2max)
    ymax = np.minimum(y1max, y2max)

    inter_w = np.maximum(xmax - xmin, 0.0)
    inter_h = np.maximum(ymax - ymin, 0.0)
    inter_area = inter_w * inter_h

    area1 = (x1max - x1min) * (y1max - y1min)
    area2 = (x2max - x2min) * (y2max - y2min)

    union_area = area1 + area2 - inter_area
    return inter_area / union_area

# ---------------- 中心点转矩形（xyxy格式） ----------------
def center_to_box(center, width, height):
    cx, cy = center
    x1 = cx - width / 2
    y1 = cy - height / 2
    x2 = cx + width / 2
    y2 = cy + height / 2
    return [x1, y1, x2, y2]

# ---------------- 定义可行域（保证矩形不越界） ----------------
cx_min = site_x + rect_w / 2
cx_max = site_x + site_width - rect_w / 2
cy_min = site_y + rect_h / 2
cy_max = site_y + site_height - rect_h / 2

# ---------------- clamp函数（限制数值范围） ----------------
def clamp(v, lo, hi):
    return max(lo, min(hi, v))

# ---------------- 判断矩形是否与其他矩形重叠 ----------------
def overlaps_others(centers, idx, center):
    new_box = center_to_box(center, rect_w, rect_h)

    for j in range(len(centers)):
        if j == idx:
            continue

        other_box = center_to_box(centers[j], rect_w, rect_h)

        if box_iou_xyxy(new_box, other_box) > 0:
            return True

    return False

# ---------------- 目标函数（距离主入口 + 次入口的总和） ----------------
def objective(centers):
    total = 0.0
    for c in centers:
        total += math.dist(c, main_entrance) + math.dist(c, secondary_entrance)
    return total

# ---------------- 初始化阶段 ----------------

centers = []
boxes = []

while len(centers) < n_rect:
    c = (
        random.uniform(cx_min, cx_max),
        random.uniform(cy_min, cy_max),
    )

    box = center_to_box(c, rect_w, rect_h)

    if not any(box_iou_xyxy(box, b) > 0 for b in boxes):
        centers.append(c)
        boxes.append(box)

best_centers = centers[:]
best_value = objective(best_centers)

# ⭐ ---------------- 路径记录（优化轨迹） ----------------
paths = [[c] for c in best_centers]


# ---------------- 优化迭代阶段 ----------------

step = 10.0
n_iter = 3000

for _ in range(n_iter):

    idx = random.randrange(n_rect)

    new_centers = list(best_centers)

    old_x, old_y = new_centers[idx]

    new_x = clamp(old_x + random.uniform(-step, step), cx_min, cx_max)
    new_y = clamp(old_y + random.uniform(-step, step), cy_min, cy_max)

    new_centers[idx] = (new_x, new_y)

    if overlaps_others(new_centers, idx, new_centers[idx]):
        continue

    new_value = objective(new_centers)

    if new_value < best_value:
        best_centers = new_centers
        best_value = new_value

        paths[idx].append((new_x, new_y))

# ---------------- 输出结果 ----------------

print("优化后中心点组：")
for i, c in enumerate(best_centers, start=1):
    print(f"  第{i}个: ({round(c[0], 3)}, {round(c[1], 3)})")

print("最小距离和:", round(best_value, 3))

# ---------------- 绘制方法：可视化 ----------------

fig, ax = plt.subplots(figsize=(8, 5))

# ---------------- 绘制方法：基地轮廓 ----------------
ax.add_patch(Rectangle(
    (site_x, site_y),
    site_width,
    site_height,
    fill=False,
    linewidth=2
))

# ---------------- 绘制方法：入口点 ----------------
ax.scatter(*main_entrance, s=90)
ax.scatter(*secondary_entrance, s=90)

# ---------------- 绘制方法：优化路径 ----------------
for path in paths:
    xs = [p[0] for p in path]
    ys = [p[1] for p in path]
    ax.plot(xs, ys, linewidth=1)

# ---------------- 绘制方法：最终矩形 ----------------
for c in best_centers:
    ax.add_patch(Rectangle(
        (c[0] - rect_w / 2, c[1] - rect_h / 2),
        rect_w,
        rect_h,
        fill=False,
        linewidth=2
    ))

# ---------------- 绘制方法：坐标轴设置 ----------------
ax.set_xlim(-5, site_width + 5)
ax.set_ylim(-5, site_height + 5)
ax.set_aspect("equal")
ax.grid(False)

plt.show()
```
