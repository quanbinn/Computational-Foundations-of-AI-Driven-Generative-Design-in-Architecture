# 多矩形中心点距离和计算

## 开始做实体实验

![](/images/二维空间布局算法/随机布局生成算法/空间点距离和最小化/多矩形中心点距离和计算/1a1.jpg)

```python
import math
import random
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.patches import Rectangle

# ------ 定义基地（左下角坐标 + 宽高） ------
site_x, site_y = 0, 0
site_width, site_height = 240, 200

# ------ 定义入口坐标（位于边界） ------
main_entrance = (120, 0)         # 主入口
secondary_entrance = (240, 100)  # 次入口

# ------ 定义矩形尺寸：长50米、宽30米 ------
rect_w = 50
rect_h = 30

# ------ 需要随机生成的矩形数量 ------
n_rect = 4

# ------ IoU计算函数（参照你的 box_iou_xyxy 写法） ------
def box_iou_xyxy(box1, box2):
    x1min, y1min, x1max, y1max = box1
    x2min, y2min, x2max, y2max = box2

    area1 = (x1max - x1min) * (y1max - y1min)
    area2 = (x2max - x2min) * (y2max - y2min)

    xmin = np.maximum(x1min, x2min)
    ymin = np.maximum(y1min, y2min)
    xmax = np.minimum(x1max, x2max)
    ymax = np.minimum(y1max, y2max)

    inter_w = np.maximum(xmax - xmin, 0.0)
    inter_h = np.maximum(ymax - ymin, 0.0)
    inter_area = inter_w * inter_h

    union_area = area1 + area2 - inter_area
    return inter_area / union_area

# ------ 工具函数：由中心点得到矩形边界 [x1, y1, x2, y2] ------
def center_to_box(center, width, height):
    cx, cy = center
    x1 = cx - width / 2
    y1 = cy - height / 2
    x2 = cx + width / 2
    y2 = cy + height / 2
    return [x1, y1, x2, y2]

# ------ 随机生成中心点（确保矩形边界不超出基地边界） ------
cx_min = site_x + rect_w / 2
cx_max = site_x + site_width - rect_w / 2
cy_min = site_y + rect_h / 2
cy_max = site_y + site_height - rect_h / 2

centers = []
boxes = []

max_trials = 5000
trials = 0

while len(centers) < n_rect and trials < max_trials:
    trials += 1

    candidate_center = (
        random.uniform(cx_min, cx_max),
        random.uniform(cy_min, cy_max),
    )
    candidate_box = center_to_box(candidate_center, rect_w, rect_h)

    # 与已有矩形做IoU检查：IoU==0 才接受（不重叠）
    overlap = False
    for old_box in boxes:
        if box_iou_xyxy(candidate_box, old_box) > 0.0:
            overlap = True
            break

    if not overlap:
        centers.append(candidate_center)
        boxes.append(candidate_box)

if len(centers) < n_rect:
    raise RuntimeError("随机采样次数不足，未生成足够数量的不重叠矩形。")

# ------ 创建画布 ------
fig, ax = plt.subplots(figsize=(8, 5))

# ------ 绘制基地轮廓 ------
ax.add_patch(Rectangle(
    (site_x, site_y),
    site_width,
    site_height,
    fill=False,
    edgecolor="black",
    linewidth=2
))

# ------ 绘制入口点 ------
ax.scatter(*main_entrance, color="red", s=90, zorder=3)
ax.scatter(*secondary_entrance, color="blue", s=90, zorder=3)

# ------ 绘制标注 ------
ax.text(main_entrance[0] + 1, main_entrance[1] + 1.5, "Main", color="red")
ax.text(secondary_entrance[0] + 1, secondary_entrance[1] - 3, "Secondary", color="blue")

# ------ 计算并输出：每个矩形中心点到主次入口的距离和 ------
total_sum = 0.0
for i, center in enumerate(centers, start=1):
    d_main = math.dist(center, main_entrance)
    d_secondary = math.dist(center, secondary_entrance)
    d_sum = d_main + d_secondary
    total_sum += d_sum

    print(f"第{i}个矩形中心点坐标:", (round(center[0], 3), round(center[1], 3)))
    print("  矩形尺寸（长×宽，米）:", rect_w, "×", rect_h)
    print("  到主入口距离:", round(d_main, 3))
    print("  到次入口距离:", round(d_secondary, 3))
    print("  距离和:", round(d_sum, 3))
    print("-----")

    rect_x = center[0] - rect_w / 2
    rect_y = center[1] - rect_h / 2

    # 绘制矩形与中心点
    ax.add_patch(Rectangle(
        (rect_x, rect_y),
        rect_w,
        rect_h,
        fill=False,
        edgecolor="green",
        linewidth=2
    ))
    ax.scatter(*center, color="green", s=70, zorder=4)

    # 绘制中心点到主入口和次入口的连线
    ax.plot(
        [center[0], main_entrance[0]],
        [center[1], main_entrance[1]],
        "r--",
        linewidth=1.2,
        alpha=0.7
    )
    ax.plot(
        [center[0], secondary_entrance[0]],
        [center[1], secondary_entrance[1]],
        "b--",
        linewidth=1.2,
        alpha=0.7
    )

print("全部矩形中心点距离总和:", round(total_sum, 3))

# ------ 坐标轴设置 ------
ax.set_xlim(-5, site_width + 5)
ax.set_ylim(-5, site_height + 5)
ax.set_aspect("equal")
ax.set_xlabel("X")
ax.set_ylabel("Y")

# 取消网格显示
ax.grid(False)

plt.show()
```
