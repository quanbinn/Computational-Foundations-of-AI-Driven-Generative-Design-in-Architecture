# 单矩形最小距离中心点迭代求解

## 开始做实体实验

![](/images/二维空间布局算法/随机布局生成算法/空间点距离和最小化/单矩形最小距离中心点迭代求解/1a1.jpg)

```python
import math
import random
import matplotlib.pyplot as plt
from matplotlib.patches import Rectangle

# ------ 定义基地（左下角坐标 + 宽高） ------
site_x, site_y = 0, 0
site_width, site_height = 240, 200

# ------ 定义入口坐标 ------
main_entrance = (120, 0)
secondary_entrance = (240, 100)

# ------ 定义矩形尺寸 ------
rect_w, rect_h = 50, 30

# ------ 可行域（保证矩形不出界） ------
cx_min = site_x + rect_w / 2
cx_max = site_x + site_width - rect_w / 2
cy_min = site_y + rect_h / 2
cy_max = site_y + site_height - rect_h / 2


# ------ 目标函数：距离和（到主入口 + 次入口） ------
def objective(center):
    return math.dist(center, main_entrance) + math.dist(center, secondary_entrance)


# ------ 初始化（随机一个矩形中心点） ------
center = [
    random.uniform(cx_min, cx_max),
    random.uniform(cy_min, cy_max)
]

best_value = objective(center)


# ------ 参数设置（控制搜索行为） ------
step = 8.0        # 每次扰动的最大步长
n_iter = 200      # 迭代次数

# ------ 记录轨迹（用于可视化优化过程） ------
path = [tuple(center)]


print("========== 优化开始 ==========")
print("初始中心点:", (round(center[0], 3), round(center[1], 3)))
print("初始距离和:", round(best_value, 3))
print("-----")

# ------ 随机优化（核心算法：局部扰动 + 贪心接受） ------
for i in range(n_iter):

    # 在当前点附近随机扰动（局部搜索）
    new_x = center[0] + random.uniform(-step, step)
    new_y = center[1] + random.uniform(-step, step)

    # 限制在基地内（保证矩形不出界）
    new_x = max(cx_min, min(cx_max, new_x))
    new_y = max(cy_min, min(cy_max, new_y))

    new_center = [new_x, new_y]
    new_value = objective(new_center)

    # 贪心策略：只接受更优解
    if new_value < best_value:
        center = new_center
        best_value = new_value
        print(f"第{i+1}步：接受 → 距离和 = {round(best_value,3)}")
    else:
        print(f"第{i+1}步：拒绝")

    # 记录当前状态（用于画路径）
    path.append(tuple(center))

print("========== 优化结束 ==========")
print("最终中心点:", (round(center[0], 3), round(center[1], 3)))
print("最小距离和:", round(best_value, 3))

# ------ 创建画布 ------
fig, ax = plt.subplots(figsize=(8, 5))

# ---------------- 绘制方法：基地 ----------------
ax.add_patch(Rectangle(
    (site_x, site_y),
    site_width,
    site_height,
    fill=False,
    edgecolor="black",
    linewidth=2
))

# ---------------- 绘制方法：入口 ----------------
ax.scatter(*main_entrance, color="red", s=90)
ax.scatter(*secondary_entrance, color="blue", s=90)

# ---------------- 绘制方法：路径 ----------------
path_x = [p[0] for p in path]
path_y = [p[1] for p in path]

ax.plot(path_x, path_y, color="orange", linewidth=1.5)

# ---------------- 绘制方法：起点与终点 ----------------
# 起点
ax.scatter(path_x[0], path_y[0], color="purple", s=60)
ax.text(path_x[0], path_y[0], "Start")

# 终点
ax.scatter(path_x[-1], path_y[-1], color="green", s=80)
ax.text(path_x[-1], path_y[-1], "End")

# ---------------- 绘制方法：最终矩形 ----------------
rect_x = center[0] - rect_w / 2
rect_y = center[1] - rect_h / 2

ax.add_patch(Rectangle(
    (rect_x, rect_y),
    rect_w,
    rect_h,
    fill=False,
    edgecolor="green",
    linewidth=2
))

# ---------------- 绘制方法：距离连线 ----------------
ax.plot(
    [center[0], main_entrance[0]],
    [center[1], main_entrance[1]],
    "r--"
)

ax.plot(
    [center[0], secondary_entrance[0]],
    [center[1], secondary_entrance[1]],
    "b--"
)

# ---------------- 绘制方法：坐标轴 ----------------
ax.set_xlim(-5, site_width + 5)
ax.set_ylim(-5, site_height + 5)
ax.set_aspect("equal")
ax.grid(False)

plt.show()
```
