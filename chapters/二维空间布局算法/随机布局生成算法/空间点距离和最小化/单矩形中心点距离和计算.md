# 单矩形中心点距离和计算

## 开始做实体实验

![](/images/二维空间布局算法/随机布局生成算法/空间点距离和最小化/单矩形中心点距离和计算/1a1.jpg)

```python
import math
import random
import matplotlib.pyplot as plt
from matplotlib.patches import Rectangle

# ------ 定义基地（左下角坐标 + 宽高） ------
site_x, site_y = 0, 0
site_width, site_height = 240, 200

# ------ 定义入口坐标（位于边界） ------
main_entrance = (120, 0)        # 主入口
secondary_entrance = (240, 100)  # 次入口

# ------ 定义矩形：长度 50m、宽度 30m；以中心点表示，在基地内随机生成 ------
rect_w = 50   # 长度（水平方向尺寸，米）
rect_h = 30   # 宽度（竖直方向尺寸，米）

# 矩形边界不越界：中心点必须在 [rect_w/2, site_width-rect_w/2] × [rect_h/2, site_height-rect_h/2]
cx_min = site_x + rect_w / 2
cx_max = site_x + site_width - rect_w / 2
cy_min = site_y + rect_h / 2
cy_max = site_y + site_height - rect_h / 2

rect_center = (
    random.uniform(cx_min, cx_max),
    random.uniform(cy_min, cy_max),
)
rect_x = rect_center[0] - rect_w / 2
rect_y = rect_center[1] - rect_h / 2

# ------ 计算距离和：中心点到主入口 + 到次入口 ------
d_main = math.dist(rect_center, main_entrance)
d_secondary = math.dist(rect_center, secondary_entrance)
d_sum = d_main + d_secondary

# ------ 创建画布 ------
fig, ax = plt.subplots(figsize=(8, 5))

# ------ 绘制基地轮廓 ------
ax.add_patch(Rectangle(
    (site_x, site_y),
    site_width,
    site_height,
    fill=False,
    edgecolor="black",
    linewidth=2
))

# ------ 绘制入口点 ------
ax.scatter(*main_entrance, color="red", s=90, zorder=3)
ax.scatter(*secondary_entrance, color="blue", s=90, zorder=3)

# ------ 绘制标注 ------
ax.text(main_entrance[0] + 1, main_entrance[1] + 1.5, "Main", color="red")
ax.text(secondary_entrance[0] + 1, secondary_entrance[1] - 3, "Secondary", color="blue")

# ------ 绘制矩形轮廓与中心点 ------
ax.add_patch(Rectangle(
    (rect_x, rect_y),
    rect_w,
    rect_h,
    fill=False,
    edgecolor="green",
    linewidth=2
))
ax.scatter(*rect_center, color="green", s=70, zorder=4)

# ------ 绘制距离连线 ------
ax.plot(
    [rect_center[0], main_entrance[0]],
    [rect_center[1], main_entrance[1]],
    "r--",
    linewidth=1.5,
    alpha=0.8
)
ax.plot(
    [rect_center[0], secondary_entrance[0]],
    [rect_center[1], secondary_entrance[1]],
    "b--",
    linewidth=1.5,
    alpha=0.8
)

# ------ 坐标轴设置 ------
ax.set_xlim(-5, site_width + 5)
ax.set_ylim(-5, site_height + 5)
ax.set_aspect("equal")
ax.set_xlabel("X")
ax.set_ylabel("Y")

# 取消网格显示
ax.grid(False)

# ------ 输出信息 ------
print("矩形中心点坐标:", rect_center)
print("矩形尺寸（长×宽，米）:", rect_w, "×", rect_h)
print("到主入口距离:", round(d_main, 3))
print("到次入口距离:", round(d_secondary, 3))
print("距离和:", round(d_sum, 3))

plt.show()
```
