    def getyOfCenter(self): return self.yOfCenter
    def getx1(self): return self.x1
    def gety1(self): return self.y1
    def getx2(self): return self.x2
    def gety2(self): return self.y2
    def getWidth(self): return self.width
    def getHeight(self): return self.height

    # ---------------- 面积计算 ----------------
    def area(self):
        return self.width * self.height

    # ---------------- 中心点输出 ----------------
    def centralCoordinates(self):
        return [self.xOfCenter, self.yOfCenter]

    # ---------------- 绘制方法 ----------------

        p1 = [self.x1, self.y1]
        p2 = [self.x2, self.y1]
        p3 = [self.x2, self.y2]
        p4 = [self.x1, self.y2]

        plt.plot([p1[0], p2[0]], [p1[1], p2[1]], color="green")
        plt.plot([p2[0], p3[0]], [p2[1], p3[1]], color="green")
        plt.plot([p3[0], p4[0]], [p3[1], p4[1]], color="green")
        plt.plot([p4[0], p1[0]], [p4[1], p1[1]], color="green")

# ---------------- IoU计算函数 ----------------
def box_iou_xyxy(box1, box2):
    x1min, y1min, x1max, y1max = box1
    x2min, y2min, x2max, y2max = box2

    area1 = (x1max - x1min) * (y1max - y1min)
    area2 = (x2max - x2min) * (y2max - y2min)

    # ---------------- 交集计算 ----------------
    xmin = np.maximum(x1min, x2min)
    ymin = np.maximum(y1min, y2min)
    xmax = np.minimum(x1max, x2max)
    ymax = np.minimum(y1max, y2max)

    w = np.maximum(xmax - xmin, 0.0)
    h = np.maximum(ymax - ymin, 0.0)
    inter_area = w * h

    # ---------------- 并集计算 ----------------
    union_area = area1 + area2 - inter_area

    return inter_area / union_area

    area2 = (x2max - x2min) * (y2max - y2min)

    # ---------------- 交集计算 ----------------
    xmin = np.maximum(x1min, x2min)
    ymin = np.maximum(y1min, y2min)
    xmax = np.minimum(x1max, x2max)
    ymax = np.minimum(y1max, y2max)

    w = np.maximum(xmax - xmin, 0.0)
    h = np.maximum(ymax - ymin, 0.0)
    inter = w * h

    # ---------------- 并集计算 ----------------
    union = area1 + area2 - inter

    return inter / union
