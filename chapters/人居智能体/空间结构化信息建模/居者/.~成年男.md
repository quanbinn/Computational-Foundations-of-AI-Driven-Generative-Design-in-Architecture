#### 1个标准成年男子站立在房间正中心，面朝南

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 房间尺寸 ----------------
L, W, H = 4.2, 3.6, 3.0
t = 0.2
glass_t = 0.01
door_t = 0.02

# ---------------- 窗户 ----------------
windows = [
    {"width": 2.1, "height": 1.8, "x": 1.05, "z": 0.75}
]

# ---------------- 门 ----------------
doors = [
    {"x0": 0.2, "x1": 1.1, "z0": 0, "z1": 2.1}
]

# ---------------- 桌子 ----------------
table_size = (1.2, 0.6, 0.75)

# ---------------- 顶点 ----------------
P = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

# ---------------- cuboid ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz
    return [
        [(x,y,z),(x1,y,z),(x1,y1,z),(x,y1,z)],
        [(x,y,z1),(x1,y,z1),(x1,y1,z1),(x,y1,z1)],
        [(x,y,z),(x1,y,z),(x1,y,z1),(x,y,z1)],
        [(x1,y,z),(x1,y1,z),(x1,y1,z1),(x1,y,z1)],
        [(x1,y1,z),(x,y1,z),(x,y1,z1),(x1,y1,z1)],
        [(x,y1,z),(x,y,z),(x,y,z1),(x,y1,z1)],
    ]

# ---------------- 墙体 ----------------
def cut_wall_south(L, t, H, windows):
    faces, cursor = [], 0

    for w in windows:
        wx0, wx1 = w["x"], w["x"] + w["width"]
        wz0, wz1 = w["z"], w["z"] + w["height"]

        if wx0 > cursor:
            faces.extend(cuboid(cursor, -t/2, 0, wx0-cursor, t, H))

        if wz0 > 0:
            faces.extend(cuboid(wx0, -t/2, 0, w["width"], t, wz0))

        if wz1 < H:
            faces.extend(cuboid(wx0, -t/2, wz1, w["width"], t, H-wz1))

        cursor = wx1

    if cursor < L:
        faces.extend(cuboid(cursor, -t/2, 0, L-cursor, t, H))

    return faces


def cut_wall_north(L, W, t, H, doors):
    faces = []

    for d in doors:
        dx0, dx1, dz1 = d["x0"], d["x1"], d["z1"]

        if dx0 > 0:
            faces.extend(cuboid(0, W-t/2, 0, dx0, t, H))

        if dx1 < L:
            faces.extend(cuboid(dx1, W-t/2, 0, L-dx1, t, H))

        if dz1 < H:
            faces.extend(cuboid(dx0, W-t/2, dz1, dx1-dx0, t, H-dz1))

    return faces


# ---------------- 门窗 ----------------
def make_door_solid(W, doors, door_t):
    faces = []

    for d in doors:
        faces.extend(
            cuboid(
                d["x0"],
                W-door_t/2,
                d["z0"],
                d["x1"]-d["x0"],
                door_t,
                d["z1"]-d["z0"]
            )
        )

    return faces


def make_window_glass(windows, glass_t):
    faces = []

    for w in windows:
        faces.extend(
            cuboid(
                w["x"],
                -glass_t/2,
                w["z"],
                w["width"],
                glass_t,
                w["height"]
            )
        )

    return faces


# ---------------- 🪞 镜子（wall-4） ----------------
def make_mirror():
    mirror_w = 1.2
    mirror_h = 2.4
    mirror_t = 0.01

    y0 = (W - mirror_w) / 2
    z0 = 0.0
    x0 = t/2

    faces = []
    faces.extend(cuboid(x0, y0, z0, mirror_t, mirror_w, mirror_h))

    return faces


# ---------------- 桌子 ----------------
def make_table():
    tw, td, th = table_size

    top_t = 0.05
    leg_h = th - top_t
    leg_w = 0.05

    cx = windows[0]["x"] + windows[0]["width"]/2

    x0 = cx - tw/2
    y0 = 0.10

    faces = []

    # 桌面
    faces.extend(cuboid(x0, y0, leg_h, tw, td, top_t))

    # 桌腿
    legs = [
        (x0, y0),
        (x0+tw-leg_w, y0),
        (x0, y0+td-leg_w),
        (x0+tw-leg_w, y0+td-leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, leg_h))

    return faces, (x0, y0, tw, td)


# ---------------- 椅子 ----------------
def make_chair(table_info):
    x0, y0, tw, td = table_info

    cw, cd, ch = 0.45, 0.45, 0.45
    back_h = 0.45
    leg_w = 0.04

    cx = x0 + tw/2 - cw/2
    cy = y0 + td + 0.15

    faces = []

    # 座面
    faces.extend(cuboid(cx, cy, ch, cw, cd, 0.04))

    # 靠背
    faces.extend(cuboid(cx, cy+cd-0.04, ch, cw, 0.04, back_h))

    # 椅腿
    legs = [
        (cx, cy),
        (cx+cw-leg_w, cy),
        (cx, cy+cd-leg_w),
        (cx+cw-leg_w, cy+cd-leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, ch))

    return faces


# ---------------- 床 ----------------
def make_bed(dx=0.0, dy=0.0):
    bl, bw, bh = 1.8, 1.5, 0.45

    head_h = 0.9
    head_t = 0.08

    leg_w = 0.06
    top_t = 0.08
    leg_h = bh - top_t

    cx, cy = L/2, W/2

    x0 = cx - bl/2 + dx
    y0 = cy - bw/2 + dy

    faces = []

    # 床面
    faces.extend(cuboid(x0, y0, leg_h, bl, bw, top_t))

    # 床腿
    legs = [
        (x0, y0),
        (x0+bl-leg_w, y0),
        (x0, y0+bw-leg_w),
        (x0+bl-leg_w, y0+bw-leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, leg_h))

    # 床头
    faces.extend(
        cuboid(
            x0 + bl - head_t,
            y0,
            bh,
            head_t,
            bw,
            head_h - bh
        )
    )

    return faces

# ---------------- 成年男子（1.75m） ----------------
def make_man():
    """
    中国标准成年男子（简化体元表达）
    身高：1.75m
    站立于房间正中心
    """

    faces = []

    # 房间中心
    cx = L / 2
    cy = W / 2

    # ---------------- 身体参数 ----------------
    leg_h = 0.90
    torso_h = 0.55
    neck_h = 0.05
    head_h = 0.25

    shoulder_w = 0.38
    torso_d = 0.22

    leg_w = 0.10
    gap = 0.06

    arm_w = 0.08
    arm_d = 0.08
    arm_h = 0.60

    # ---------------- 双腿 ----------------
    left_leg_x = cx - gap/2 - leg_w
    right_leg_x = cx + gap/2

    y0 = cy - leg_w/2

    faces.extend(cuboid(left_leg_x, y0, 0, leg_w, leg_w, leg_h))
    faces.extend(cuboid(right_leg_x, y0, 0, leg_w, leg_w, leg_h))

    # ---------------- 躯干 ----------------
    torso_x = cx - shoulder_w/2
    torso_y = cy - torso_d/2

    faces.extend(
        cuboid(
            torso_x,
            torso_y,
            leg_h,
            shoulder_w,
            torso_d,
            torso_h
        )
    )

    # ---------------- 双臂 ----------------
    arm_z = leg_h + 0.05

    faces.extend(
        cuboid(
            torso_x - arm_w,
            cy - arm_d/2,
            arm_z,
            arm_w,
            arm_d,
            arm_h
        )
    )

    faces.extend(
        cuboid(
            torso_x + shoulder_w,
            cy - arm_d/2,
            arm_z,
            arm_w,
            arm_d,
            arm_h
        )
    )

    # ---------------- 头部 ----------------
    head_w = 0.22

    head_x = cx - head_w/2
    head_y = cy - head_w/2

    faces.extend(
        cuboid(
            head_x,
            head_y,
            leg_h + torso_h + neck_h,
            head_w,
            head_w,
            head_h
        )
    )

    return faces


# ---------------- flatten ----------------
def flatten(faces):
    return [np.array(f) for f in faces if np.array(f).ndim == 2]


# ---------------- add ----------------
def add(ax, faces, color):
    ax.add_collection3d(
        Poly3DCollection(
            faces,
            facecolors=color,
            edgecolors="black",
            linewidths=0.3,
            alpha=color[3]
        )
    )


# ---------------- 场景 ----------------
fig = plt.figure(figsize=(10,7))
ax = fig.add_subplot(111, projection='3d')

wall_faces = []

wall_faces.extend(cut_wall_south(L, t, H, windows))
wall_faces.extend(cuboid(L-t/2, 0, 0, t, W, H))
wall_faces.extend(cut_wall_north(L, W, t, H, doors))
wall_faces.extend(cuboid(-t/2, 0, 0, t, W, H))

table_faces, table_info = make_table()

chair_faces = make_chair(table_info)

bed_faces = make_bed(dx=1.1, dy=0.95)

mirror_faces = make_mirror()

man_faces = make_man()


# ---------------- 渲染 ----------------
add(ax, flatten(wall_faces), (0.2,0.6,1,0.25))

# 地面
add(ax, flatten(cuboid(0,0,-t,L,W,t)), (0.7,0.7,0.7,1))

# 窗玻璃
add(ax, flatten(make_window_glass(windows, glass_t)), (0.3,0.8,1,0.15))

# 门
add(ax, flatten(make_door_solid(W, doors, door_t)), (1,1,1,1))

# 家具
add(ax, flatten(table_faces), (0.55,0.35,0.2,1))
add(ax, flatten(chair_faces), (0.4,0.25,0.15,1))
add(ax, flatten(bed_faces), (0.8,0.7,0.6,1))

# 镜子
add(ax, flatten(mirror_faces), (0.95,0.95,1,0.6))

# 成年男子
add(ax, flatten(man_faces), (0.85,0.72,0.58,1))


# ---------------- 标注 ----------------
ax.text(L/2,0,H/2,"WALL-1")
ax.text(L,W/2,H/2,"WALL-2")
ax.text(L/2,W,H/2,"WALL-3")
ax.text(0,W/2,H/2,"WALL-4")

for name,(x,y,z) in P.items():
    ax.scatter(x,y,z,color="red",s=25)
    ax.text(x+0.05,y+0.05,z+0.05,name)

# 人体标注
ax.text(L/2, W/2, 1.85, "MAN-1 (1.75m)")


# ---------------- 视角 ----------------
ax.view_init(elev=80, azim=300)

ax.set_xlim(0,L)
ax.set_ylim(0,W)
ax.set_zlim(0,H)

ax.set_box_aspect([L,W,H])

plt.tight_layout()
plt.show()
```

```python
# 成年男子（坐在椅子上）
# =========================================================
def make_seated_man(table_info):

    x0, y0, tw, td = table_info

    faces = []

    # ---------------- 椅子位置 ----------------
    cw, cd, ch = 0.45, 0.45, 0.45

    chair_x = x0 + tw/2 - cw/2
    chair_y = y0 + td + 0.15

    # ---------------- 人体参数 ----------------
    hip_h = 0.45

    torso_h = 0.60
    torso_w = 0.32
    torso_d = 0.20

    thigh_h = 0.12
    thigh_l = 0.42

    calf_h = 0.43
    leg_w = 0.09

    arm_w = 0.07
    arm_d = 0.07
    arm_h = 0.38

    head_size = 0.22

    # ---------------- 人体中心 ----------------
    cx = chair_x + cw/2
    cy = chair_y + cd/2

    # =====================================================
    # 大腿（朝向桌子，即 -y）
    # =====================================================
    thigh_y = cy - thigh_l

    faces.extend(
        cuboid(
            cx - 0.11,
            thigh_y,
            hip_h - thigh_h,
            0.10,
            thigh_l,
            thigh_h
        )
    )

    faces.extend(
        cuboid(
            cx + 0.01,
            thigh_y,
            hip_h - thigh_h,
            0.10,
            thigh_l,
            thigh_h
        )
    )

    # =====================================================
    # 小腿
    # =====================================================
    shin_y = thigh_y - leg_w + 0.02

    faces.extend(
        cuboid(
            cx - 0.11,
            shin_y,
            0,
            leg_w,
            leg_w,
            calf_h
        )
    )

    faces.extend(
        cuboid(
            cx + 0.02,
            shin_y,
            0,
            leg_w,
            leg_w,
            calf_h
        )
    )

    # =====================================================
    # 躯干
    # =====================================================
    faces.extend(
        cuboid(
            cx - torso_w/2,
            cy - torso_d/2,
            hip_h,
            torso_w,
            torso_d,
            torso_h
        )
    )

    # =====================================================
    # 双臂
    # =====================================================
    arm_z = hip_h + 0.10

    faces.extend(
        cuboid(
            cx - torso_w/2 - arm_w,
            cy - arm_d/2,
            arm_z,
            arm_w,
            arm_d,
            arm_h
        )
    )

    faces.extend(
        cuboid(
            cx + torso_w/2,
            cy - arm_d/2,
```

```python
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 房间尺寸 ----------------
L, W, H = 4.2, 3.6, 3.0
t = 0.2

door_t = 0.02

    {"width": 2.1, "height": 1.8, "x": 1.05, "z": 0.75}

# ---------------- 门 ----------------

    {"x0": 0.2, "x1": 1.1, "z0": 0, "z1": 2.1}

    "P3": (0, W, 0),

    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

# ---------------- cuboid ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz
    return [
        [(x,y,z),(x1,y,z),(x1,y1,z),(x,y1,z)],
        [(x,y,z1),(x1,y,z1),(x1,y1,z1),(x,y1,z1)],
        [(x,y,z),(x1,y,z),(x1,y,z1),(x,y,z1)],
        [(x1,y,z),(x1,y1,z),(x1,y1,z1),(x1,y,z1)],
        [(x1,y1,z),(x,y1,z),(x,y1,z1),(x1,y1,z1)],
        [(x,y1,z),(x,y,z),(x,y,z1),(x,y1,z1)],
    ]

# ---------------- 墙体 ----------------
def cut_wall_south(L, t, H, windows):
    faces, cursor = [], 0
    for w in windows:
        wx0, wx1 = w["x"], w["x"] + w["width"]
        wz0, wz1 = w["z"], w["z"] + w["height"]

        if wx0 > cursor:
            faces.extend(cuboid(cursor, -t/2, 0, wx0-cursor, t, H))
        if wz0 > 0:
            faces.extend(cuboid(wx0, -t/2, 0, w["width"], t, wz0))
        if wz1 < H:
            faces.extend(cuboid(wx0, -t/2, wz1, w["width"], t, H-wz1))

        cursor = wx1

    if cursor < L:
        faces.extend(cuboid(cursor, -t/2, 0, L-cursor, t, H))

    return faces

def cut_wall_north(L, W, t, H, doors):
    faces = []
    for d in doors:
        dx0, dx1, dz1 = d["x0"], d["x1"], d["z1"]

        if dx0 > 0:
            faces.extend(cuboid(0, W-t/2, 0, dx0, t, H))
        if dx1 < L:
            faces.extend(cuboid(dx1, W-t/2, 0, L-dx1, t, H))
        if dz1 < H:
            faces.extend(cuboid(dx0, W-t/2, dz1, dx1-dx0, t, H-dz1))

    return faces

# ---------------- 门窗 ----------------
def make_door_solid(W, doors, door_t):
    faces = []
    for d in doors:
        faces.extend(cuboid(d["x0"], W-door_t/2, d["z0"],
                             d["x1"]-d["x0"], door_t, d["z1"]-d["z0"]))
    return faces

def make_window_glass(windows, glass_t):
    faces = []
    for w in windows:
        faces.extend(cuboid(w["x"], -glass_t/2, w["z"],
                             w["width"], glass_t, w["height"]))
    return faces

# ---------------- 🪞 镜子 ----------------

    mirror_w = 1.2

    mirror_t = 0.01

    z0 = 0.0
    x0 = t / 2

    faces.extend(cuboid(x0, y0, z0, mirror_t, mirror_w, mirror_h))
    return faces

# ---------------- 桌子 ----------------
def make_table():
    tw, td, th = table_size
    top_t = 0.05
    leg_h = th - top_t
    leg_w = 0.05

    cx = windows[0]["x"] + windows[0]["width"] / 2
    x0 = cx - tw / 2
    y0 = 0.10

    faces = []
    faces.extend(cuboid(x0, y0, leg_h, tw, td, top_t))

    legs = [
        (x0, y0),
        (x0 + tw - leg_w, y0),
        (x0, y0 + td - leg_w),
        (x0 + tw - leg_w, y0 + td - leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, leg_h))

    return faces, (x0, y0, tw, td)

# ---------------- 椅子 ----------------
def make_chair(table_info):
    x0, y0, tw, td = table_info

    cw, cd, ch = 0.45, 0.45, 0.45
    back_h = 0.45
    leg_w = 0.04

    cx = x0 + tw / 2 - cw / 2
    cy = y0 + td + 0.15

    faces = []
    faces.extend(cuboid(cx, cy, ch, cw, cd, 0.04))
    faces.extend(cuboid(cx, cy + cd - 0.04, ch, cw, 0.04, back_h))

    legs = [
        (cx, cy),
        (cx + cw - leg_w, cy),
        (cx, cy + cd - leg_w),
        (cx + cw - leg_w, cy + cd - leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, ch))

    return faces

# ---------------- 床 ----------------
def make_bed(dx=0.0, dy=0.0):
    bl, bw, bh = 1.8, 1.5, 0.45
    head_h, head_t = 0.9, 0.08
    leg_w, top_t = 0.06, 0.08
    leg_h = bh - top_t

    cx, cy = L / 2, W / 2
    x0, y0 = cx - bl / 2 + dx, cy - bw / 2 + dy

    faces = []
    faces.extend(cuboid(x0, y0, leg_h, bl, bw, top_t))

    legs = [
        (x0, y0),
        (x0 + bl - leg_w, y0),
        (x0, y0 + bw - leg_w),
        (x0 + bl - leg_w, y0 + bw - leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, leg_h))

    faces.extend(cuboid(x0 + bl - head_t, y0, bh, head_t, bw, head_h - bh))
    return faces

# ---------------- 成年男子（平躺，头部紧贴东墙） ----------------
def make_lying_man(dx=1.1, dy=0.95, shift_x=0.405):
    """
    人体平躺，头朝西（脚在东，头在西？需确认）：
    实际代码中头在东侧（X正向），为了让头贴东墙，整体向东平移 shift_x。
    此时头在最东侧，脸朝西。
    """
    faces = []

    bl, bw, bh = 1.8, 1.5, 0.45

    cx, cy = L / 2, W / 2
    bed_x0 = cx - bl / 2 + dx
    bed_y0 = cy - bw / 2 + dy

    center_x = bed_x0 + bl / 2
    center_y = bed_y0 + bw / 2

    body_z = bh + 0.36  # 人体抬高，避免被床遮挡

    # 人体参数
    torso_l = 0.75
    shoulder_w = 0.42
    torso_h = 0.22

    head_l = head_w = head_h = 0.22

    leg_l, leg_w, leg_h = 0.78, 0.12, 0.12
    arm_l, arm_w, arm_h = 0.65, 0.08, 0.08

    # 躯干
    tx = center_x - torso_l / 2
    ty = center_y - shoulder_w / 2
    faces.extend(cuboid(tx + shift_x, ty, body_z, torso_l, shoulder_w, torso_h))

    # 头（东侧）——平移后头部最东端 = tx + torso_l + shift_x + head_l
    hx = tx + torso_l
    faces.extend(cuboid(hx + shift_x, center_y - head_w / 2, body_z, head_l, head_w, head_h))

    # 腿（西侧）
    lx = tx - leg_l
    faces.extend(cuboid(lx + shift_x, center_y - 0.12, body_z, leg_l, leg_w, leg_h))
    faces.extend(cuboid(lx + shift_x, center_y, body_z, leg_l, leg_w, leg_h))

    # 手臂
    faces.extend(cuboid(tx + 0.05 + shift_x, ty - arm_w, body_z, arm_l, arm_w, arm_h))
    faces.extend(cuboid(tx + 0.05 + shift_x, ty + shoulder_w, body_z, arm_l, arm_w, arm_h))

    return faces
```
