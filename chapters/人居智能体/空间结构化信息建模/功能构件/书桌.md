# 书桌

> 注明:
>  
- 通过多个长方体体元（Cuboid）的组合（Composition），构建桌子构件；其中包括桌面（Table Top）与桌腿（Table Legs），实现“分部—组合”的构造表达。
- 桌子不涉及布尔减运算，而是通过几何拼装形成整体实体，属于典型的装配式构件（Assembly Component）。

#### 书桌构件的创建过程如下：

1. 定义桌子参数：设定整体尺寸（tw, td, th）及构造参数（桌面厚度 top_t、桌腿截面 leg_w、桌腿高度 leg_h = th − top_t）。
2. 确定桌子位置：以南墙窗户中心为基准沿 x 方向居中布置，并在 y 方向向室内偏移一定距离（如 0.10 m），得到桌子基点（x0, y0）。
3. 生成桌面：在 (x0, y0, leg_h) 处创建尺寸为（tw, td, top_t）的长方体，形成位于桌腿之上的桌面构件。
4. 生成桌腿：在桌面四角位置布置 4 个尺寸为（leg_w, leg_w, leg_h）的竖向长方体，形成支撑桌面的结构单元。
5. 构件组合：将桌面与四个桌腿体元进行几何并集运算（Union），形成完整书桌实体。
6. 语义标注：标注为“TABLE”，并建立空间关系——靠近 WALL-1（南墙）、位于窗下（Under Window）、沿窗中心对称布置。

### 俯视室内

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 房间尺寸 ----------------
L, W, H = 4.2, 3.6, 3.0
t = 0.2
glass_t = 0.01
door_t = 0.02  # 20mm门厚

# ---------------- 顶点（新增） ----------------
P = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

# ---------------- 窗户（WALL-1） ----------------
windows = [
    {"width": 2.1, "height": 1.8, "x": 1.05, "z": 0.75}
]

# ---------------- 门（WALL-3） ----------------
doors = [
    {
        "x0": 0.2,
        "x1": 0.2 + 0.9,
        "z0": 0,
        "z1": 2.1
    }
]

# ---------------- cuboid ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz
    return [
        [(x, y, z), (x1, y, z), (x1, y1, z), (x, y1, z)],
        [(x, y, z1), (x1, y, z1), (x1, y1, z1), (x, y1, z1)],
        [(x, y, z), (x1, y, z), (x1, y, z1), (x, y, z1)],
        [(x1, y, z), (x1, y1, z), (x1, y1, z1), (x1, y, z1)],
        [(x1, y1, z), (x, y1, z), (x, y1, z1), (x1, y1, z1)],
        [(x, y1, z), (x, y, z), (x, y, z1), (x, y1, z1)],
    ]

# ---------------- 南墙开窗 ----------------
def cut_wall_south(L, t, H, windows):
    faces = []
    cursor = 0

    for w in windows:
        wx0 = w["x"]
        wx1 = wx0 + w["width"]
        wz0 = w["z"]
        wz1 = wz0 + w["height"]

        if wx0 > cursor:
            faces.extend(cuboid(cursor, -t/2, 0, wx0 - cursor, t, H))

        if wz0 > 0:
            faces.extend(cuboid(wx0, -t/2, 0, w["width"], t, wz0))

        if wz1 < H:
            faces.extend(cuboid(wx0, -t/2, wz1, w["width"], t, H - wz1))

        cursor = wx1

    if cursor < L:
        faces.extend(cuboid(cursor, -t/2, 0, L - cursor, t, H))

    return faces

# ---------------- 北墙开门（挖空） ----------------
def cut_wall_north(L, W, t, H, doors):
    faces = []

    for d in doors:
        dx0 = d["x0"]
        dx1 = d["x1"]
        dz1 = d["z1"]

        if dx0 > 0:
            faces.extend(cuboid(0, W - t/2, 0, dx0, t, H))

        if dx1 < L:
            faces.extend(cuboid(dx1, W - t/2, 0, L - dx1, t, H))

        if dz1 < H:
            faces.extend(cuboid(dx0, W - t/2, dz1, dx1 - dx0, t, H - dz1))

    return faces

# ---------------- 门实体 ----------------
def make_door_solid(W, doors, door_t):
    faces = []
    for d in doors:
        faces.extend(
            cuboid(
                d["x0"],
                W - door_t/2,
                d["z0"],
                d["x1"] - d["x0"],
                door_t,
                d["z1"] - d["z0"]
            )
        )
    return faces

# ---------------- 玻璃 ----------------
def make_window_glass(windows, glass_t):
    faces = []
    for w in windows:
        faces.extend(
            cuboid(
                w["x"],
                -glass_t/2,
                w["z"],
                w["width"],
                glass_t,
                w["height"]
            )
        )
    return faces

# ---------------- flatten ----------------
def flatten(faces):
    out = []
    for f in faces:
        arr = np.array(f)
        if arr.ndim == 2:
            out.append(arr)
    return out

# ---------------- add ----------------
def add(ax, faces, color):
    poly = Poly3DCollection(
        faces,
        facecolors=color,
        edgecolors="black",
        linewidths=0.3,
        alpha=color[3]
    )
    ax.add_collection3d(poly)

# ---------------- plot ----------------
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')

wall_faces = []
wall_faces.extend(cut_wall_south(L, t, H, windows))
wall_faces.extend(cuboid(L - t/2, 0, 0, t, W, H))
wall_faces.extend(cut_wall_north(L, W, t, H, doors))
wall_faces.extend(cuboid(-t/2, 0, 0, t, W, H))

add(ax, flatten(wall_faces), (0.2, 0.6, 1.0, 0.25))
add(ax, flatten(cuboid(0, 0, -t, L, W, t)), (0.7, 0.7, 0.7, 1.0))
add(ax, flatten(make_window_glass(windows, glass_t)), (0.3, 0.8, 1.0, 0.15))
add(ax, flatten(make_door_solid(W, doors, door_t)), (1.0, 1.0, 1.0, 1.0))

# ---------------- 墙体标注 ----------------
ax.text(L/2, 0, H/2, "WALL-1 (S)")
ax.text(L, W/2, H/2, "WALL-2 (E)")
ax.text(L/2, W, H/2, "WALL-3 (N - DOOR)")
ax.text(0, W/2, H/2, "WALL-4 (W)")

# ---------------- 顶点显示（新增） ----------------
for name, (x, y, z) in P.items():
    ax.scatter(x, y, z, color="red", s=25)
    ax.text(x + 0.05, y + 0.05, z + 0.05, name)

# ---------------- 视角 ----------------
ax.view_init(elev=75, azim=235)

ax.set_xlim(0, L)
ax.set_ylim(0, W)
ax.set_zlim(0, H)
ax.set_box_aspect([L, W, H])

plt.tight_layout()
plt.show()
```

### 把立方体桌子变成真实家具结构：桌面和四个桌腿

def make_table():

    top_t = 0.05
    leg_h = th - top_t
    leg_w = 0.05

    cx = 1.05 + 2.1 / 2

        (x0, y0),
        (x0 + tw - leg_w, y0),
        (x0, y0 + td - leg_w),
        (x0 + tw - leg_w, y0 + td - leg_w)
