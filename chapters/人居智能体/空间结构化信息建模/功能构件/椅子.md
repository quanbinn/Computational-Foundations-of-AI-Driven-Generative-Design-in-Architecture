# 椅子

> 注明:
>  
- 通过多个长方体体元（Cuboid）的组合（Composition），构建椅子构件；包括座面（Seat）、靠背（Backrest）与椅腿（Legs），实现“分部—组合”的构造表达。
- 椅子的位置由书桌构件驱动生成，属于依附于桌子的从属家具构件（Dependent Furniture Component），体现“家具—家具”的空间约束关系。

#### 椅子构件的创建过程如下：
1. 定义椅子参数：设定座面尺寸（cw, cd, ch）、靠背高度（back_h）及椅腿截面尺寸（leg_w），用于控制整体比例与构造关系。
2. 确定椅子位置：以书桌为基准，沿 x 方向与桌面中心对齐，并在 y 方向位于桌子外侧一定距离（如 td + 0.15 m），得到椅子基点（cx, cy）。
3. 生成座面：在 (cx, cy, ch) 处创建厚度较小的长方体，作为承坐构件（Seat）。
4. 生成靠背：在座面后侧沿 y 方向生成竖向长方体（高度 back_h），形成背部支撑构件（Backrest）。
5. 生成椅腿：在座面四角布置 4 个竖向长方体（高度 ch），形成支撑体系（Legs）。
6. 构件组合：将座面、靠背与椅腿体元进行几何并集运算（Union），形成完整椅子实体。
7. 语义标注：标注为“CHAIR”，并建立空间关系——位于书桌前方（Facing Table）、与书桌中心对齐（Center Alignment）、相对于 WALL-1 呈内向布置。

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 房间尺寸 ----------------
L, W, H = 4.2, 3.6, 3.0
t = 0.2
glass_t = 0.01
door_t = 0.02

# ---------------- 窗户 ----------------
windows = [
    {"width": 2.1, "height": 1.8, "x": 1.05, "z": 0.75}
]

# ---------------- 门 ----------------
doors = [
    {"x0": 0.2, "x1": 1.1, "z0": 0, "z1": 2.1}
]

# ---------------- 桌子尺寸 ----------------
table_size = (1.2, 0.6, 0.75)

# ---------------- 8个拓扑顶点 ----------------
P = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

# ---------------- cuboid ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz
    return [
        [(x,y,z),(x1,y,z),(x1,y1,z),(x,y1,z)],
        [(x,y,z1),(x1,y,z1),(x1,y1,z1),(x,y1,z1)],
        [(x,y,z),(x1,y,z),(x1,y,z1),(x,y,z1)],
        [(x1,y,z),(x1,y1,z),(x1,y1,z1),(x1,y,z1)],
        [(x1,y1,z),(x,y1,z),(x,y1,z1),(x1,y1,z1)],
        [(x,y1,z),(x,y,z),(x,y,z1),(x,y1,z1)],
    ]

# ---------------- 墙体 ----------------
def cut_wall_south(L, t, H, windows):
    faces = []
    cursor = 0
    for w in windows:
        wx0, wx1 = w["x"], w["x"] + w["width"]
        wz0, wz1 = w["z"], w["z"] + w["height"]

        if wx0 > cursor:
            faces.extend(cuboid(cursor, -t/2, 0, wx0-cursor, t, H))
        if wz0 > 0:
            faces.extend(cuboid(wx0, -t/2, 0, w["width"], t, wz0))
        if wz1 < H:
            faces.extend(cuboid(wx0, -t/2, wz1, w["width"], t, H-wz1))

        cursor = wx1

    if cursor < L:
        faces.extend(cuboid(cursor, -t/2, 0, L-cursor, t, H))

    return faces

def cut_wall_north(L, W, t, H, doors):
    faces = []
    for d in doors:
        dx0, dx1, dz1 = d["x0"], d["x1"], d["z1"]

        if dx0 > 0:
            faces.extend(cuboid(0, W-t/2, 0, dx0, t, H))
        if dx1 < L:
            faces.extend(cuboid(dx1, W-t/2, 0, L-dx1, t, H))
        if dz1 < H:
            faces.extend(cuboid(dx0, W-t/2, dz1, dx1-dx0, t, H-dz1))

    return faces

# ---------------- 门窗 ----------------
def make_door_solid(W, doors, door_t):
    faces = []
    for d in doors:
        faces.extend(cuboid(d["x0"], W-door_t/2, d["z0"],
                             d["x1"]-d["x0"], door_t, d["z1"]-d["z0"]))
    return faces

def make_window_glass(windows, glass_t):
    faces = []
    for w in windows:
        faces.extend(cuboid(w["x"], -glass_t/2, w["z"],
                             w["width"], glass_t, w["height"]))
    return faces

# ---------------- 桌子 ----------------
def make_table():
    tw, td, th = table_size
    top_t = 0.05
    leg_h = th - top_t
    leg_w = 0.05

    window_center_x = 1.05 + 2.1/2
    x0 = window_center_x - tw/2
    y0 = 0.10

    faces = []

    faces.extend(cuboid(x0, y0, leg_h, tw, td, top_t))

    legs = [
        (x0, y0),
        (x0+tw-leg_w, y0),
        (x0, y0+td-leg_w),
        (x0+tw-leg_w, y0+td-leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, leg_h))

    return faces, (x0, y0, tw, td)

# ---------------- 椅子 ----------------
def make_chair(table_info):
    x0, y0, tw, td = table_info

    cw, cd, ch = 0.45, 0.45, 0.45
    back_h = 0.45
    leg_w = 0.04

    cx = x0 + tw/2 - cw/2
    cy = y0 + td + 0.15

    faces = []

    faces.extend(cuboid(cx, cy, ch, cw, cd, 0.04))
    faces.extend(cuboid(cx, cy+cd-0.04, ch, cw, 0.04, back_h))

    legs = [
        (cx, cy),
        (cx+cw-leg_w, cy),
        (cx, cy+cd-leg_w),
        (cx+cw-leg_w, cy+cd-leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, ch))

    return faces

# ---------------- 工具 ----------------
def flatten(faces):
    out = []
    for f in faces:
        arr = np.array(f)
        if arr.ndim == 2:
            out.append(arr)
    return out

def add(ax, faces, color):
    poly = Poly3DCollection(faces, facecolors=color,
                            edgecolors="black", linewidths=0.3,
                            alpha=color[3])
    ax.add_collection3d(poly)

# ---------------- 场景 ----------------
fig = plt.figure(figsize=(10,7))
ax = fig.add_subplot(111, projection='3d')

wall_color = (0.2,0.6,1,0.25)
floor_color = (0.7,0.7,0.7,1)
glass_color = (0.3,0.8,1,0.15)
door_color = (1,1,1,1)
table_color = (0.55,0.35,0.2,1)
chair_color = (0.4,0.25,0.15,1)

# 房间
wall_faces = []
wall_faces.extend(cut_wall_south(L,t,H,windows))
wall_faces.extend(cuboid(L-t/2,0,0,t,W,H))
wall_faces.extend(cut_wall_north(L,W,t,H,doors))
wall_faces.extend(cuboid(-t/2,0,0,t,W,H))

floor_faces = cuboid(0,0,-t,L,W,t)
glass_faces = make_window_glass(windows, glass_t)
door_faces = make_door_solid(W, doors, door_t)

# 家具
table_faces, table_info = make_table()
chair_faces = make_chair(table_info)

# 渲染
add(ax, flatten(wall_faces), wall_color)
add(ax, flatten(floor_faces), floor_color)
add(ax, flatten(glass_faces), glass_color)
add(ax, flatten(door_faces), door_color)
add(ax, flatten(table_faces), table_color)
add(ax, flatten(chair_faces), chair_color)

# ---------------- 墙体标注 ----------------
ax.text(L/2, 0, H/2, "WALL-1 (S)")
ax.text(L, W/2, H/2, "WALL-2 (E)")
ax.text(L/2, W, H/2, "WALL-3 (N)")
ax.text(0, W/2, H/2, "WALL-4 (W)")

# ---------------- 顶点 ----------------
for name, (x, y, z) in P.items():
    ax.scatter(x, y, z, color="red", s=25)
    ax.text(x + 0.05, y + 0.05, z + 0.05, name)

# ---------------- 视角 ----------------
ax.view_init(elev=75, azim=235)
ax.set_xlim(0,L)
ax.set_ylim(0,W)
ax.set_zlim(0,H)
ax.set_box_aspect([L,W,H])

plt.tight_layout()
plt.show()
```
