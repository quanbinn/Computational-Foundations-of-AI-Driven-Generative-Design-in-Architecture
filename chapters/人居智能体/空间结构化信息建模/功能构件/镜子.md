# 镜子

> 注明:
>  
- 镜子构件通过单一长方体体元（Cuboid）直接生成，属于典型的薄片式墙附着构件（Wall-attached Thin Primitive），用于表达室内垂直界面的反射性功能单元。其构造方式不依赖多部件组合，而采用“整体几何 + 参数化定位”的轻量表达策略。
- 镜子构件的空间定位依附于 wall-4（x=0）内侧表面，通过“墙体厚度偏移 + 居中约束”的方式实现精确布置，体现典型的界面附着型构件生成机制（Surface-constrained Primitive Placement）。

#### 镜子构件的创建过程如下：

1. 定义镜子参数：设定镜子宽度（1.2 m）、高度（2.4 m）和厚度（0.01 m），其中宽度沿 y 方向，高度沿 z 方向，厚度沿 x 方向。
2. 确定镜子位置：以 wall-4（x = 0）为基准，在墙体内侧通过 x = t/2 贴墙定位，并在 y 方向居中（y = (W − 1.2)/2），在 z 方向从地面（z = 0）起布置。
3. 生成镜面主体：在确定位置处生成一个长方体体元，形成尺寸为 0.01 × 1.2 × 2.4 的薄板结构。
4. 构件表达：镜子作为单一几何体存在，不进行拆分或组合，仅作为独立附着构件表达。
5. 构件关系：镜子依附于 wall-4 内侧表面，居中布置并落地设置，不改变墙体结构。
6. 语义标注：标注为“MIRROR”，空间关系为附着于 wall-4、居中、落地、朝向室内。

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 房间尺寸 ----------------
L, W, H = 4.2, 3.6, 3.0
t = 0.2
glass_t = 0.01
door_t = 0.02

# ---------------- 窗户 ----------------
windows = [
    {"width": 2.1, "height": 1.8, "x": 1.05, "z": 0.75}
]

# ---------------- 门 ----------------
doors = [
    {"x0": 0.2, "x1": 1.1, "z0": 0, "z1": 2.1}
]

# ---------------- 桌子 ----------------
table_size = (1.2, 0.6, 0.75)

# ---------------- 顶点 ----------------
P = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

# ---------------- cuboid ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz
    return [
        [(x,y,z),(x1,y,z),(x1,y1,z),(x,y1,z)],
        [(x,y,z1),(x1,y,z1),(x1,y1,z1),(x,y1,z1)],
        [(x,y,z),(x1,y,z),(x1,y,z1),(x,y,z1)],
        [(x1,y,z),(x1,y1,z),(x1,y1,z1),(x1,y,z1)],
        [(x1,y1,z),(x,y1,z),(x,y1,z1),(x1,y1,z1)],
        [(x,y1,z),(x,y,z),(x,y,z1),(x,y1,z1)],
    ]

# ---------------- 墙体 ----------------
def cut_wall_south(L, t, H, windows):
    faces, cursor = [], 0
    for w in windows:
        wx0, wx1 = w["x"], w["x"] + w["width"]
        wz0, wz1 = w["z"], w["z"] + w["height"]

        if wx0 > cursor:
            faces.extend(cuboid(cursor, -t/2, 0, wx0-cursor, t, H))
        if wz0 > 0:
            faces.extend(cuboid(wx0, -t/2, 0, w["width"], t, wz0))
        if wz1 < H:
            faces.extend(cuboid(wx0, -t/2, wz1, w["width"], t, H-wz1))

        cursor = wx1

    if cursor < L:
        faces.extend(cuboid(cursor, -t/2, 0, L-cursor, t, H))

    return faces

def cut_wall_north(L, W, t, H, doors):
    faces = []
    for d in doors:
        dx0, dx1, dz1 = d["x0"], d["x1"], d["z1"]

        if dx0 > 0:
            faces.extend(cuboid(0, W-t/2, 0, dx0, t, H))
        if dx1 < L:
            faces.extend(cuboid(dx1, W-t/2, 0, L-dx1, t, H))
        if dz1 < H:
            faces.extend(cuboid(dx0, W-t/2, dz1, dx1-dx0, t, H-dz1))

    return faces

# ---------------- 门窗 ----------------
def make_door_solid(W, doors, door_t):
    faces = []
    for d in doors:
        faces.extend(cuboid(d["x0"], W-door_t/2, d["z0"],
                             d["x1"]-d["x0"], door_t, d["z1"]-d["z0"]))
    return faces

def make_window_glass(windows, glass_t):
    faces = []
    for w in windows:
        faces.extend(cuboid(w["x"], -glass_t/2, w["z"],
                             w["width"], glass_t, w["height"]))
    return faces

# ---------------- 🪞 镜子（wall-4） ----------------
def make_mirror():
    mirror_w = 1.2   # y方向宽度
    mirror_h = 2.4   # z方向高度
    mirror_t = 0.01  # x方向厚度

    y0 = (W - mirror_w) / 2   # 居中
    z0 = 0.0                  # 落地
    x0 = t/2                  # 贴 wall-4 内侧

    faces = []
    faces.extend(cuboid(x0, y0, z0, mirror_t, mirror_w, mirror_h))
    return faces

# ---------------- 桌子 ----------------
def make_table():
    tw, td, th = table_size
    top_t = 0.05
    leg_h = th - top_t
    leg_w = 0.05

    cx = windows[0]["x"] + windows[0]["width"]/2
    x0 = cx - tw/2
    y0 = 0.10

    faces = []
    faces.extend(cuboid(x0, y0, leg_h, tw, td, top_t))

    legs = [
        (x0, y0),
        (x0+tw-leg_w, y0),
        (x0, y0+td-leg_w),
        (x0+tw-leg_w, y0+td-leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, leg_h))

    return faces, (x0, y0, tw, td)

# ---------------- 椅子 ----------------
def make_chair(table_info):
    x0, y0, tw, td = table_info

    cw, cd, ch = 0.45, 0.45, 0.45
    back_h = 0.45
    leg_w = 0.04

    cx = x0 + tw/2 - cw/2
    cy = y0 + td + 0.15

    faces = []
    faces.extend(cuboid(cx, cy, ch, cw, cd, 0.04))
    faces.extend(cuboid(cx, cy+cd-0.04, ch, cw, 0.04, back_h))

    legs = [
        (cx, cy),
        (cx+cw-leg_w, cy),
        (cx, cy+cd-leg_w),
        (cx+cw-leg_w, cy+cd-leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, ch))

    return faces

# ---------------- 床 ----------------
def make_bed(dx=0.0, dy=0.0):
    bl, bw, bh = 1.8, 1.5, 0.45
    head_h, head_t = 0.9, 0.08

    leg_w = 0.06
    top_t = 0.08
    leg_h = bh - top_t

    cx, cy = L/2, W/2
    x0, y0 = cx - bl/2 + dx, cy - bw/2 + dy

    faces = []

    faces.extend(cuboid(x0, y0, leg_h, bl, bw, top_t))

    legs = [
        (x0, y0),
        (x0+bl-leg_w, y0),
        (x0, y0+bw-leg_w),
        (x0+bl-leg_w, y0+bw-leg_w)
    ]

    for lx, ly in legs:
        faces.extend(cuboid(lx, ly, 0, leg_w, leg_w, leg_h))

    faces.extend(cuboid(
        x0 + bl - head_t,
        y0,
        bh,
        head_t,
        bw,
        head_h - bh
    ))

    return faces

# ---------------- flatten ----------------
def flatten(faces):
    return [np.array(f) for f in faces if np.array(f).ndim == 2]

# ---------------- add ----------------
def add(ax, faces, color):
    ax.add_collection3d(Poly3DCollection(
        faces, facecolors=color,
        edgecolors="black", linewidths=0.3,
        alpha=color[3]
    ))

# ---------------- 场景 ----------------
fig = plt.figure(figsize=(10,7))
ax = fig.add_subplot(111, projection='3d')

wall_faces = []
wall_faces.extend(cut_wall_south(L,t,H,windows))
wall_faces.extend(cuboid(L-t/2,0,0,t,W,H))
wall_faces.extend(cut_wall_north(L,W,t,H,doors))
wall_faces.extend(cuboid(-t/2,0,0,t,W,H))

table_faces, table_info = make_table()
chair_faces = make_chair(table_info)
bed_faces = make_bed(dx=1.1, dy=0.95)
mirror_faces = make_mirror()

# ---------------- 渲染 ----------------
add(ax, flatten(wall_faces), (0.2,0.6,1,0.25))
add(ax, flatten(cuboid(0,0,-t,L,W,t)), (0.7,0.7,0.7,1))
add(ax, flatten(make_window_glass(windows, glass_t)), (0.3,0.8,1,0.15))
add(ax, flatten(make_door_solid(W, doors, door_t)), (1,1,1,1))
add(ax, flatten(table_faces), (0.55,0.35,0.2,1))
add(ax, flatten(chair_faces), (0.4,0.25,0.15,1))
add(ax, flatten(bed_faces), (0.8,0.7,0.6,1))
add(ax, flatten(mirror_faces), (0.95,0.95,1,0.6))

# ---------------- 标注 ----------------
ax.text(L/2,0,H/2,"WALL-1")
ax.text(L,W/2,H/2,"WALL-2")
ax.text(L/2,W,H/2,"WALL-3")
ax.text(0,W/2,H/2,"WALL-4")

for name,(x,y,z) in P.items():
    ax.scatter(x,y,z,color="red",s=25)
    ax.text(x+0.05,y+0.05,z+0.05,name)

ax.view_init(elev=80, azim=300)
ax.set_xlim(0,L)
ax.set_ylim(0,W)
ax.set_zlim(0,H)
ax.set_box_aspect([L,W,H])

plt.tight_layout()
plt.show()
```
