# 墙体

### 创建一个长宽高分别为：4.2m, 3.6m, 和3.0m的立方体房间 

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection

# ---------------- 房间尺寸 ----------------
L = 4.2   # X方向长度
W = 3.6   # Y方向宽度
H = 3.0   # Z方向高度

# ---------------- 定义8个顶点 ----------------
p0 = (0, 0, 0)
p1 = (L, 0, 0)
p2 = (L, W, 0)
p3 = (0, W, 0)

p4 = (0, 0, H)
p5 = (L, 0, H)
p6 = (L, W, H)
p7 = (0, W, H)

# ---------------- 定义6个面 ----------------
floor = [p0, p1, p2, p3]
ceiling = [p4, p5, p6, p7]

wall1 = [p0, p1, p5, p4]
wall2 = [p1, p2, p6, p5]
wall3 = [p2, p3, p7, p6]
wall4 = [p3, p0, p4, p7]

faces = [floor, ceiling, wall1, wall2, wall3, wall4]

# ---------------- 绘图 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

# ---------------- 面颜色设置 ----------------
face_colors = [
    (0.8, 0.8, 0.8, 1.0),  # 地面（不透明）

    # 顶面（改为透明）
    (0.9, 0.9, 0.9, 0.15),

    # 四面墙（半透明）
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
]

for face, color in zip(faces, face_colors):
    poly = Poly3DCollection([face], facecolors=color, edgecolors='k', linewidths=0.8)
    ax.add_collection3d(poly)

# ---------------- 坐标轴 ----------------
ax.set_xlim(0, L)
ax.set_ylim(0, W)
ax.set_zlim(0, H)

ax.set_xlabel("X (4.2m)")
ax.set_ylabel("Y (3.6m)")
ax.set_zlabel("Z (3.0m)")

ax.set_box_aspect([L, W, H])

# ---------------- 视角 ----------------
ax.view_init(elev=20, azim=235)

plt.title("3D Room (Transparent Ceiling)")
plt.show()
```

### 在立方体房间上显示出所有的墙（WALL-1~4）
```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 房间尺寸 ----------------
L = 4.2
W = 3.6
H = 3.0

# ---------------- 8个顶点 ----------------
points = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

p0, p1, p2, p3 = points["P0"], points["P1"], points["P2"], points["P3"]
p4, p5, p6, p7 = points["P4"], points["P5"], points["P6"], points["P7"]

# ---------------- 六个面 ----------------
floor = [p0, p1, p2, p3]
ceiling = [p4, p5, p6, p7]

wall1 = [p0, p1, p5, p4]  # 前墙
wall2 = [p1, p2, p6, p5]  # 右墙
wall3 = [p2, p3, p7, p6]  # 后墙
wall4 = [p3, p0, p4, p7]  # 左墙

faces = [floor, ceiling, wall1, wall2, wall3, wall4]

face_names = ["FLOOR", "CEILING", "WALL-1", "WALL-2", "WALL-3", "WALL-4"]

# ---------------- 绘图 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

# ---------------- 面颜色 ----------------
face_colors = [
    (0.8, 0.8, 0.8, 1.0),   # 地面
    (0.9, 0.9, 0.9, 0.15),  # 顶面透明
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
]

# ---------------- 画面 ----------------
for face, color in zip(faces, face_colors):
    poly = Poly3DCollection([face], facecolors=color, edgecolors='k', linewidths=0.8)
    ax.add_collection3d(poly)

# ---------------- 墙面编号 ----------------
for i, face in enumerate(faces):
    centroid = np.mean(np.array(face), axis=0)
    ax.text(
        centroid[0], centroid[1], centroid[2],
        face_names[i],
        color="black",
        fontsize=11,
        ha="center"
    )

# ---------------- 顶点编号 ----------------
for name, (x, y, z) in points.items():
    ax.scatter(x, y, z, color='red', s=25)
    ax.text(
        x + 0.05,
        y + 0.05,
        z + 0.05,
        name,
        color="red",
        fontsize=10
    )

# ---------------- 坐标轴 ----------------
ax.set_xlim(0, L)
ax.set_ylim(0, W)
ax.set_zlim(0, H)

ax.set_xlabel("X (4.2m)")
ax.set_ylabel("Y (3.6m)")
ax.set_zlabel("Z (3.0m)")

ax.set_box_aspect([L, W, H])
ax.view_init(elev=20, azim=235)

plt.title("3D Room with Vertex + Wall Labels")
plt.show()
```

### 把墙体和楼板均采用立方体进行三维建模表达

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection

# ---------------- 参数 ----------------
L = 4.2
W = 3.6
H = 3.0
t = 0.2  # 200mm

# ---------------- 立方体生成 ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz

    return [
        [(x, y, z), (x1, y, z), (x1, y1, z), (x, y1, z)],
        [(x, y, z1), (x1, y, z1), (x1, y1, z1), (x, y1, z1)],
        [(x, y, z), (x1, y, z), (x1, y, z1), (x, y, z1)],
        [(x1, y, z), (x1, y1, z), (x1, y1, z1), (x1, y, z1)],
        [(x1, y1, z), (x, y1, z), (x, y1, z1), (x1, y1, z1)],
        [(x, y1, z), (x, y, z), (x, y, z1), (x, y1, z1)],
    ]


# ---------------- 添加几何 ----------------
def add(ax, faces, color):
    poly = Poly3DCollection(
        faces,
        facecolors=color,
        edgecolors="black",
        linewidths=0.4,
        alpha=color[3]
    )
    ax.add_collection3d(poly)

# ---------------- 画布 ----------------
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')

wall_color = (0.2, 0.6, 1.0, 0.25)
floor_color = (0.7, 0.7, 0.7, 1.0)

# ---------------- 墙体（200mm厚） ----------------
walls = [
    cuboid(0, W - t/2, 0, L, t, H),     # 上墙
    cuboid(0, -t/2, 0, L, t, H),        # 下墙
    cuboid(-t/2, 0, 0, t, W, H),        # 左墙
    cuboid(L - t/2, 0, 0, t, W, H),     # 右墙
]

# ---------------- 楼板（0 ~ -0.2m） ----------------
floor = cuboid(0, 0, -t, L, W, t)

# ---------------- 绘制墙体 ----------------
for w in walls:
    add(ax, w, wall_color)

# ---------------- 绘制楼板 ----------------
add(ax, floor, floor_color)

# ---------------- 坐标系统 ----------------
ax.set_xlim(-0.5, L + 0.5)
ax.set_ylim(-0.5, W + 0.5)
ax.set_zlim(-0.3, H)

ax.set_box_aspect([L, W, H])

ax.set_xlabel("X (4.2m)")
ax.set_ylabel("Y (3.6m)")
ax.set_zlabel("Z (3.0m)")

ax.view_init(elev=20, azim=235)

plt.title("BIM Room (200mm Walls + Slab)")
plt.tight_layout()
plt.show()
```

### 在立方体房间上显示出所有的墙（WALL-1~4）
```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 参数 ----------------
L = 4.2
W = 3.6
H = 3.0
t = 0.2  # 墙厚 200mm

# ---------------- 立方体生成 ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz
    return [
        [(x, y, z), (x1, y, z), (x1, y1, z), (x, y1, z)],  # bottom
        [(x, y, z1), (x1, y, z1), (x1, y1, z1), (x, y1, z1)],  # top
        [(x, y, z), (x1, y, z), (x1, y, z1), (x, y, z1)],  # front
        [(x1, y, z), (x1, y1, z), (x1, y1, z1), (x1, y, z1)],  # right
        [(x1, y1, z), (x, y1, z), (x, y1, z1), (x1, y1, z1)],  # back
        [(x, y1, z), (x, y, z), (x, y, z1), (x, y1, z1)],  # left
    ]

# ---------------- 添加几何 ----------------
def add(ax, faces, color):
    poly = Poly3DCollection(
        faces,
        facecolors=color,
        edgecolors="black",
        linewidths=0.4,
        alpha=color[3]
    )
    ax.add_collection3d(poly)

# ---------------- 画布 ----------------
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')

wall_color = (0.2, 0.6, 1.0, 0.25)
floor_color = (0.7, 0.7, 0.7, 1.0)

# =========================================================
# ---------------- 建筑拓扑顺序（核心） ----------------
# 底面：P0 → P1 → P2 → P3（逆时针）
# 墙体：南 → 东 → 北 → 西（闭合环）
# =========================================================

# ---------------- 墙体 ----------------
walls = [
    cuboid(0, -t/2, 0, L, t, H),      # WALL-1 南 (P0→P1)
    cuboid(L - t/2, 0, 0, t, W, H),   # WALL-2 东 (P1→P2)
    cuboid(0, W - t/2, 0, L, t, H),   # WALL-3 北 (P2→P3)
    cuboid(-t/2, 0, 0, t, W, H),      # WALL-4 西 (P3→P0)
]

# ---------------- 墙体标注 ----------------
wall_labels = [
    ("WALL-1 (S)", L/2, 0, H/2),
    ("WALL-2 (E)", L, W/2, H/2),
    ("WALL-3 (N)", L/2, W, H/2),
    ("WALL-4 (W)", 0, W/2, H/2),
]

# ---------------- 楼板 ----------------
floor = cuboid(0, 0, -t, L, W, t)

# ---------------- 绘制墙体 ----------------
for w in walls:
    add(ax, w, wall_color)

# ---------------- 绘制楼板 ----------------
add(ax, floor, floor_color)

# ---------------- 墙体编号 ----------------
for name, x, y, z in wall_labels:
    ax.text(x, y, z, name, fontsize=11, color="black", weight="bold")

# ---------------- 8个标准点（净空间） ----------------
points = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

for name, (x, y, z) in points.items():
    ax.scatter(x, y, z, color="red", s=25)
    ax.text(x + 0.05, y + 0.05, z + 0.05, name, fontsize=9)

# ---------------- 坐标系统 ----------------
ax.set_xlim(-0.5, L + 0.5)
ax.set_ylim(-0.5, W + 0.5)
ax.set_zlim(-0.3, H)

ax.set_box_aspect([L, W, H])

ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")

# ---------------- 视角 ----------------
ax.view_init(elev=20, azim=235)

plt.title("BIM Room (Topological Walls + Slab + Labels)")
plt.tight_layout()
plt.show()
```
