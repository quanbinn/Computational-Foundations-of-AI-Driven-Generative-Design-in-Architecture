# 墙体

### 创建一个长宽高分别为：4.2m, 3.6m, 和3.0m的立方体房间 

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection

# ---------------- 房间尺寸 ----------------
L = 4.2   # X方向长度
W = 3.6   # Y方向宽度
H = 3.0   # Z方向高度

# ---------------- 定义8个顶点 ----------------
p0 = (0, 0, 0)
p1 = (L, 0, 0)
p2 = (L, W, 0)
p3 = (0, W, 0)

p4 = (0, 0, H)
p5 = (L, 0, H)
p6 = (L, W, H)
p7 = (0, W, H)

# ---------------- 定义6个面 ----------------
floor = [p0, p1, p2, p3]
ceiling = [p4, p5, p6, p7]

wall1 = [p0, p1, p5, p4]
wall2 = [p1, p2, p6, p5]
wall3 = [p2, p3, p7, p6]
wall4 = [p3, p0, p4, p7]

faces = [floor, ceiling, wall1, wall2, wall3, wall4]

# ---------------- 绘图 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

# ---------------- 面颜色设置 ----------------
face_colors = [
    (0.8, 0.8, 0.8, 1.0),  # 地面（不透明）

    # 顶面（改为透明）
    (0.9, 0.9, 0.9, 0.15),

    # 四面墙（半透明）
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
]

for face, color in zip(faces, face_colors):
    poly = Poly3DCollection([face], facecolors=color, edgecolors='k', linewidths=0.8)
    ax.add_collection3d(poly)

# ---------------- 坐标轴 ----------------
ax.set_xlim(0, L)
ax.set_ylim(0, W)
ax.set_zlim(0, H)

ax.set_xlabel("X (4.2m)")
ax.set_ylabel("Y (3.6m)")
ax.set_zlabel("Z (3.0m)")

ax.set_box_aspect([L, W, H])

# ---------------- 视角 ----------------
ax.view_init(elev=20, azim=235)

plt.title("3D Room (Transparent Ceiling)")
plt.show()
```

### 在立方体房间上显示出所有的点（P0–P7）
```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection

# ---------------- 房间尺寸 ----------------
L = 4.2
W = 3.6
H = 3.0

# ---------------- 8个顶点 ----------------
points = {
    "p0": (0, 0, 0),
    "p1": (L, 0, 0),
    "p2": (L, W, 0),
    "p3": (0, W, 0),
    "p4": (0, 0, H),
    "p5": (L, 0, H),
    "p6": (L, W, H),
    "p7": (0, W, H),
}

p0, p1, p2, p3 = points["p0"], points["p1"], points["p2"], points["p3"]
p4, p5, p6, p7 = points["p4"], points["p5"], points["p6"], points["p7"]

# ---------------- 六个面 ----------------
floor = [p0, p1, p2, p3]
ceiling = [p4, p5, p6, p7]

wall1 = [p0, p1, p5, p4]
wall2 = [p1, p2, p6, p5]
wall3 = [p2, p3, p7, p6]
wall4 = [p3, p0, p4, p7]

faces = [floor, ceiling, wall1, wall2, wall3, wall4]

# ---------------- 绘图 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

# ---------------- 面颜色 ----------------
face_colors = [
    (0.8, 0.8, 0.8, 1.0),
    (0.9, 0.9, 0.9, 0.15),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
]

for face, color in zip(faces, face_colors):
    poly = Poly3DCollection([face], facecolors=color, edgecolors='k', linewidths=0.8)
    ax.add_collection3d(poly)

# ---------------- 标注点 ----------------
for name, (x, y, z) in points.items():
    # 点
    ax.scatter(x, y, z, color='red', s=30)

    # 文本标签（稍微偏移避免遮挡）
    ax.text(x + 0.05, y + 0.05, z + 0.05, name, color='black', fontsize=9)

# ---------------- 坐标轴 ----------------
ax.set_xlim(0, L)
ax.set_ylim(0, W)
ax.set_zlim(0, H)

ax.set_xlabel("X (4.2m)")
ax.set_ylabel("Y (3.6m)")
ax.set_zlabel("Z (3.0m)")

ax.set_box_aspect([L, W, H])

ax.view_init(elev=20, azim=235)

plt.title("3D Room with Labeled Vertices")
plt.show()
```

### 在立方体房间上显示出所有的墙（WALL-1~4）
```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 房间尺寸 ----------------
L = 4.2
W = 3.6
H = 3.0

# ---------------- 8个顶点 ----------------
points = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

p0, p1, p2, p3 = points["P0"], points["P1"], points["P2"], points["P3"]
p4, p5, p6, p7 = points["P4"], points["P5"], points["P6"], points["P7"]

# ---------------- 六个面 ----------------
floor = [p0, p1, p2, p3]
ceiling = [p4, p5, p6, p7]

wall1 = [p0, p1, p5, p4]  # 前墙
wall2 = [p1, p2, p6, p5]  # 右墙
wall3 = [p2, p3, p7, p6]  # 后墙
wall4 = [p3, p0, p4, p7]  # 左墙

faces = [floor, ceiling, wall1, wall2, wall3, wall4]

face_names = ["FLOOR", "CEILING", "WALL-1", "WALL-2", "WALL-3", "WALL-4"]

# ---------------- 绘图 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

# ---------------- 面颜色 ----------------
face_colors = [
    (0.8, 0.8, 0.8, 1.0),   # 地面
    (0.9, 0.9, 0.9, 0.15),  # 顶面透明
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
    (0.2, 0.6, 1.0, 0.3),
]

# ---------------- 画面 ----------------
for face, color in zip(faces, face_colors):
    poly = Poly3DCollection([face], facecolors=color, edgecolors='k', linewidths=0.8)
    ax.add_collection3d(poly)

# ---------------- 墙面编号 ----------------
for i, face in enumerate(faces):
    centroid = np.mean(np.array(face), axis=0)
    ax.text(
        centroid[0], centroid[1], centroid[2],
        face_names[i],
        color="black",
        fontsize=11,
        ha="center"
    )

# ---------------- 顶点编号 ----------------
for name, (x, y, z) in points.items():
    ax.scatter(x, y, z, color='red', s=25)
    ax.text(
        x + 0.05,
        y + 0.05,
        z + 0.05,
        name,
        color="red",
        fontsize=10
    )

# ---------------- 坐标轴 ----------------
ax.set_xlim(0, L)
ax.set_ylim(0, W)
ax.set_zlim(0, H)

ax.set_xlabel("X (4.2m)")
ax.set_ylabel("Y (3.6m)")
ax.set_zlabel("Z (3.0m)")

ax.set_box_aspect([L, W, H])
ax.view_init(elev=20, azim=235)

plt.title("3D Room with Vertex + Wall Labels")
plt.show()
```
