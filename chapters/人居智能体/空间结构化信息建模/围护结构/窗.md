# 窗

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 房间尺寸 ----------------
L, W, H = 4.2, 3.6, 3.0

# ---------------- 窗户参数（支持扩展多个） ----------------
windows = [
    {
        "width": 2.0,
        "height": 2.0,
        "x": 1.0,
        "z": 0.75
    }
]

# ---------------- 顶点定义 ----------------
points = {
    "P0": (0, 0, 0), "P1": (L, 0, 0), "P2": (L, W, 0), "P3": (0, W, 0),
    "P4": (0, 0, H), "P5": (L, 0, H), "P6": (L, W, H), "P7": (0, W, H),
}

p0, p1, p2, p3 = points["P0"], points["P1"], points["P2"], points["P3"]
p4, p5, p6, p7 = points["P4"], points["P5"], points["P6"], points["P7"]

# ---------------- 房间面 ----------------
room_faces = [
    {"name": "FLOOR",   "verts": [p0, p1, p2, p3], "color": (0.8, 0.8, 0.8, 1.0)},
    {"name": "CEILING", "verts": [p4, p5, p6, p7], "color": (0.9, 0.9, 0.9, 0.15)},
    {"name": "WALL-1",  "verts": [p0, p1, p5, p4], "color": (0.2, 0.6, 1.0, 0.2)},
    {"name": "WALL-2",  "verts": [p1, p2, p6, p5], "color": (0.2, 0.6, 1.0, 0.3)},
    {"name": "WALL-3",  "verts": [p2, p3, p7, p6], "color": (0.2, 0.6, 1.0, 0.3)},
    {"name": "WALL-4",  "verts": [p3, p0, p4, p7], "color": (0.2, 0.6, 1.0, 0.3)},
]

# ---------------- 生成窗户几何 ----------------
def create_window(window):
    x0, z0 = window["x"], window["z"]
    x1 = x0 + window["width"]
    z1 = z0 + window["height"]

    return [
        (x0, 0, z0),
        (x1, 0, z0),
        (x1, 0, z1),
        (x0, 0, z1),
    ]

# ---------------- 绘图 ----------------
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')

# ---------------- 1. 绘制房间 ----------------
for face in room_faces:
    poly = Poly3DCollection(
        [face["verts"]],
        facecolors=face["color"],
        edgecolors='k',
        linewidths=0.8
    )
    ax.add_collection3d(poly)

    # 面中心标注
    centroid = np.mean(np.array(face["verts"]), axis=0)
    ax.text(*centroid, face["name"], fontsize=9, ha='center')

# ---------------- 2. 绘制窗户 ----------------
for w in windows:
    win_face = create_window(w)

    poly = Poly3DCollection(
        [win_face],
        facecolors=(0.7, 0.9, 1.0, 0.8),
        edgecolors='darkblue',
        linewidths=2
    )
    ax.add_collection3d(poly)

    # 窗户标注
    centroid = np.mean(np.array(win_face), axis=0)
    ax.text(
        centroid[0],
        -0.1,
        centroid[2],
        f'WINDOW\n({w["width"]}m × {w["height"]}m)',
        color='darkblue',
        fontsize=9,
        ha='center'
    )

# ---------------- 3. 顶点 ----------------
for name, (x, y, z) in points.items():
    ax.scatter(x, y, z, color='red', s=20)
    ax.text(x + 0.05, y + 0.05, z + 0.05, name, fontsize=8)

# ---------------- 坐标轴 ----------------
ax.set_xlim(0, L)
ax.set_ylim(0, W)
ax.set_zlim(0, H)

ax.set_xlabel("X (m)")
ax.set_ylabel("Y (m)")
ax.set_zlabel("Z (m)")

ax.set_box_aspect([L, W, H])

# ---------------- 视角 ----------------
ax.view_init(elev=20, azim=235)

plt.title("3D Room (Structured + Parametric Window)")
plt.show()
```
