# 窗

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 参数 ----------------
L, W, H = 4.2, 3.6, 3.0
t = 0.2

# ---------------- 窗户 ----------------
windows = [
    {"width": 2.0, "height": 2.0, "x": 1.0, "z": 0.75}
]

# ---------------- 8个顶点 ----------------
P = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

# ---------------- cuboid ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz
    return [
        [(x, y, z), (x1, y, z), (x1, y1, z), (x, y1, z)],
        [(x, y, z1), (x1, y, z1), (x1, y1, z1), (x, y1, z1)],
        [(x, y, z), (x1, y, z), (x1, y, z1), (x, y, z1)],
        [(x1, y, z), (x1, y1, z), (x1, y1, z1), (x1, y, z1)],
        [(x1, y1, z), (x, y1, z), (x, y1, z1), (x1, y1, z1)],
        [(x, y1, z), (x, y, z), (x, y, z1), (x, y1, z1)],
    ]

# ---------------- Wall-2开窗 ----------------
def cut_wall(L, t, H, windows):
    faces = []
    cursor = 0

    for w in windows:
        wx0 = w["x"]
        wx1 = wx0 + w["width"]
        wz0 = w["z"]
        wz1 = wz0 + w["height"]

        if wx0 > cursor:
            faces.extend(cuboid(cursor, -t/2, 0, wx0 - cursor, t, H))

        if wz0 > 0:
            faces.extend(cuboid(wx0, -t/2, 0, w["width"], t, wz0))

        if wz1 < H:
            faces.extend(cuboid(wx0, -t/2, wz1, w["width"], t, H - wz1))

        cursor = wx1

    if cursor < L:
        faces.extend(cuboid(cursor, -t/2, 0, L - cursor, t, H))

    return faces

# ---------------- flatten ----------------
def flatten(faces):
    out = []
    for f in faces:
        arr = np.array(f)
        if arr.ndim == 2 and arr.shape[1] == 3:
            out.append(arr)
    return out

# ---------------- add ----------------
def add(ax, faces, color):
    poly = Poly3DCollection(
        faces,
        facecolors=color,
        edgecolors="black",
        linewidths=0.3,
        alpha=color[3]
    )
    ax.add_collection3d(poly)

# ---------------- 画布 ----------------
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')

wall_color = (0.2, 0.6, 1.0, 0.25)
floor_color = (0.7, 0.7, 0.7, 1.0)

# ---------------- 墙体 ----------------
wall_faces = []

# WALL-1（后墙）
wall_faces.extend(cuboid(0, W - t/2, 0, L, t, H))

# WALL-2（开窗墙）
wall_faces.extend(cut_wall(L, t, H, windows))

# WALL-3
wall_faces.extend(cuboid(-t/2, 0, 0, t, W, H))

# WALL-4
wall_faces.extend(cuboid(L - t/2, 0, 0, t, W, H))

# ---------------- 楼板 ----------------
floor_faces = cuboid(0, 0, -t, L, W, t)

# ---------------- 绘制 ----------------
add(ax, flatten(wall_faces), wall_color)
add(ax, flatten(floor_faces), floor_color)

# ---------------- 墙标注 ----------------
ax.text(L/2, W - t/2, H + 0.05, "WALL-1", ha="center", color="black")
ax.text(L/2, -t/2, H/2, "WALL-2", ha="center", color="black")
ax.text(-t/2, W/2, H/2, "WALL-3", ha="center", rotation=90)
ax.text(L + t/2, W/2, H/2, "WALL-4", ha="center", rotation=90)

# ---------------- 顶点显示 ----------------
for name, (x, y, z) in P.items():
    ax.scatter(x, y, z, color="red", s=25)
    ax.text(x, y, z, name, color="red")

# ---------------- 窗标注 ----------------
for w in windows:
    cx = w["x"] + w["width"]/2
    cy = -t/2
    cz = w["z"] + w["height"]/2
    ax.text(cx, cy, cz, "WINDOW", color="blue")

# ---------------- 视图 ----------------
ax.set_xlim(-0.5, L + 0.5)
ax.set_ylim(-0.5, W + 0.5)
ax.set_zlim(0, H)

ax.set_box_aspect([L, W, H])
ax.view_init(elev=20, azim=235)

ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")

plt.title("BIM Room with Points + Walls + Window")
plt.tight_layout()
plt.show()
```
