# 门

> 注明:
>  
- 通过在墙体长方体中进行局部布尔减运算（Opening），形成门洞；并在门洞位置嵌入具有厚度的门体体元（Door Cuboid），实现“开口—填充”的构造表达。

门构件的创建过程如下：

1. 定义门参数：包括门的水平范围（x0 → x1）及竖向范围（z0 → z1），用于确定门洞的位置与尺寸。
2. 定位门洞范围：在对应墙体（如 WALL-3）上，确定门洞的水平区间（x0 → x1）及竖向区间（z0 → z1）。
3. 墙体局部分段：将门洞所在区域从原墙体中剔除，对剩余部分进行分段表达：
- 左侧墙体段
- 右侧墙体段
- 门上墙体段
4.（等价表达）门洞生成：该过程在几何上等价于：原墙体 − 门洞体元（Cuboid） = 开门墙体
5. 生成门实体：在门洞位置创建一个具有厚度（door_t）的长方体，其尺寸与门洞一致。
6. 构件组合：将开门后的墙体与门实体进行组合，形成完整门构件。
7. 进行语义标注：对门构件标注为“DOOR”，并与所属墙体建立空间关联关系。

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 房间尺寸 ----------------
L, W, H = 4.2, 3.6, 3.0
t = 0.2
glass_t = 0.01
door_t = 0.02  # 20mm门厚

# ---------------- 顶点（新增） ----------------
P = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

# ---------------- 窗户（WALL-1） ----------------
windows = [
    {"width": 2.1, "height": 1.8, "x": 1.05, "z": 0.75}
]

# ---------------- 门（WALL-3） ----------------
doors = [
    {
        "x0": 0.2,
        "x1": 0.2 + 0.9,
        "z0": 0,
        "z1": 2.1
    }
]

# ---------------- cuboid ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz
    return [
        [(x, y, z), (x1, y, z), (x1, y1, z), (x, y1, z)],
        [(x, y, z1), (x1, y, z1), (x1, y1, z1), (x, y1, z1)],
        [(x, y, z), (x1, y, z), (x1, y, z1), (x, y, z1)],
        [(x1, y, z), (x1, y1, z), (x1, y1, z1), (x1, y, z1)],
        [(x1, y1, z), (x, y1, z), (x, y1, z1), (x1, y1, z1)],
        [(x, y1, z), (x, y, z), (x, y, z1), (x, y1, z1)],
    ]

# ---------------- 南墙开窗 ----------------
def cut_wall_south(L, t, H, windows):
    faces = []
    cursor = 0

    for w in windows:
        wx0 = w["x"]
        wx1 = wx0 + w["width"]
        wz0 = w["z"]
        wz1 = wz0 + w["height"]

        if wx0 > cursor:
            faces.extend(cuboid(cursor, -t/2, 0, wx0 - cursor, t, H))

        if wz0 > 0:
            faces.extend(cuboid(wx0, -t/2, 0, w["width"], t, wz0))

        if wz1 < H:
            faces.extend(cuboid(wx0, -t/2, wz1, w["width"], t, H - wz1))

        cursor = wx1

    if cursor < L:
        faces.extend(cuboid(cursor, -t/2, 0, L - cursor, t, H))

    return faces

# ---------------- 北墙开门（挖空） ----------------
def cut_wall_north(L, W, t, H, doors):
    faces = []

    for d in doors:
        dx0 = d["x0"]
        dx1 = d["x1"]
        dz1 = d["z1"]

        if dx0 > 0:
            faces.extend(cuboid(0, W - t/2, 0, dx0, t, H))

        if dx1 < L:
            faces.extend(cuboid(dx1, W - t/2, 0, L - dx1, t, H))

        if dz1 < H:
            faces.extend(cuboid(dx0, W - t/2, dz1, dx1 - dx0, t, H - dz1))

    return faces

# ---------------- 门实体 ----------------
def make_door_solid(W, doors, door_t):
    faces = []
    for d in doors:
        dx0 = d["x0"]
        dx1 = d["x1"]
        dz0 = d["z0"]
        dz1 = d["z1"]

        faces.extend(
            cuboid(
                dx0,
                W - door_t/2,
                dz0,
                dx1 - dx0,
                door_t,
                dz1 - dz0
            )
        )
    return faces

# ---------------- 玻璃 ----------------
def make_window_glass(windows, glass_t):
    faces = []
    for w in windows:
        faces.extend(
            cuboid(
                w["x"],
                -glass_t/2,
                w["z"],
                w["width"],
                glass_t,
                w["height"]
            )
        )
    return faces

# ---------------- flatten ----------------
def flatten(faces):
    out = []
    for f in faces:
        arr = np.array(f)
        if arr.ndim == 2:
            out.append(arr)
    return out

# ---------------- add ----------------
def add(ax, faces, color):
    poly = Poly3DCollection(
        faces,
        facecolors=color,
        edgecolors="black",
        linewidths=0.3,
        alpha=color[3]
    )
    ax.add_collection3d(poly)

# ---------------- plot ----------------
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')

wall_color = (0.2, 0.6, 1.0, 0.25)
floor_color = (0.7, 0.7, 0.7, 1.0)
glass_color = (0.3, 0.8, 1.0, 0.15)
door_color = (1.0, 1.0, 1.0, 1.0)

wall_faces = []
wall_faces.extend(cut_wall_south(L, t, H, windows))
wall_faces.extend(cuboid(L - t/2, 0, 0, t, W, H))
wall_faces.extend(cut_wall_north(L, W, t, H, doors))
wall_faces.extend(cuboid(-t/2, 0, 0, t, W, H))

floor_faces = cuboid(0, 0, -t, L, W, t)
glass_faces = make_window_glass(windows, glass_t)
door_faces = make_door_solid(W, doors, door_t)

add(ax, flatten(wall_faces), wall_color)
add(ax, flatten(floor_faces), floor_color)
add(ax, flatten(glass_faces), glass_color)
add(ax, flatten(door_faces), door_color)

# ---------------- 墙体标注（保留语义，仅补充一致性） ----------------
ax.text(L/2, 0, H/2, "WALL-1 (S)", ha="center")
ax.text(L, W/2, H/2, "WALL-2 (E)", ha="center")
ax.text(L/2, W, H/2, "WALL-3 (N - DOOR)", ha="center")
ax.text(0, W/2, H/2, "WALL-4 (W)", ha="center")

# ---------------- 顶点显示（新增） ----------------
for name, (x, y, z) in P.items():
    ax.scatter(x, y, z, color="red", s=25)
    ax.text(x + 0.05, y + 0.05, z + 0.05, name)

# ---------------- 视角 ----------------
ax.view_init(elev=20, azim=235)

ax.set_xlim(0, L)
ax.set_ylim(0, W)
ax.set_zlim(0, H)
ax.set_box_aspect([L, W, H])

plt.tight_layout()
plt.show()
```
