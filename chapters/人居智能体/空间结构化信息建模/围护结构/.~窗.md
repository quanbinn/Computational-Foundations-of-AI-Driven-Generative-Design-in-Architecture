# 窗

> 注明:
>  
- 通过在墙体长方体中进行局部布尔减运算（Opening），形成窗洞；并在窗洞位置嵌入具有厚度的玻璃体元（Glass Cuboid），实现“开口—填充”的构造表达。

窗构件的创建过程如下：

1. 定义窗的基本参数：设定窗的宽度（2.1 m）、高度（1.8 m）、窗左下角距南墙起点的水平距离（1.05 m）以及窗下沿距地面的高度（0.75 m）。
2. 定位窗洞范围：在对应墙体（如 WALL-1）上，确定窗洞的水平区间（x → x + width）及竖向区间（z → z + height）。
3. 切割南墙：将 WALL-1 拆分为四个独立的长方体构件：
- 左侧墙体段
- 窗下墙体段
- 窗上墙体段
- 右侧墙体段
4. 重建墙体实体：对上述各分段分别进行厚度外推（t），生成多个长方体，从而替代原始完整墙体，实现窗洞的“隐式切割”。
5. 窗洞生成机制：上述分段过程在几何上等价于：原墙体 − 窗洞体元（Cuboid） = 开窗墙体
6. 生成窗填充构件（玻璃）：在窗洞位置创建一个具有厚度（glass_t） 为0.01 m的长方体，其尺寸与窗洞一致，材质为半透明玻璃。
7. 设置微偏移（避免共面冲突）：在法向方向（如Y方向）对玻璃构件施加微小偏移（ε），避免渲染中的 Z-fighting 问题。
8. 完成构件组合：将墙体（Wall）、窗洞（Opening）与玻璃（Glass）组合，形成完整的“墙-窗系统”。
9. 进行语义标注：对所有顶点（P0–P7）、墙体（WALL-1~4）、楼板（Floor）以及窗（Window）进行编号和可视化标注。。

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# ---------------- 参数 ----------------
L, W, H = 4.2, 3.6, 3.0
t = 0.2
glass_t = 0.01  # 10mm 玻璃厚度

# ---------------- 窗户 ----------------
windows = [
    {"width": 2.1, "height": 1.8, "x": 1.05, "z": 0.75}
]

# ---------------- 8个顶点 ----------------
P = {
    "P0": (0, 0, 0),
    "P1": (L, 0, 0),
    "P2": (L, W, 0),
    "P3": (0, W, 0),
    "P4": (0, 0, H),
    "P5": (L, 0, H),
    "P6": (L, W, H),
    "P7": (0, W, H),
}

# ---------------- cuboid ----------------
def cuboid(x, y, z, dx, dy, dz):
    x1, y1, z1 = x + dx, y + dy, z + dz
    return [
        [(x, y, z), (x1, y, z), (x1, y1, z), (x, y1, z)],
        [(x, y, z1), (x1, y, z1), (x1, y1, z1), (x, y1, z1)],
        [(x, y, z), (x1, y, z), (x1, y, z1), (x, y, z1)],
        [(x1, y, z), (x1, y1, z), (x1, y1, z1), (x1, y, z1)],
        [(x1, y1, z), (x, y1, z), (x, y1, z1), (x1, y1, z1)],
        [(x, y1, z), (x, y, z), (x, y, z1), (x, y1, z1)],
    ]

# ---------------- 南墙开窗 ----------------
def cut_wall_south(L, t, H, windows):
    faces = []
    cursor = 0

    for w in windows:
        wx0 = w["x"]
        wx1 = wx0 + w["width"]
        wz0 = w["z"]
        wz1 = wz0 + w["height"]

        # 左墙
        if wx0 > cursor:
            faces.extend(cuboid(cursor, -t/2, 0, wx0 - cursor, t, H))

        # 窗下
        if wz0 > 0:
            faces.extend(cuboid(wx0, -t/2, 0, w["width"], t, wz0))

        # 窗上
        if wz1 < H:
            faces.extend(cuboid(wx0, -t/2, wz1, w["width"], t, H - wz1))

        cursor = wx1

    # 右墙
    if cursor < L:
        faces.extend(cuboid(cursor, -t/2, 0, L - cursor, t, H))

    return faces

# ---------------- 窗户玻璃 ----------------
def make_window_glass(windows, glass_t):
    faces = []
    for w in windows:
        wx0 = w["x"]
        wz0 = w["z"]

        # 微小偏移避免闪烁
        y_offset = 0.001

        faces.extend(
            cuboid(
                wx0,
                -glass_t/2 + y_offset,
                wz0,
                w["width"],
                glass_t,
                w["height"]
            )
        )
    return faces

# ---------------- flatten ----------------
def flatten(faces):
    out = []
    for f in faces:
        arr = np.array(f)
        if arr.ndim == 2:
            out.append(arr)
    return out

# ---------------- add ----------------
def add(ax, faces, color):
    poly = Poly3DCollection(
        faces,
        facecolors=color,
        edgecolors="black",
        linewidths=0.3,
        alpha=color[3]
    )
    ax.add_collection3d(poly)

# ---------------- 画布 ----------------
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')

wall_color = (0.2, 0.6, 1.0, 0.25)
floor_color = (0.7, 0.7, 0.7, 1.0)
glass_color = (0.3, 0.8, 1.0, 0.15)

# ---------------- 墙体 ----------------
wall_faces = []

# WALL-1（南）
wall_faces.extend(cut_wall_south(L, t, H, windows))

# WALL-2（东）
wall_faces.extend(cuboid(L - t/2, 0, 0, t, W, H))

# WALL-3（北）
wall_faces.extend(cuboid(0, W - t/2, 0, L, t, H))

# WALL-4（西）
wall_faces.extend(cuboid(-t/2, 0, 0, t, W, H))

# ---------------- 楼板 ----------------
floor_faces = cuboid(0, 0, -t, L, W, t)

# ---------------- 玻璃 ----------------
glass_faces = make_window_glass(windows, glass_t)

# ---------------- 绘制 ----------------
add(ax, flatten(wall_faces), wall_color)
add(ax, flatten(floor_faces), floor_color)
add(ax, flatten(glass_faces), glass_color)

# ---------------- 墙体标注 ----------------
ax.text(L/2, 0, H/2, "WALL-1 (S)", ha="center")
ax.text(L, W/2, H/2, "WALL-2 (E)", ha="center")
ax.text(L/2, W, H/2, "WALL-3 (N)", ha="center")
ax.text(0, W/2, H/2, "WALL-4 (W)", ha="center")

# ---------------- 顶点 ----------------
for name, (x, y, z) in P.items():
    ax.scatter(x, y, z, color="red", s=25)
    ax.text(x + 0.05, y + 0.05, z + 0.05, name)

# ---------------- 窗标注 ----------------
for w in windows:
    cx = w["x"] + w["width"]/2
    cy = 0
    cz = w["z"] + w["height"]/2
    ax.text(cx, cy, cz, "WINDOW", color="blue")

# ---------------- 视图 ----------------
ax.set_xlim(-0.5, L + 0.5)
ax.set_ylim(-0.5, W + 0.5)
ax.set_zlim(0, H)

ax.set_box_aspect([L, W, H])
ax.view_init(elev=20, azim=235)

ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")

plt.title("Topological BIM Room (Wall + Opening + Glass)")
plt.tight_layout()
plt.show()
```
