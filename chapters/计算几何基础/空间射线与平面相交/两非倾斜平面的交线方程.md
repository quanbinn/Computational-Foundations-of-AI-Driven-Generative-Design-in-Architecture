# 两非倾斜平面的交线方程

```python
import numpy as np
import matplotlib.pyplot as plt

# ---------------- 两非倾斜（坐标平行）平面及其交线 ----------------
# 平面 π1：z = z0（水平面）
# 平面 π2：x = x0（竖直面）
# 交线 L：同时满足 z = z0 且 x = x0，向量为沿 y 轴方向，参数方程为
#   (x, y, z) = (x0, t, z0)，t ∈ ℝ

z0, x0 = 0.0, 1.5

fig = plt.figure()
ax = fig.add_subplot(111, projection="3d")

# ---------------- 绘制平面 π1：z = z0（浅蓝色） ----------------
X1, Y1 = np.meshgrid(np.linspace(-2, 4, 12), np.linspace(-2, 4, 12))
Z1 = np.full_like(X1, z0)
ax.plot_surface(X1, Y1, Z1, alpha=0.35, color="C0")

# ---------------- 绘制平面 π2：x = x0（浅绿色） ----------------
Y2, Z2 = np.meshgrid(np.linspace(-2, 4, 12), np.linspace(-2, 4, 12))
X2 = np.full_like(Y2, x0)
ax.plot_surface(X2, Y2, Z2, alpha=0.35, color="C2")

# ---------------- 交线：直线 (x0, t, z0) ----------------
t_line = np.linspace(-2, 4, 80)
ax.plot(np.full_like(t_line, x0), t_line, np.full_like(t_line, z0), color="red", linewidth=2.5, label="Line of intersection L")

ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")
ax.set_title("Intersection of axis-aligned planes: z = {:.1f} and x = {:.1f}".format(z0, x0))
ax.legend()
plt.show()
```
