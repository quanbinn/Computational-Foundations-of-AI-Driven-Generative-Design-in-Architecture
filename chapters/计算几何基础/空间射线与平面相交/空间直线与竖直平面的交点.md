# 空间直线与竖直平面的交点

```python
import numpy as np
import matplotlib.pyplot as plt

# ---------------- 空间直线 P(t) = O + t * d 与竖直平面 x = x_plane 的交点 ----------------
def line_vertical_plane_x_intersection(O, d, x_plane, eps=1e-9):
    """
    竖直平面：x = 常数（法向量沿 X 轴）
    """
    ox, oy, oz = O
    dx, dy, dz = d
    if abs(dx) < eps:
        if abs(ox - x_plane) < eps:
            return None
        return None
    t = (x_plane - ox) / dx
    return [ox + t * dx, oy + t * dy, oz + t * dz]


O = np.array([0.0, 1.0, 0.5])
d = np.array([1.2, -0.3, 0.7])
x_plane = 2.0
P = line_vertical_plane_x_intersection(O.tolist(), d.tolist(), x_plane)
print("与竖直平面 x = {:.1f} 的交点：".format(x_plane), P)

fig = plt.figure()
ax = fig.add_subplot(111, projection="3d")

Yg, Zg = np.meshgrid(np.linspace(-1, 3, 8), np.linspace(-1, 3, 8))
Xg = np.full_like(Yg, x_plane)
ax.plot_surface(Xg, Yg, Zg, alpha=0.35, color="C2")

ts = np.linspace(-0.5, 2.5, 100)
line_pts = O.reshape(1, 3) + ts.reshape(-1, 1) * d.reshape(1, 3)
ax.plot(line_pts[:, 0], line_pts[:, 1], line_pts[:, 2], color="gray", label="Line")
if P is not None:
    ax.scatter([P[0]], [P[1]], [P[2]], color="red", s=60, label="Intersection")

ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")
ax.legend()
plt.show()
```
