# 空间直线与水平平面的交点

```python
import numpy as np
import matplotlib.pyplot as plt

# ---------------- 空间直线 P(t) = O + t * d 与水平面 z = z_plane 的交点 ----------------
def line_horizontal_plane_intersection(O, d, z_plane, eps=1e-9):
    """
    O: 直线上的点 [ox, oy, oz]
    d: 方向向量 [dx, dy, dz]（可为非单位向量）
    z_plane: 水平面高度 z = 常数
    返回交点坐标；若无唯一交点（与平面平行且不在面上）则返回 None
    """
    ox, oy, oz = O
    dx, dy, dz = d
    if abs(dz) < eps:
        if abs(oz - z_plane) < eps:
            return None  # 直线落在平面内，不取单点
        return None  # 与平面平行且不相交
    t = (z_plane - oz) / dz
    return [ox + t * dx, oy + t * dy, oz + t * dz]


O = np.array([0.5, -0.2, 1.0])
d = np.array([1.0, 0.5, -0.8])
z_plane = 0.0
P = line_horizontal_plane_intersection(O.tolist(), d.tolist(), z_plane)
print("与水平面 z = {:.1f} 的交点：".format(z_plane), P)

# ---------------- 可视化 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection="3d")

xx, yy = np.meshgrid(np.linspace(-1, 3, 8), np.linspace(-1, 3, 8))
zz = np.full_like(xx, z_plane)
ax.plot_surface(xx, yy, zz, alpha=0.35, color="C0")

ts = np.linspace(-0.5, 2.5, 100)
line_pts = O.reshape(1, 3) + ts.reshape(-1, 1) * d.reshape(1, 3)
ax.plot(line_pts[:, 0], line_pts[:, 1], line_pts[:, 2], color="gray", label="Line")
if P is not None:
    ax.scatter([P[0]], [P[1]], [P[2]], color="red", s=60, label="Intersection")

ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")
ax.legend()
plt.show()
```
