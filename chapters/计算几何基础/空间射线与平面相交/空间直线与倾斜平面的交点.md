# 空间直线与倾斜平面的交点

```python
import numpy as np
import matplotlib.pyplot as plt

# ---------------- 空间直线 P(t) = O + t * d 与一般平面 ax + by + cz + d0 = 0 的交点 ----------------
def line_plane_intersection(O, direction, plane_abcd, eps=1e-9):
    """
    plane_abcd: (a, b, c, d0) 满足 a*x + b*y + c*z + d0 = 0
    """
    a, b, c, d0 = plane_abcd
    ox, oy, oz = O
    dx, dy, dz = direction
    denom = a * dx + b * dy + c * dz
    if abs(denom) < eps:
        return None  # 与平面平行（或在平面内)
    num = -(a * ox + b * oy + c * oz + d0)
    t = num / denom
    return [ox + t * dx, oy + t * dy, oz + t * dz]


# 倾斜平面示例：x + y + z - 3 = 0
O = np.array([0.5, 0.0, 1.0])
d = np.array([0.0, 1.0, -0.5])
plane = (1.0, 1.0, 1.0, -3.0)
P = line_plane_intersection(O.tolist(), d.tolist(), plane)
print("与平面 x+y+z-3=0 的交点：", P)

fig = plt.figure()
ax = fig.add_subplot(111, projection="3d")

a, b, c, d0 = plane
grid = np.linspace(-1, 4, 15)
X, Y = np.meshgrid(grid, grid)
Z = (-(a * X + b * Y + d0)) / c  # 假定 c != 0
ax.plot_surface(X, Y, Z, alpha=0.35, color="C1")

ts = np.linspace(-1, 4, 120)
line_pts = O.reshape(1, 3) + ts.reshape(-1, 1) * d.reshape(1, 3)
ax.plot(line_pts[:, 0], line_pts[:, 1], line_pts[:, 2], color="gray", label="Line")
if P is not None:
    ax.scatter([P[0]], [P[1]], [P[2]], color="red", s=60, label="Intersection")

ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")
ax.legend()
plt.show()
```
