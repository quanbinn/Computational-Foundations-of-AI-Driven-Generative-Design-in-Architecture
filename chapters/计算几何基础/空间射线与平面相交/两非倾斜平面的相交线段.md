# 两非倾斜平面的相交线段

```python
import numpy as np
import matplotlib.pyplot as plt

# ---------------- 两非倾斜平面的交线被区间截得的线段 ----------------
# 沿用：π1: z = z0，π2: x = x0，无限交线为 (x0, t, z0)
# 截取 t ∈ [t_min, t_max] 得到一条有限线段

z0, x0 = 0.0, 1.0
t_min, t_max = -0.5, 2.5

fig = plt.figure()
ax = fig.add_subplot(111, projection="3d")

X1, Y1 = np.meshgrid(np.linspace(-1, 3, 10), np.linspace(-1, 3, 10))
Z1 = np.full_like(X1, z0)
ax.plot_surface(X1, Y1, Z1, alpha=0.3, color="C0")

Y2, Z2 = np.meshgrid(np.linspace(-1, 3, 10), np.linspace(-1, 3, 10))
X2 = np.full_like(Y2, x0)
ax.plot_surface(X2, Y2, Z2, alpha=0.3, color="C2")

# ---------------- 交线上的线段 ----------------
t_seg = np.linspace(t_min, t_max, 50)
ax.plot(
    np.full_like(t_seg, x0),
    t_seg,
    np.full_like(t_seg, z0),
    color="red",
    linewidth=3,
    label="Intersection segment",
)

ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")
ax.set_title("Intersection segment for t in [{:.1f}, {:.1f}]".format(t_min, t_max))
ax.legend()
plt.show()
```
