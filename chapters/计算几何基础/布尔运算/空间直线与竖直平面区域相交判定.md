# 空间直线与竖直平面区域相交判定

## 开始做实体实验

![](/images/计算几何基础/布尔运算/空间直线与竖直平面区域相交判定/1a1.jpg)

```python
import numpy as np
import matplotlib.pyplot as plt

def line_vertical_plane_x_rect(O, d, x_plane, y_range, z_range, eps=1e-9):
    """
    空间直线 P(t)=O+t*d 与「竖直平面 x=x_plane 上的矩形区域」是否相交
    矩形：y ∈ y_range, z ∈ z_range
    """
    ox, oy, oz = O
    dx, dy, dz = d
    if abs(dx) < eps:
        return False, None
    t = (x_plane - ox) / dx
    x = ox + t * dx
    y = oy + t * dy
    z = oz + t * dz
    if y_range[0] - eps <= y <= y_range[1] + eps and z_range[0] - eps <= z <= z_range[1] + eps:
        return True, [x, y, z]
    return False, None

O = np.array([-0.5, 1.0, 1.0])
d = np.array([1.0, 0.2, -0.3])
x_plane = 2.0
y_rng, z_rng = (0.0, 3.0), (0.0, 2.5)
hit, P = line_vertical_plane_x_rect(O, d, x_plane, y_rng, z_rng)
print("与竖直平面矩形区域相交：", hit, P)

fig = plt.figure()
ax = fig.add_subplot(111, projection="3d")
Yg, Zg = np.meshgrid(np.linspace(*y_rng, 10), np.linspace(*z_rng, 10))
Xg = np.full_like(Yg, x_plane)
ax.plot_surface(Xg, Yg, Zg, alpha=0.35, color="C2")
ts = np.linspace(-0.5, 4.0, 100)
L = O.reshape(1, 3) + ts.reshape(-1, 1) * d.reshape(1, 3)
ax.plot(L[:, 0], L[:, 1], L[:, 2], color="gray", linewidth=2, label="Line")
if P:
    ax.scatter([P[0]], [P[1]], [P[2]], color="red", s=60, label="Intersection in region")
ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")
ax.set_title("Line vs. rectangle on vertical plane (x = const.)")
ax.legend()

plt.show()
```
