```python
import matplotlib.pyplot as plt

def segment_intersect(a, b, c, d, eps=1e-9):
    """
    判定二维线段 AB 与 CD 是否相交（含端点相接视为相交）
    a,b,c,d: [x, y]
    思路：求直线交点参数 t,u ∈ [0,1]
    """
    x1, y1 = a
    x2, y2 = b
    x3, y3 = c
    x4, y4 = d
    denom = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4)
    if abs(denom) < eps:
        # 平行或共线：此处返回 False；更精细的共线重叠需另行处理
        return False, None
    t = ((x1 - x3) * (y3 - y4) - (y1 - y3) * (x3 - x4)) / denom
    u = ((x1 - x3) * (y1 - y2) - (y1 - y3) * (x1 - x2)) / denom
    if -eps <= t <= 1 + eps and -eps <= u <= 1 + eps:
        ix = x1 + t * (x2 - x1)
        iy = y1 + t * (y2 - y1)
        return True, [ix, iy]
    return False, None

# ---------------- 示例 ----------------
seg1 = ([1, 1], [5, 4])
seg2 = ([2, 0], [3, 6])
ok, pt = segment_intersect(seg1[0], seg1[1], seg2[0], seg2[1])
print("是否相交：", ok, "交点近似：", pt)

plt.axis("equal")
plt.plot([seg1[0][0], seg1[1][0]], [seg1[0][1], seg1[1][1]], "b-", linewidth=2, label="Segment AB")
plt.plot([seg2[0][0], seg2[1][0]], [seg2[0][1], seg2[1][1]], "g-", linewidth=2, label="Segment CD")
if pt:
    plt.scatter([pt[0]], [pt[1]], color="red", s=80, zorder=5, label="Intersection")
plt.legend()
plt.title("Segment intersection test")
plt.xlabel("x")
plt.ylabel("y")

plt.show()
```
