# 空间直线与倾斜平面区域相交判定

## 开始做实体实验

![](/images/计算几何基础/布尔运算/空间直线与倾斜平面区域相交判定/1a1.jpg)

```python
import numpy as np
import matplotlib.pyplot as plt

def line_parallelogram_patch(O, d, C0, u, v, eps=1e-9):
    """
    直线 P(k)=O+k*d 与倾斜平面上平行四边形区域是否相交
    区域：{ C0 + s*u + t*v | 0<=s<=1, 0<=t<=1 }
    解 k*d - s*u - t*v = C0 - O
    """
    O = np.asarray(O, dtype=float)
    d = np.asarray(d, dtype=float)
    C0 = np.asarray(C0, dtype=float)
    u = np.asarray(u, dtype=float)
    v = np.asarray(v, dtype=float)
    M = np.column_stack([d, -u, -v])
    rhs = C0 - O
    if abs(np.linalg.det(M)) < eps:
        return False, None
    sol = np.linalg.solve(M, rhs)
    k, s, t = sol
    if -eps <= s <= 1 + eps and -eps <= t <= 1 + eps:
        P = O + k * d
        return True, P.tolist()
    return False, None

O = np.array([0.0, 0.0, 2.5])
d = np.array([0.6, 0.5, -1.0])
C0 = np.array([0.0, 0.0, 0.5])
u = np.array([2.0, 0.0, -0.5])
v = np.array([0.0, 2.0, -0.5])

hit, P = line_parallelogram_patch(O, d, C0, u, v)
print("与倾斜平面上平行四边形区域相交：", hit, P)

fig = plt.figure()
ax = fig.add_subplot(111, projection="3d")
su = np.linspace(0, 1, 12)
tv = np.linspace(0, 1, 12)
S, T = np.meshgrid(su, tv)
X = C0[0] + S * u[0] + T * v[0]
Y = C0[1] + S * u[1] + T * v[1]
Z = C0[2] + S * u[2] + T * v[2]
ax.plot_surface(X, Y, Z, alpha=0.4, color="C1")

ks = np.linspace(-0.5, 4.0, 100)
L = O.reshape(1, 3) + ks.reshape(-1, 1) * d.reshape(1, 3)
ax.plot(L[:, 0], L[:, 1], L[:, 2], color="gray", linewidth=2, label="Line")

if P:
    ax.scatter([P[0]], [P[1]], [P[2]], color="red", s=60, label="Intersection in region")
corners = [C0, C0 + u, C0 + u + v, C0 + v, C0]
cx, cy, cz = zip(*corners)
ax.plot(cx, cy, cz, "k--", linewidth=1, alpha=0.7, label="Region boundary")

ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")
ax.set_title("Line vs. parallelogram patch on oblique plane")
ax.legend()
plt.show()
```
