# 空间直线与水平平面区域相交判定

## 开始做实体实验

![](/images/计算几何基础/布尔运算/空间直线与水平平面区域相交判定/1a1.jpg)

```python
import numpy as np
import matplotlib.pyplot as plt

def line_horizontal_plane_rect(O, d, z_plane, x_range, y_range, eps=1e-9):
    """
    空间直线 P(t)=O+t*d 与「水平面 z=z_plane 上的矩形区域」是否相交
    x_range, y_range: (min, max)
    返回 (是否相交, 交点或 None)
    """
    ox, oy, oz = O
    dx, dy, dz = d
    if abs(dz) < eps:
        return False, None
    t = (z_plane - oz) / dz
    x = ox + t * dx
    y = oy + t * dy
    z = oz + t * dz
    if x_range[0] - eps <= x <= x_range[1] + eps and y_range[0] - eps <= y <= y_range[1] + eps:
        return True, [x, y, z]
    return False, None

O = np.array([0.0, 0.0, 2.0])
d = np.array([1.0, 0.7, -1.2])
z_plane = 0.0
x_rng, y_rng = (-0.5, 3.0), (-0.5, 3.0)
hit, P = line_horizontal_plane_rect(O, d, z_plane, x_rng, y_rng)
print("与水平面矩形区域相交：", hit, P)

fig = plt.figure()
ax = fig.add_subplot(111, projection="3d")
xs, ys = np.meshgrid(np.linspace(*x_rng, 10), np.linspace(*y_rng, 10))
zs = np.full_like(xs, z_plane)
ax.plot_surface(xs, ys, zs, alpha=0.35, color="C0")
ts = np.linspace(-0.2, 3.0, 80)
L = O.reshape(1, 3) + ts.reshape(-1, 1) * d.reshape(1, 3)
ax.plot(L[:, 0], L[:, 1], L[:, 2], color="gray", linewidth=2, label="Line")
if P:
    ax.scatter([P[0]], [P[1]], [P[2]], color="red", s=60, label="Intersection in region")
ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")
ax.set_title("Line vs. rectangle on horizontal plane")
ax.legend()

plt.show()
```
