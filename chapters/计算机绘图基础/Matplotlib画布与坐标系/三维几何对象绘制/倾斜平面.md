# 倾斜平面

### 倾斜平面1

```python
import numpy as np
from matplotlib import pyplot as plt

# ---------------- 3D坐标系初始化 ----------------
ax = plt.figure().add_subplot(111, projection='3d')

# ---------------- 生成一维采样点 ----------------
x = np.linspace(0, 9, 2)
y = np.linspace(0, 9, 2)

# ---------------- 构造二维网格 ----------------
X, Y = np.meshgrid(x, y)

# print(X)
# print(Y)

# ---------------- 绘制方法 ----------------
# 平面 z=2y 的部分
ax.plot_surface(
    X = X,        # x方向展开
    Y = Y,        # y方向展开
    Z = Y * 2     # z方向由y决定：z = 2y
)

# ---------------- 显示图形 ----------------
plt.show()
```

### 使用曲面的参数方程渲染出1个倾斜平面

```python
import numpy as np
import matplotlib.pyplot as plt

# ---------------- 3D坐标系初始化 ----------------
ax = plt.figure().add_subplot(111, projection='3d')

# ---------------- 参数化平面定义说明 ----------------
# 通过给定三个点的坐标(x0，y0，z0), (d1x，d1y，d1z)和(d2x，d2y，d2z)
# 然后使用平面的参数方程绘出平面上的多个点
# 其中(x0=7，y0=8，z0=9)为平面通过的一点坐标
# (d1x=100，d1y=200，d1z=300)和(d2x=-400，d2y=500，d2z=800)
# 为两条方向向量（矢量）
# 曲面的参数方程如下：
# x = x0 + c1 * d1x + c2 * d2x
# y = y0 + c1 * d1y + c2 * d2y
# z = z0 + c1 * d1z + c2 * d2z

# ---------------- 参数系数（平面展开变量） ----------------
c1 = np.linspace(0, 9, 10)
c2 = np.linspace(0, 9, 10)

# ---------------- 参数化平面生成 ----------------
x = 100000 + c1 * 100 + c2 * (-400)
y = 100 + c1 * 200 + c2 * (500)
z = 900 + c1 * 300 + c2 * (800)

# ---------------- 网格构造（用于曲面绘制） ----------------
X, Y = np.meshgrid(x, y)
print(X)
print(Y)

X, Z = np.meshgrid(x, z)
print(Z)

# ---------------- 绘制方法 ----------------
ax.plot_surface(
    X,
    Y,
    Z
)

# ---------------- 显示图形 ----------------
plt.show()
```

### 使用曲面的参数方程渲染出2个倾斜平面

```python
import numpy as np
import matplotlib.pyplot as plt

# ---------------- 3D坐标系初始化 ----------------
ax = plt.figure().add_subplot(111, projection='3d')

# ---------------- 参数化平面说明（第一组） ----------------
# 通过给定三个点的坐标(x0，y0，z0), (d1x，d1y，d1z)和(d2x，d2y，d2z)
# 然后使用平面的参数方程绘出平面上的多个点
# 其中(x0=7，y0=8，z0=9)为平面通过的一点坐标
# (d1x=100，d1y=200，d1z=300)和(d2x=-400，d2y=500，d2z=800)
# 为两条方向向量（矢量）
# 曲面的参数方程如下：
# x = x0 + c1 * d1x + c2 * d2x
# y = y0 + c1 * d1y + c2 * d2y
# z = z0 + c1 * d1z + c2 * d2z

# ---------------- 第一组参数（正方向展开） ----------------
c1 = np.linspace(0, 90, 10)
c2 = np.linspace(0, 90, 10)

x = 100000 + c1 * 100 + c2 * (-400)
y = 100 + c1 * 200 + c2 * (500)
z = 900 + c1 * 300 + c2 * (800)

# ---------------- 网格构造与数据输出（第一组） ----------------
X, Y = np.meshgrid(x, y)
print(X)
print(Y)

X, Z = np.meshgrid(x, z)
print(Z)

# ---------------- 绘制方法（第一组平面） ----------------
ax.plot_surface(
    X,
    Y,
    Z
)

# ---------------- 参数化平面说明（第二组） ----------------
# 同样采用参数化平面方程，但方向系数取负值，用于生成反向平面

# ---------------- 第二组参数（反方向展开） ----------------
c1 = np.linspace(0, -90, 10)
c2 = np.linspace(0, -90, 10)

x = 10 + c1 * 500 + c2 * (-100)
y = 10 + c1 * 300 + c2 * (400)
z = 90 + c1 * 100 + c2 * (300)

# ---------------- 网格构造与数据输出（第二组） ----------------
X, Y = np.meshgrid(x, y)
print(X)
print(Y)

X, Z = np.meshgrid(x, z)
print(Z)

# ---------------- 绘制方法（第二组平面） ----------------
ax.plot_surface(
    X,
    Y,
    Z
)

# ---------------- 显示图形 ----------------
plt.show()
```
