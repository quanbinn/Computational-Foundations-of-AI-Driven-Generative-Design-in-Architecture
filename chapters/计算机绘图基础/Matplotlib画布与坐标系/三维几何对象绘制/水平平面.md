# 水平平面

```python
import numpy as np
from matplotlib import pyplot as plt

# ---------------- 3D坐标系初始化 ----------------
ax = plt.figure().add_subplot(111, projection='3d')

# ---------------- 生成网格点 ----------------
x = np.linspace(0, 9, 2)
y = np.linspace(0, 9, 2)

# 通过meshgrid生成二维网格
X, Y = np.meshgrid(x, y)

print(X)
print(Y)

# ---------------- 构造水平面高度 ----------------
Z = X * 0 + 1.5
print(Z)

# ---------------- 绘制方法 ----------------
# 绘出1.5(米)的水平面(平均身高男性眼眼睛瞳孔的高度)
ax.plot_surface(
    X,
    Y,
    Z = X * 0 + 1.5
)

# ---------------- 显示图形 ----------------
plt.show()
```
