# 网格点约束放置

### 基于随机网格点为底面中心的单个长方体生成方法

```python
import matplotlib.pyplot as plt
import random

# ---------------- 3D坐标系初始化 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.set(xlim=(0, 300), ylim=(0, 300), zlim=(0, 300))

# ---------------- 场地参数 ----------------
length = 240
width = 200

site = [
    [0, 0, 0],
    [length, 0, 0],
    [length, width, 0],
    [0, width, 0]
]

# ---------------- 绘制方法：场地闭合结构 ----------------
# 通过依次连接相邻顶点，并使用取模实现首尾闭合
for i in range(4):
    start = site[i]                         # 当前点
    end = site[(i + 1) % 4]                 # 下一个点（自动闭合）

    ax.plot([start[0], end[0]],             # x坐标
            [start[1], end[1]],             # y坐标
            [start[2], end[2]],             # z坐标
            color="green")                 # 边界颜色

# ---------------- 网格系统生成 ----------------
distance_in_length = 40
distance_in_width = 40

segments_in_length = int(length / distance_in_length)   # x方向分段数
segments_in_width = int(width / distance_in_width)     # y方向分段数

# ---------------- 生成x方向网格坐标 ----------------
xArrayInLength = []
for i in range(1, segments_in_length):
    xArrayInLength.append(distance_in_length * i)

# ---------------- 生成y方向网格坐标 ----------------
yArrayInWidth = []
for i in range(1, segments_in_width):
    yArrayInWidth.append(distance_in_width * i)

# ---------------- 生成网格交点 ----------------
intersections = []
for j in range(1, segments_in_width):
    intersections.append([])

numberOfFloor = 0   # 当前楼层（z值）

# 构建二维网格点矩阵 intersections[j][i] = [x, y, z]
for j in range(0, segments_in_width - 1):
    for i in range(1, segments_in_length):
        intersections[j].append([
            xArrayInLength[i - 1],
            yArrayInWidth[j],
            numberOfFloor
        ])

# ---------------- 绘制方法：网格点 ----------------
for j in range(len(intersections)):
    for i in range(len(intersections[j])):
        ax.scatter(intersections[j][i][0],   # x
                   intersections[j][i][1],   # y
                   intersections[j][i][2],   # z
                   c="red")                 # 点颜色

# ---------------- Cuboid类 ----------------
class Cuboid:
    def __init__(self, x, y, z, length, width, height):
        self.x = x
        self.y = y
        self.z = z        
        self.length = length
        self.width = width   
        self.height = height

    def getx(self): return self.x
    def gety(self): return self.y
    def getz(self): return self.z    
    def getlength(self): return self.length
    def getwidth(self): return self.width
    def getheight(self): return self.height

    def areaInBase(self):
        return self.length * self.width          # 底面积

    def volume(self):
        return self.length * self.width * self.height   # 体积        

    def coordinates_of_base_center(self):
        x = self.x + self.length / 2
        y = self.y + self.width / 2
        z = self.z
        return [x, y, z]                        # 底面中心点

    # ---------------- 绘制方法 ----------------
    def render(self):
        ax.bar3d(self.x, self.y, self.z,
                 self.length, self.width, self.height,
                 color="green", alpha=0.5)      # 半透明体块

# ---------------- 随机生成Cuboid ----------------
length_c = 24
width_c = 20
height_c = 4

# 从网格点中随机选取一个位置
xIndex = random.randint(0, len(intersections) - 1)
yIndex = random.randint(0, len(intersections[xIndex]) - 1)

x = intersections[xIndex][yIndex][0]
y = intersections[xIndex][yIndex][1]
z = 1   # 略微抬高，避免与地面重合

cuboid1 = Cuboid(x, y, z, length_c, width_c, height_c)
cuboid1.render()

# ---------------- 显示 ----------------
plt.show()
```
