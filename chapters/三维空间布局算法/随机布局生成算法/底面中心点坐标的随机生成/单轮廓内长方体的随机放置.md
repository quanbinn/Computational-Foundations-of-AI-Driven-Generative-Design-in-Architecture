### 创建长方形基地轮廓
```python
import matplotlib.pyplot as plt

# ---------------- 创建3D图形并设置坐标轴范围 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.set(xlim=(0, 300), ylim=(0, 300), zlim=(0, 300))

# ---------------- 定义场地尺寸参数 ----------------
length, width = 240, 200    # 给长方形场地的长度和宽度赋值

# ---------------- 定义场地四个角点坐标 ----------------
site = [[0,0,0],[length,0,0],[length,width,0],[0,width,0]]     # 给场地四个角点坐标赋值（z=0）

# ---------------- 绘制方法：边界线段连接 ----------------
# 绘制闭合的场地边界(绿色)
# 对于点A(x1,y1,z1)和点B(x2,y2,z2)的连线,代码表达为ax.plot([x1,x2],[y1,y2],[z1,z2])
for i in range(4):
    start = site[i]                  # 当前起点
    end = site[(i + 1) % 4]          # 下一个点，最后回到起点（利用取模实现闭合）

    ax.plot([start[0], end[0]],      # x坐标
            [start[1], end[1]],      # y坐标
            [start[2], end[2]],      # z坐标
            color="green")           # 线颜色

# ---------------- 显示图形 ----------------
plt.show()
```

```python
import matplotlib.pyplot as plt
import random

# ---------------- 3D坐标系初始化 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.set_xlim((0, 300))
ax.set_ylim((0, 300))
ax.set_zlim((0, 300))

# ---------------- 场地类（二维/三维边界） ----------------
class Site:
    def __init__(self, data_of_points):
        self.data = data_of_points   # 多边形顶点数据（按顺序存储）

    def getData(self): 
        return self.data             # 返回全部顶点数据

    def edges(self): 
        return len(self.data)        # 边数 = 顶点数

    # ---------------- 绘制方法 ----------------
    def render(self):
        data = self.data

        # ---------------- 闭合多边形绘制 ----------------
        for i in range(len(data)):
            start = data[i]                          # 当前点
            end = data[(i + 1) % len(data)]          # 下一个点（取模实现首尾闭合）

            ax.plot([start[0], end[0]],              # x坐标
                    [start[1], end[1]],              # y坐标
                    [start[2], end[2]],              # z坐标
                    color="green")                  # 线颜色

# ---------------- 场地定义 ----------------
length = 240
width = 200

site = [
    [0, 0, 0],
    [length, 0, 0],
    [length, width, 0],
    [0, width, 0]
]

offset = 10   # 红线内缩距离（控制可建设范围）

site_of_property_line = [
    [0 + offset, 0 + offset, 0],
    [length - offset, 0 + offset, 0],
    [length - offset, width - offset, 0],
    [0 + offset, width - offset, 0]
]

# ---------------- 绘制场地 ----------------
s = Site(site)
s.render()     # 绘制外轮廓

s1 = Site(site_of_property_line)
s1.render()    # 绘制红线内边界

# ---------------- 长方体类 ----------------
class Cuboid:
    def __init__(self, x, y, z, length, width, height):
        self.x = x
        self.y = y
        self.z = z        
        self.length = length
        self.width = width   
        self.height = height

    def getx(self): return self.x
    def gety(self): return self.y
    def getz(self): return self.z    
    def getlength(self): return self.length
    def getwidth(self): return self.width
    def getheight(self): return self.height

    def areaInBase(self):
        return self.length * self.width     # 底面积

    def volume(self):
        return self.length * self.width * self.height   # 体积        

    def coordinates_of_base_center(self):
        x = self.x + self.length / 2
        y = self.y + self.width / 2
        z = self.z
        return [x, y, z]                    # 返回底面中心点坐标

    # ---------------- 绘制方法 ----------------
    def render(self):
        ax.bar3d(self.x, self.y, self.z,
                 self.length, self.width, self.height,
                 color="green", alpha=0.5)  # 半透明显示体块

# ---------------- 随机生成一个立方体 ----------------
length_of_cuboid = 80
width_of_cuboid = 60
height_of_cuboid = 100

# 在红线范围内随机生成左下角点（保证不越界）
x_of_cuboid = random.randint(0 + offset, length - offset - length_of_cuboid)
y_of_cuboid = random.randint(0 + offset, width - offset - width_of_cuboid)
z_of_cuboid = 0

cuboid1 = Cuboid(x_of_cuboid, y_of_cuboid, z_of_cuboid,
                 length_of_cuboid, width_of_cuboid, height_of_cuboid)

cuboid1.render()

# ---------------- 显示结果 ----------------
plt.show()
```

### 单轮廓内两个长方体的随机放置

import matplotlib.pyplot as plt
import random

# ---------------- 3D坐标系初始化 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.set(xlim=(0, 300), ylim=(0, 300), zlim=(0, 300))

# ---------------- 场地类 ----------------
class Site:
    def __init__(self, data_of_points):
        self.data = data_of_points            # 按顺序存储的多边形顶点

    def getData(self): 
        return self.data                      # 返回顶点数据

    def edges(self): 
        return len(self.data)                 # 边数 = 顶点数

    # ---------------- 绘制方法（闭合多边形） ----------------
    def render(self):
        data = self.data

        # ---------------- 场地闭合绘制 ----------------
        for i in range(len(data)):
            start = data[i]                  # 当前顶点
            end = data[(i + 1) % len(data)]  # 下一个顶点（取模实现闭合）

            ax.plot([start[0], end[0]],      # x坐标
                    [start[1], end[1]],      # y坐标
                    [start[2], end[2]],      # z坐标
                    color="green")          # 边界颜色

# ---------------- 场地数据 ----------------
length = 240
width = 200

site = [
    [0, 0, 0],
    [length, 0, 0],
    [length, width, 0],
    [0, width, 0]
]

offset = 10   # 红线内缩距离（控制可建设范围）

site_of_property_line = [
    [0 + offset, 0 + offset, 0],
    [length - offset, 0 + offset, 0],
    [length - offset, width - offset, 0],
    [0 + offset, width - offset, 0]

# ---------------- 绘制场地 ----------------
s = Site(site)
s.render()     # 外轮廓

s1 = Site(site_of_property_line)
s1.render()    # 红线范围

# ---------------- 长方体类 ----------------
class Cuboid:
    def __init__(self, x, y, z, length, width, height):
        self.x = x
        self.y = y
        self.z = z        
        self.length = length
        self.width = width   
        self.height = height

    def getx(self): return self.x
    def gety(self): return self.y
    def getz(self): return self.z    
    def getlength(self): return self.length
    def getwidth(self): return self.width
    def getheight(self): return self.height

    def areaInBase(self):
        return self.length * self.width        # 底面积

    def volume(self):
        return self.length * self.width * self.height  # 体积        

    def coordinates_of_base_center(self):
        x = self.x + self.length / 2
        y = self.y + self.width / 2
        z = self.z
        return [x, y, z]                      # 底面中心点

    # ---------------- 绘制方法 ----------------
    def render(self):
        ax.bar3d(self.x, self.y, self.z,
                 self.length, self.width, self.height,
                 color="green", alpha=0.5)    # 半透明体块

# ---------------- 参数定义 ----------------
length_of_cuboid = 80
width_of_cuboid = 60
height_of_cuboid = 100

# ---------------- 随机生成Cuboid 1 ----------------
# 在红线范围内随机生成左下角点（保证不越界）
x_of_cuboid = random.randint(0 + offset, length - offset - length_of_cuboid)
y_of_cuboid = random.randint(0 + offset, width - offset - width_of_cuboid)
z_of_cuboid = 0

cuboid1 = Cuboid(x_of_cuboid, y_of_cuboid, z_of_cuboid,
                 length_of_cuboid, width_of_cuboid, height_of_cuboid)
cuboid1.render()

# ---------------- 随机生成Cuboid 2 ----------------
# 再次随机生成一个位置（可能与第一个重叠，此处不做约束）
x_of_cuboid = random.randint(0 + offset, length - offset - length_of_cuboid)
y_of_cuboid = random.randint(0 + offset, width - offset - width_of_cuboid)
z_of_cuboid = 0

cuboid2 = Cuboid(x_of_cuboid, y_of_cuboid, z_of_cuboid,
