# 两轮廓内两个不相交长方体的随机生成

```python
import matplotlib.pyplot as plt
import numpy as np 
import random

# ---------------- 3D坐标系初始化 ----------------
ax = plt.figure().add_subplot(111, projection='3d')
ax.set_xlim((0, 250))
ax.set_ylim((0, 250))
ax.set_zlim((0, 200))

# ---------------- 场地参数 ----------------
length = 240
width = 200

# ---------------- 两层轮廓（z = 0 / 100） ----------------
z_levels = [0, 100]

# 不同层颜色（教学用）
colors = ["green", "blue"]

sites = []
for z in z_levels:
    sites.append([[0,0,z],[length,0,z],[length,width,z],[0,width,z]])

# ---------------- 场地边界绘制 ----------------
for k in range(len(sites)):
    site = sites[k]
    color = colors[k]

    ax.plot([site[0][0], site[1][0]], [site[0][1], site[1][1]], [site[0][2], site[1][2]], color=color)  
    ax.plot([site[1][0], site[2][0]], [site[1][1], site[2][1]], [site[1][2], site[2][2]], color=color)
    ax.plot([site[2][0], site[3][0]], [site[2][1], site[3][1]], [site[2][2], site[3][2]], color=color)
    ax.plot([site[3][0], site[0][0]], [site[3][1], site[0][1]], [site[3][2], site[0][2]], color=color)

# ---------------- 网格参数 ----------------
distance_in_length = 20
distance_in_width = 20

segments_in_length = int(length / distance_in_length)
segments_in_width  = int(width / distance_in_width)

# ---------------- 网格X方向 ----------------
xArrayInLength = []
for i in range(1, segments_in_length):
    xArrayInLength.append(distance_in_length * i)

# ---------------- 网格Y方向 ----------------
yArrayInWidth = []
for i in range(1, segments_in_width):
    yArrayInWidth.append(distance_in_width * i)

# ---------------- 网格结构初始化 ----------------
intersections = []
for l in range(1, segments_in_width):
    intersections.append([])

print(intersections)

# ---------------- 网格交点生成（2层） ----------------
for j in range(0, segments_in_width - 1):
    for i in range(1, segments_in_length):
        for z in z_levels:
            intersections[j].append([
                xArrayInLength[i - 1],
                yArrayInWidth[j],
                z
            ])

print(intersections)
print(len(intersections))

# ---------------- 网格点绘制 ----------------
for j in range(0, len(intersections)):
    for i in range(0, len(intersections[j])):

        x, y, z = intersections[j][i]

        # ---------------- 绘制方法 ----------------
        color = colors[z_levels.index(z)]
        ax.scatter(x, y, z, c=color)

# ---------------- Cuboid类 ----------------
class Cuboid:
    def __init__(self, x, y, z, length, width, height):
        self.x = x
        self.y = y
        self.z = z        
        self.length = length
        self.width = width   
        self.height = height

    def getx(self): return self.x
    def gety(self): return self.y
    def getz(self): return self.z    
    def getlength(self): return self.length
    def getwidth(self): return self.width
    def getheight(self): return self.height

    # ---------------- 绘制方法 ----------------
    def render(self):
        color = colors[z_levels.index(self.z)]
        ax.bar3d(self.x, self.y, self.z,
                 self.length, self.width, self.height,
                 color=color, alpha=0.5)

# ---------------- box参数 ----------------
length = 30
width = 40
height = 20

# ---------------- IoU函数 ----------------
def box_iou_xyzxyz(box1, box2):
    x1min, y1min, z1min, x1max, y1max, z1max = box1
    volume1 = (x1max - x1min) * (y1max - y1min) * (z1max - z1min)
    print('长方体1的体积是', volume1, '立方米')

    x2min, y2min, z2min, x2max, y2max, z2max = box2
    volume2 = (x2max - x2min) * (y2max - y2min) * (z2max - z2min)
    print('长方体2的体积是', volume2, '立方米')

    xmin = np.maximum(x1min, x2min)
    xmax = np.minimum(x1max, x2max)
    ymin = np.maximum(y1min, y2min)
    ymax = np.minimum(y1max, y2max)
    zmin = np.maximum(z1min, z2min)
    zmax = np.minimum(z1max, z2max)

    l = np.maximum(xmax - xmin, 0.)
    w = np.maximum(ymax - ymin, 0.)
    h = np.maximum(zmax - zmin, 0.)

    inter = l * w * h
    print('两个长方体相交的体积是', inter, '立方米')

    union = volume1 + volume2 - inter
    print('两个长方体叠加在一起的总体积是', union, '立方米')

    return inter / union

# ---------------- 通用生成函数 ----------------
def generate_non_overlap(existing_boxes):
    while True:
        xIndex = random.randint(0, len(intersections)-1)
        yIndex = random.randint(0, len(intersections[xIndex])-1)

        x, y, z = intersections[xIndex][yIndex]

        cuboid = Cuboid(x, y, z, length, width, height)

        box = [
            cuboid.getx(), cuboid.gety(), cuboid.getz(),
            cuboid.getx() + cuboid.getlength(),
            cuboid.gety() + cuboid.getwidth(),
            cuboid.getz() + cuboid.getheight()
        ]

        print(box)

        valid = True
        for old in existing_boxes:
            iou = box_iou_xyzxyz(box, old)
            print('与已有长方体的交并比是：', iou)
            if iou > 0:
                valid = False
                break

        if valid:
            return cuboid, box

# ---------------- 生成2个长方体 ----------------
cuboid1, box1 = generate_non_overlap([])
cuboid2, box2 = generate_non_overlap([box1])

# ---------------- 渲染 ----------------
cuboid1.render()
cuboid2.render()

plt.show()
```
