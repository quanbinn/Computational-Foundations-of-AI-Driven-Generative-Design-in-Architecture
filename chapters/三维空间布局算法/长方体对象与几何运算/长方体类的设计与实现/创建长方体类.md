# 创建长方体类

### 定义带有构造函数的类

```python
class Cuboid:
    """长方体类：存储位置与尺寸信息"""

    def __init__(self, x, y, z, length, width, height):
        """
        初始化长方体

        参数：
        x, y, z : 左下角前点坐标
        length  : 长（X方向）
        width   : 宽（Y方向）
        height  : 高（Z方向）
        """
        self.x = x
        self.y = y
        self.z = z        
        self.length = length
        self.width = width   
        self.height = height

    # ---------------- 属性访问方法 ----------------
    def getX(self): return self.x
    def getY(self): return self.y
    def getZ(self): return self.z    
    def getLength(self): return self.length
    def getWidth(self): return self.width
    def getHeight(self): return self.height

# ---------------- 使用示例（创建长方体对象） ----------------
cuboid1 = Cuboid(20, 30, 8, 16, 8, 4)

# ---------------- 属性输出 ----------------
print(cuboid1.getX())
print(cuboid1.getY())
print(cuboid1.getZ())
print(cuboid1.getLength())
print(cuboid1.getWidth())
print(cuboid1.getHeight())
```

###  **在类中增加底面积、体积与底面中心点计算方法**

![](/images/三维空间布局算法/长方体操作/创建长方体类/创建长方体类/2a1.jpg)

```python
class Cuboid:
    """长方体类：描述位置、尺寸及基本几何计算"""

    def __init__(self, x, y, z, length, width, height):
        """
        初始化长方体
        x, y, z：左下角前点坐标
        length, width, height：长、宽、高
        """
        self.x = x
        self.y = y
        self.z = z        
        self.length = length
        self.width = width   
        self.height = height

    # ---------------- 属性访问 ----------------
    def getX(self): return self.x
    def getY(self): return self.y
    def getZ(self): return self.z    
    def getLength(self): return self.length
    def getWidth(self): return self.width
    def getHeight(self): return self.height

    # ---------------- 几何计算 ----------------
    def baseArea(self):
        """计算底面积"""
        return self.length * self.width

    def volume(self):
        """计算体积"""
        return self.length * self.width * self.height        

    def baseCenter(self):
        """计算底面中心点坐标"""
        x = self.x + self.length / 2
        y = self.y + self.width / 2
        z = self.z
        return [x, y, z]

# ---------------- 使用示例（长方体对象） ----------------
cuboid1 = Cuboid(20, 30, 8, 16, 8, 4)

# ---------------- 输出结果 ----------------
print(cuboid1.getWidth())
print(cuboid1.baseArea())
print(cuboid1.volume())
print(cuboid1.baseCenter())
```

### 在类中增加长方体六面渲染方法

![](/images/三维空间布局算法/长方体操作/创建长方体类/创建长方体类/4a1.jpg)

```python
import matplotlib.pyplot as plt

# ---------- 创建三维坐标系 ----------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

# 设置坐标轴范围
ax.set_xlim(0, 100)
ax.set_ylim(0, 100)
ax.set_zlim(0, 100)

class Cuboid:
    """长方体类：描述位置、尺寸及绘制方法"""

    def __init__(self, x, y, z, length, width, height):
        """
        初始化长方体
        x, y, z：左下角前点坐标
        length, width, height：长、宽、高
        """
        self.x = x
        self.y = y
        self.z = z        
        self.length = length
        self.width = width   
        self.height = height
    
    # ---------- 属性访问 ----------
    def getX(self): return self.x
    def getY(self): return self.y
    def getZ(self): return self.z    
    def getLength(self): return self.length
    def getWidth(self): return self.width
    def getHeight(self): return self.height

    # ---------- 几何计算 ----------
    def baseArea(self):
        """计算底面积"""
        return self.length * self.width

    def volume(self):
        """计算体积"""
        return self.length * self.width * self.height        

    def baseCenter(self):
        """计算底面中心点"""
        x = self.x + self.length / 2
        y = self.y + self.width / 2
        z = self.z
        return [x, y, z]
        
    # ---------- 绘制 ----------
    def render(self, color="lightblue", alpha=0.5):
        """
        绘制长方体
        color：颜色（默认浅蓝）
        alpha：透明度（0~1）
        """
        ax.bar3d(self.x, self.y, self.z,
                 self.length, self.width, self.height,
                 color=color, alpha=alpha)

# ---------------- 使用示例（创建长方体对象） ----------------
cuboid1 = Cuboid(20, 30, 8, 16, 8, 4)
cuboid1.render()  # 默认浅蓝色

cuboid2 = Cuboid(60, 70, 0, 16, 16, 8)
cuboid2.render(color="orange")  # 指定不同颜色

plt.show()
```
