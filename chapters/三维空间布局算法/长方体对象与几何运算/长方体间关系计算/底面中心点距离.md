# 两个长方体的底面中心点距离

```python
import matplotlib.pyplot as plt
import numpy as np
import math

# ---------------- 创建3D绘图坐标系 ----------------
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.set_xlim((0, 50))
ax.set_ylim((0, 50))
ax.set_zlim((0, 50))

# ---------------- 3D长方体类定义 ----------------
class Cuboid:
    """长方体类"""

    def __init__(self, x, y, z, length, width, height):
        """
        初始化长方体
        x, y, z - 底面左下角坐标
        length, width, height - 尺寸
        """
        self.x = x
        self.y = y
        self.z = z        
        self.length = length
        self.width = width   
        self.height = height

    # ---------------- 几何方法：底面中心 ----------------
    def base_center(self):
        x_center = self.x + self.length / 2
        y_center = self.y + self.width / 2
        z_center = self.z
        return [x_center, y_center, z_center]

    # ---------------- 绘制方法 ----------------
    def render(self):
        ax.bar3d(self.x, self.y, self.z,
                 self.length, self.width, self.height,
                 color="green", alpha=0.5)

# ---------------- 距离计算方法 ----------------
def distance_between_base_centers(cuboid1, cuboid2):
    """
    计算两个立方体底面中心点的水平距离（忽略高度 z）
    """
    x1, y1, _ = cuboid1.base_center()
    x2, y2, _ = cuboid2.base_center()

    distance = math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)
    return distance

# ---------------- 示例：创建并绘制长方体 ----------------
cuboid1 = Cuboid(10, 20, 8, 16, 8, 16)
cuboid1.render()

cuboid2 = Cuboid(15, 20, 4, 16, 16, 24)
cuboid2.render()

# ---------------- 计算并输出距离 ----------------
distance = distance_between_base_centers(cuboid1, cuboid2)
print("两个立方体底面中心点的水平距离：", distance)

plt.show()
```
