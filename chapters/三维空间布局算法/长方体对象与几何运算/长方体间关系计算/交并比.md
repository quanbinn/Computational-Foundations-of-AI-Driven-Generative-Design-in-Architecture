# 交并比

```python
import matplotlib.pyplot as plt
import numpy as np

# 创建3D绘图坐标系
ax = plt.figure().add_subplot(111, projection='3d')
ax.set_xlim((0, 50))
ax.set_ylim((0, 50))
ax.set_zlim((0, 50))

class Cuboid:
    """长方体类"""
    def __init__(self, x, y, z, length, width, height):
        """
        初始化长方体
        参数：
        x, y, z - 长方体底面左下角坐标
        length, width, height - 长方体尺寸
        """
        self.x = x
        self.y = y
        self.z = z        
        self.length = length
        self.width = width   
        self.height = height

    # 获取属性方法
    def getx(self): return self.x
    def gety(self): return self.y
    def getz(self): return self.z    
    def getlength(self): return self.length
    def getwidth(self): return self.width
    def getheight(self): return self.height

    # 计算底面面积
    def areaInBase(self):
        return self.length * self.width

    # 计算体积
    def volume(self):
        return self.length * self.width * self.height        

    # 获取底面中心坐标
    def coordinates_of_base_center(self):
        x_center = self.x + self.length / 2
        y_center = self.y + self.width / 2
        z_center = self.z
        return [x_center, y_center, z_center]

    # 绘制长方体
    def render(self):
        ax.bar3d(self.x, self.y, self.z,
                 self.length, self.width, self.height,
                 color="green", alpha=0.5)

# 创建两个长方体并绘制
cuboid1 = Cuboid(10, 20, 8, 16, 8, 16)
cuboid1.render()

cuboid2 = Cuboid(15, 20, 4, 16, 16, 24)
cuboid2.render()

def box_iou_xyzxyz(box1, box2):
    """
    计算两个长方体的交并比(IOU)
    输入：
    box1, box2 - 每个长方体的[x_min, y_min, z_min, x_max, y_max, z_max]
    输出：
    iou - 两个长方体的交并比
    """
    # 分别提取每个长方体的坐标
    x1min, y1min, z1min, x1max, y1max, z1max = box1
    x2min, y2min, z2min, x2max, y2max, z2max = box2

    # 计算体积
    volume1 = (x1max - x1min) * (y1max - y1min) * (z1max - z1min)
    volume2 = (x2max - x2min) * (y2max - y2min) * (z2max - z2min)
    print('长方体1的体积是', volume1, '立方米')
    print('长方体2的体积是', volume2, '立方米')

    # 计算交集的坐标范围
    xmin = np.maximum(x1min, x2min)
    xmax = np.minimum(x1max, x2max)
    ymin = np.maximum(y1min, y2min)
    ymax = np.minimum(y1max, y2max)
    zmin = np.maximum(z1min, z2min)
    zmax = np.minimum(z1max, z2max)

    # 计算交集长宽高
    length_of_intersection = np.maximum(xmax - xmin, 0.)
    width_of_intersection = np.maximum(ymax - ymin, 0.)
    height_of_intersection = np.maximum(zmax - zmin, 0.)

    # 计算交集体积
    volume_of_intersection = length_of_intersection * width_of_intersection * height_of_intersection
    print('两个长方体相交的体积是', volume_of_intersection, '立方米')   

    # 计算并集体积
    volume_of_union = volume1 + volume2 - volume_of_intersection
    print('两个长方体叠加在一起的总体积是', volume_of_union, '立方米')   

    # 计算IOU
    iou = volume_of_intersection / volume_of_union
    return iou

# 构建长方体坐标列表用于IOU计算
box1 = [cuboid1.getx(), cuboid1.gety(), cuboid1.getz(),
        cuboid1.getx() + cuboid1.getlength(),
        cuboid1.gety() + cuboid1.getwidth(),
        cuboid1.getz() + cuboid1.getheight()]

box2 = [cuboid2.getx(), cuboid2.gety(), cuboid2.getz(),
        cuboid2.getx() + cuboid2.getlength(),
        cuboid2.gety() + cuboid2.getwidth(),
        cuboid2.getz() + cuboid2.getheight()]

# 计算交并比
iou = box_iou_xyzxyz(box1, box2)
print('两个长方体的交并比是：', iou)

plt.show()
```
