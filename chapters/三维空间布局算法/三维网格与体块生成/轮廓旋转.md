# 轮廓旋转

### 基于数量积的向量夹角计算

```python
import math

# ---------------- 角度说明 ----------------
# 真实的角度要考虑象限和顺时针/逆时针方向影响，例如：+-degree
# 当前方法：点积 + 反余弦，只能得到 0~180°（无方向）

def angle(v1, v2):
    """
    计算两个三维向量之间的夹角（单位：度）

    方法：
        cosθ = (v1 · v2) / (|v1| |v2|)
        θ = arccos(cosθ)
    """

    # ---------------- 点积（Dot Product） ----------------
    # v1 · v2 = x1*x2 + y1*y2 + z1*z2
    dot_product = (
        v1[0] * v2[0] +
        v1[1] * v2[1] +
        v1[2] * v2[2]
    )

    # ---------------- 向量模长（Magnitude） ----------------
    magnitude_of_v1 = math.sqrt(
        v1[0]**2 + v1[1]**2 + v1[2]**2
    )

    magnitude_of_v2 = math.sqrt(
        v2[0]**2 + v2[1]**2 + v2[2]**2
    )

    # ---------------- 计算 cosθ ----------------
    value_of_cosine = dot_product / (magnitude_of_v1 * magnitude_of_v2)

    # ---------------- 反余弦（弧度） ----------------
    pi_of_angle = math.acos(value_of_cosine)

    # ---------------- 转换为角度制 ----------------
    angle_value = pi_of_angle * 180 / math.pi

    return angle_value

# ---------------- 测试 ----------------
print(angle([235, 47, 0], [1, 0, 0]))
```

### 单点旋转的几何变换过程

```python
import math

# ---------------- 旋转角度说明 ----------------
# 原代码 angle = -11.31（默认角度制）
# 必须转换为弧度，否则 cos/sin 计算错误
angle = math.radians(-11.31)

# ---------------- Z轴旋转矩阵 ----------------
matrix3dRotation = [
    [
        round(math.cos(angle), 2),   # cosθ
        round(-math.sin(angle), 2),  # -sinθ
        0
    ],
    [
        round(math.sin(angle), 2),   # sinθ
        round(math.cos(angle), 2),   # cosθ
        0
    ],
    [
        0,
        0,
        1   # z方向保持不变（齐次）
    ]
]

print("Rotation Matrix:")
print(matrix3dRotation)


def rotate(matrix, angle):
    """
    将一个点（或向量）进行旋转变换

    本质：
        new_point = R × point
    """

    # ---------------- 初始化结果向量 ----------------
    new_matrix = [0, 0, 0]

    # ---------------- 矩阵乘法（展开计算） ----------------
    # new_x = x*cosθ + y*(-sinθ)
    new_matrix[0] = round(
        matrix[0][0] * matrix3dRotation[0][0] +
        matrix[0][1] * matrix3dRotation[0][1] +
        matrix[0][2] * matrix3dRotation[0][2],
        2
    )

    # new_y = x*sinθ + y*cosθ
    new_matrix[1] = round(
        matrix[0][0] * matrix3dRotation[1][0] +
        matrix[0][1] * matrix3dRotation[1][1] +
        matrix[0][2] * matrix3dRotation[1][2],
        2
    )

    # new_z = z保持不变
    new_matrix[2] = round(
        matrix[0][0] * matrix3dRotation[2][0] +
        matrix[0][1] * matrix3dRotation[2][1] +
        matrix[0][2] * matrix3dRotation[2][2],
        2
    )

    print("Rotated Vector:")
    print(new_matrix)

    return new_matrix

# ---------------- 测试数据 ----------------
matrix = [[235, 47, 0]]

print(rotate(matrix, angle))
```

```python
        0

        round(math.sin(angle), 2),
        round(math.cos(angle), 2),
        0
    ],
    [
        0,
        0,
        1
    ]

def rotate(matrix, angle):

    # 初始化结果矩阵（保持结构一致）
    new_matrix = [[0, 0, 0] for _ in range(len(matrix))]

    print("Input Matrix:")
    print(matrix)

    # ---------------- 遍历每个点 ----------------
    for i in range(len(matrix)):

        # ---------------- x'计算 ----------------
        new_matrix[i][0] = round(
            matrix[i][0] * matrix3dRotation[0][0] +
            matrix[i][1] * matrix3dRotation[0][1] +
            matrix[i][2] * matrix3dRotation[0][2],
            2
        )

        # ---------------- y'计算 ----------------
        new_matrix[i][1] = round(
            matrix[i][0] * matrix3dRotation[1][0] +
            matrix[i][1] * matrix3dRotation[1][1] +
            matrix[i][2] * matrix3dRotation[1][2],
            2
        )

        # ---------------- z'计算（不变） ----------------
        new_matrix[i][2] = round(
            matrix[i][0] * matrix3dRotation[2][0] +
            matrix[i][1] * matrix3dRotation[2][1] +
```

### 展示旋转变换后的图形效果

```python
import matplotlib.pyplot as plt

# ---------------- 创建3D坐标系 ----------------
ax = plt.figure().add_subplot(111, projection='3d')

# 固定观察范围（教学用）
ax.set_xlim((0, 300))
ax.set_ylim((0, 300))
ax.set_zlim((0, 300))

class Site:
    def __init__(self, data_of_points):
        # data_of_points: [[x,y,z], ...]
        self.data = data_of_points

    def getData(self):
        # 返回当前点集
        return self.data

    def edges(self):
        # 返回点数量（边数）
        return len(self.data)

    def move(self, data, p1, p2):
        """
        三维平移变换（Translation）

        本质：
            所有点统一加上同一个向量 (p2 - p1)
        """

        # ---------------- 备份当前状态 ----------------
        site = Site(data).getData()

        # 点数量
        length = len(self.data)

        # ---------------- 计算平移向量 ----------------
        xdiff = p2[0] - p1[0]
        ydiff = p2[1] - p1[1]
        zdiff = p2[2] - p1[2]

        # ---------------- 执行平移变换 ----------------
        for i in range(length):

            data[i][0] = data[i][0] + xdiff
            data[i][1] = data[i][1] + ydiff
            data[i][2] = data[i][2] + zdiff

        # ---------------- 绘制变换后结果 ----------------
        Site(site).render2()

    def render1(self):
        """
        绘制原始图形（绿色）
        ——表示变换前状态
        """

        data = self.data

        # ---------------- 绘制方法（原始图形） ----------------
        ax.plot([data[0][0], data[1][0]],
                [data[0][1], data[1][1]],
                [data[0][2], data[1][2]],
                color="green")

        ax.plot([data[1][0], data[2][0]],
                [data[1][1], data[2][1]],
                [data[1][2], data[2][2]],
                color="green")

        ax.plot([data[2][0], data[3][0]],
                [data[2][1], data[3][1]],
                [data[2][2], data[3][2]],
                color="green")

        ax.plot([data[3][0], data[0][0]],
                [data[3][1], data[0][1]],
                [data[3][2], data[0][2]],
                color="green")

    def render2(self):
        """
        绘制变换后图形（红色）
        ——表示移动后的状态
        """

        data = self.data

        # ---------------- 绘制方法（变换后图形） ----------------
        ax.plot([data[0][0], data[1][0]],
                [data[0][1], data[1][1]],
                [data[0][2], data[1][2]],
                color="red")

        ax.plot([data[1][0], data[2][0]],
                [data[1][1], data[2][1]],
                [data[1][2], data[2][2]],
                color="red")

        ax.plot([data[2][0], data[3][0]],
                [data[2][1], data[3][1]],
                [data[2][2], data[3][2]],
                color="red")

        ax.plot([data[3][0], data[0][0]],
                [data[3][1], data[0][1]],
                [data[3][2], data[0][2]],
                color="red")

# ---------------- 测试1：不规则四边形 ----------------
data = [
```
