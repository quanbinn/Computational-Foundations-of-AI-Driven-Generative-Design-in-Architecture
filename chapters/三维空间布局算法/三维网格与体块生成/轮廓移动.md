# 轮廓移动

### 基于参考向量的三维点集整体平移

```python
def move(data, p1, p2):
    """
    三维点集的整体平移

    含义：
        将 data 中的所有点按照向量 (p1 -> p2) 进行平移
    """

    # 点的数量
    length = len(data)

    # =========================
    # 计算三维平移向量
    # =========================
    # x方向偏移
    xdiff = p2[0] - p1[0]

    # y方向偏移
    ydiff = p2[1] - p1[1]

    # z方向偏移
    zdiff = p2[2] - p1[2]

    # =========================
    # 对每一个点进行平移
    # =========================
    for i in range(0, length):

        # x坐标平移
        data[i][0] = data[i][0] + xdiff

        # y坐标平移
        data[i][1] = data[i][1] + ydiff

        # z坐标平移
        data[i][2] = data[i][2] + zdiff

    # 返回平移后的点集（原地修改）
    return data


# =========================
# 测试数据（三维点）
# =========================
data = [
    [50, 30, 0],
    [285, 77, 0],
    [246, 273, 0],
    [10, 226, 0]
]

# 起点（参考点）
p1 = [50, 30, 0]

# 终点（目标点）
p2 = [0, 0, 0]

# 执行平移
new_data = move(data, p1, p2)

# 输出结果
print(new_data)
```
### 基于参考向量的三维多边形平移与可视化

import matplotlib.pyplot as plt

# =========================
# 创建 3D 坐标系
# =========================
ax = plt.figure().add_subplot(111, projection='3d')

# 设置坐标轴范围（教学用固定空间）
ax.set_xlim((0, 300))
ax.set_ylim((0, 300))
ax.set_zlim((0, 300))


class Site:
    def __init__(self, data_of_points):
        # data_of_points: [[x,y,z], [x,y,z], ...]
        self.data = data_of_points

    def getData(self):
        # 返回当前点集
        return self.data

    def edges(self):
        # 返回点的数量
        return len(self.data)

    def move(self, data, p1, p2):
        """
        三维空间中的平移变换

        含义：
            将所有点按照 (p1 -> p2) 的向量进行移动
        """

        # 重新构造数据（保持你的原写法）
        site = Site(data).getData()

        # 点数量
        length = len(self.data)

        # =========================
        # 计算三维平移向量
        # =========================
        xdiff = p2[0] - p1[0]
        ydiff = p2[1] - p1[1]
        zdiff = p2[2] - p1[2]

        # =========================
        # 对所有点进行平移
        # =========================
        for i in range(0, length):

            # x方向平移
            data[i][0] = data[i][0] + xdiff

            # y方向平移
            data[i][1] = data[i][1] + ydiff

            # z方向平移
            data[i][2] = data[i][2] + zdiff

        # 绘制平移后的图形
        Site(site).render()

    def render(self):
        """
        三维多边形绘制（硬编码四边形）
        """

        data = self.data

        # =========================
        # 逐条边绘制（闭合四边形）
        # =========================

        # 边1
        ax.plot(
            [data[0][0], data[1][0]],
            [data[0][1], data[1][1]],
            [data[0][2], data[1][2]],
            color="green"
        )

        # 边2
        ax.plot(
            [data[1][0], data[2][0]],
            [data[1][1], data[2][1]],
            [data[1][2], data[2][2]],
            color="green"
        )

        # 边3
        ax.plot(
            [data[2][0], data[3][0]],
            [data[2][1], data[3][1]],
            [data[2][2], data[3][2]],
            color="green"
        )

        # 边4（闭合）
        ax.plot(
            [data[3][0], data[0][0]],
            [data[3][1], data[0][1]],
            [data[3][2], data[0][2]],
            color="green"
        )


# =========================
# 测试数据（四点空间多边形）
# =========================
data = [
    [50, 30, 0],

    [246, 273, 0],
