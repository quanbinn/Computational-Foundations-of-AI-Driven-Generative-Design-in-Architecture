# 轮廓移动

### 基于参考向量的三维点集整体平移

```python
def move(data, p1, p2):
    """
    三维点集的整体平移

    含义：
        将 data 中的所有点按照向量 (p1 -> p2) 进行平移
    """

    # ---------------- 点的数量 ----------------
    length = len(data)

    # ---------------- 计算三维平移向量 ----------------
    xdiff = p2[0] - p1[0]
    ydiff = p2[1] - p1[1]
    zdiff = p2[2] - p1[2]

    # ---------------- 执行平移变换 ----------------
    for i in range(length):

        # x坐标平移
        data[i][0] = data[i][0] + xdiff

        # y坐标平移
        data[i][1] = data[i][1] + ydiff

        # z坐标平移
        data[i][2] = data[i][2] + zdiff

    return data

# ---------------- 测试数据（三维点） ----------------
data = [
    [50, 30, 0],
    [285, 77, 0],
    [246, 273, 0],
    [10, 226, 0]
]

p1 = [50, 30, 0]
p2 = [0, 0, 0]

# ---------------- 执行平移 ----------------
new_data = move(data, p1, p2)

print(new_data)
```
### 基于参考向量的三维多边形平移与可视化

```python
import matplotlib.pyplot as plt

# ---------------- 创建3D坐标系 ----------------
ax = plt.figure().add_subplot(111, projection='3d')

# 设置坐标轴范围（教学用固定空间）
ax.set_xlim((0, 300))
ax.set_ylim((0, 300))
ax.set_zlim((0, 300))

class Site:
    def __init__(self, data_of_points):
        # data_of_points: [[x,y,z], [x,y,z], ...]
        self.data = data_of_points

    def getData(self):
        # 返回当前点集
        return self.data

    def edges(self):
        # 返回点的数量
        return len(self.data)

    def move(self, data, p1, p2):
        """
        三维空间中的平移变换

        含义：
            将所有点按照 (p1 -> p2) 的向量进行移动
        """

        # ---------------- 备份数据 ----------------
        site = Site(data).getData()

        # 点数量
        length = len(self.data)

        # ---------------- 计算三维平移向量 ----------------
        xdiff = p2[0] - p1[0]
        ydiff = p2[1] - p1[1]
        zdiff = p2[2] - p1[2]

        # ---------------- 执行平移 ----------------
        for i in range(length):

            data[i][0] = data[i][0] + xdiff
            data[i][1] = data[i][1] + ydiff
            data[i][2] = data[i][2] + zdiff

        # ---------------- 绘制平移后结构 ----------------
        Site(site).render()

    def render(self):
        """
        三维多边形绘制（四边形）
        """

        data = self.data

        # ---------------- 绘制方法（四边形边界） ----------------

        ax.plot(
            [data[0][0], data[1][0]],
            [data[0][1], data[1][1]],
            [data[0][2], data[1][2]],
            color="green"
        )

        ax.plot(
            [data[1][0], data[2][0]],
            [data[1][1], data[2][1]],
            [data[1][2], data[2][2]],
            color="green"
        )

        ax.plot(
            [data[2][0], data[3][0]],
            [data[2][1], data[3][1]],
            [data[2][2], data[3][2]],
            color="green"
        )

        ax.plot(
            [data[3][0], data[0][0]],
            [data[3][1], data[0][1]],
            [data[3][2], data[0][2]],
            color="green"
        )

# ---------------- 测试数据（四点空间多边形） ----------------
data = [
    [50, 30, 0],
    [285, 77, 0],
    [246, 273, 0],
    [10, 226, 0]
]

Site(data).render()

# ---------------- 平移参数 ----------------
p1 = [50, 30, 0]
p2 = [0, 0, 0]

Site(data).move(data, p1, p2)

plt.show()
```

```python
import matplotlib.pyplot as plt

# ---------------- 创建3D坐标系 ----------------
ax = plt.figure().add_subplot(111, projection='3d')

# 设置三维空间范围（教学固定视窗）
ax.set_xlim((0, 300))
ax.set_ylim((0, 300))
ax.set_zlim((0, 300))

class Site:
    def __init__(self, data_of_points):
        # data_of_points: [[x,y,z], [x,y,z], ...]
        self.data = data_of_points

    def getData(self):
        # 返回当前点集
        return self.data

    def edges(self):
        # 返回点数量（边数）
        return len(self.data)

    def move(self, data, p1, p2):
        """
        三维平移变换

        含义：
            将所有点从 p1 平移到 p2
        """

        # ---------------- 备份原始数据 ----------------
        site = Site(data).getData()

        # 点数量
        length = len(self.data)

        # ---------------- 计算平移向量 ----------------
        xdiff = p2[0] - p1[0]
        ydiff = p2[1] - p1[1]
        zdiff = p2[2] - p1[2]

        # ---------------- 执行平移 ----------------
        for i in range(length):

            data[i][0] += xdiff
            data[i][1] += ydiff
            data[i][2] += zdiff

        # ---------------- 绘制变换后图形 ----------------
        Site(site).render2()

    def render1(self):
        """
        绘制原始三维图形（绿色）
        ——初始状态
        """

        data = self.data

        # ---------------- 绘制方法（原始图形） ----------------

        ax.plot([data[0][0], data[1][0]],
                [data[0][1], data[1][1]],
                [data[0][2], data[1][2]],
                color="green")

        ax.plot([data[1][0], data[2][0]],
                [data[1][1], data[2][1]],
                [data[1][2], data[2][2]],
                color="green")

        ax.plot([data[2][0], data[3][0]],
                [data[2][1], data[3][1]],
                [data[2][2], data[3][2]],
                color="green")

        ax.plot([data[3][0], data[0][0]],
                [data[3][1], data[0][1]],
                [data[3][2], data[0][2]],
                color="green")

    def render2(self):
        """
        绘制变换后三维图形（红色）
        ——移动后状态
        """

        data = self.data

        # ---------------- 绘制方法（变换后图形） ----------------

        ax.plot([data[0][0], data[1][0]],
                [data[0][1], data[1][1]],

                color="red")

        ax.plot([data[1][0], data[2][0]],
                [data[1][1], data[2][1]],

                color="red")

        ax.plot([data[2][0], data[3][0]],
                [data[2][1], data[3][1]],
                [data[2][2], data[3][2]],
                color="red")

        ax.plot([data[3][0], data[0][0]],
                [data[3][1], data[0][1]],
                [data[3][2], data[0][2]],
                color="red")
```
