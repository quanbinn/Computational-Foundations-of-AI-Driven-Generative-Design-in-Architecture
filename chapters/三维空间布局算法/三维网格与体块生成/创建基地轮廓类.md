# 创建基地轮廓类

### 定义带构造函数的类

```python
class Site:
    """场地类（由多个顶点构成的多边形）"""
    
    def __init__(self, points):
        """初始化；points：顶点坐标列表，如 [[x1,y1,z1], ...]"""
        self.points = points

    def getPoints(self):
        """获取所有顶点"""
        return self.points

    def getEdges(self):
        """获取边数（顶点数）"""
        return len(self.points)

# 创建1个 Site 实例
site = Site([[50,30,0], [285,77,0], [246,273,0], [10,226,0]])

print(site.getPoints())
print(site.getEdges())
```

### 在类中添加多边形渲染方法

```python
import matplotlib.pyplot as plt

# ================= 画布 =================
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.set(xlim=(0, 300), ylim=(0, 300), zlim=(0, 300))

# ================= Site 类 =================
class Site:
    """场地类（由多个顶点构成的多边形）"""

    def __init__(self, points):
        """初始化场地；points：顶点列表 [[x1,y1,z1], ...]"""
        self.points = points

    def getData(self):
        """获取顶点数据"""
        return self.points

    def edges(self):
        """获取边数（顶点数）"""
        return len(self.points)

    def render(self, ax):
        """在指定 matplotlib 3D 坐标系上绘制边界"""
        n = len(self.points)
        for i in range(n):
            p1 = self.points[i]
            p2 = self.points[(i + 1) % n]  # 自动闭合
            ax.plot([p1[0], p2[0]],
                    [p1[1], p2[1]],
                    [p1[2], p2[2]],
                    color="green")

# ================= 使用 =================
site = Site([[50,30,0], [285,77,0], [246,273,0], [10,226,0]])
site.render(ax)

plt.show()
```
