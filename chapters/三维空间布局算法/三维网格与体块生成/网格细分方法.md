import matplotlib.pyplot as plt

# 创建3D绘图窗口和坐标轴

ax.set_xlim((0, 300))
ax.set_ylim((0, 300))
ax.set_zlim((0, 300))

width = 200

    [0, 0, 0],        # 左下角
    [length, 0, 0],   # 右下角
    [length, width, 0], # 右上角
    [0, width, 0]     # 左上角
]  # print(site)  # 保留硬编码注释，用于调试或检查坐标

# 绘制矩形基地的四条边
ax.plot([site[0][0], site[1][0]], [site[0][1], site[1][1]], [site[0][2], site[1][2]], color="green")  
ax.plot([site[1][0], site[2][0]], [site[1][1], site[2][1]], [site[1][2], site[2][2]], color="green")
ax.plot([site[2][0], site[3][0]], [site[2][1], site[3][1]], [site[2][2], site[3][2]], color="green")
ax.plot([site[3][0], site[0][0]], [site[3][1], site[0][1]], [site[3][2], site[0][2]], color="green")

# 显示3D图形
plt.show()

### 矩形基地长边与短边的等距细分点生成及三维可视化

```python
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D  # 导入3D绘图工具

# 创建3D绘图窗口和坐标轴
ax = plt.figure().add_subplot(111, projection='3d')

# 设置坐标轴范围
ax.set_xlim((0, 300))
ax.set_ylim((0, 300))
ax.set_zlim((0, 300))

# 定义矩形基地的长宽（z方向为0，表示平面）
length = 240
width = 200
site = [
    [0, 0, 0],          # 左下角
    [length, 0, 0],     # 右下角
    [length, width, 0], # 右上角
    [0, width, 0]       # 左上角
]  # print(site)  # 保留硬编码注释，用于检查坐标

# 绘制矩形基地的四条边
ax.plot([site[0][0], site[1][0]], [site[0][1], site[1][1]], [site[0][2], site[1][2]], color="green")  
ax.plot([site[1][0], site[2][0]], [site[1][1], site[2][1]], [site[1][2], site[2][2]], color="green")
ax.plot([site[2][0], site[3][0]], [site[2][1], site[3][1]], [site[2][2], site[3][2]], color="green")
ax.plot([site[3][0], site[0][0]], [site[3][1], site[0][1]], [site[3][2], site[0][2]], color="green")

# 设置长边和短边的等距采样间距
distance_in_length = 16
distance_in_width = 16

# 计算沿长边和短边可分的段数
segments_in_length = int(length / distance_in_length)        # print(segments_in_length)
segments_in_width  = int(width / distance_in_width)          # print(segments_in_width)

# -----------------------------
# 沿长边的等距采样点 (底边)
xArrayInLength = []
yCoordInLength = []
zCoordInLength = []

for i in range(1, segments_in_length):
    xArrayInLength.append(distance_in_length * i)
    yCoordInLength.append(0)  # y固定为0
    zCoordInLength.append(0)  # z固定为0

print(xArrayInLength)  # 打印长边采样点x坐标，教学用途

# -----------------------------
# 沿短边的等距采样点 (左边)
xCoordInWidth = []
yArrayInWidth = []
zCoordInWidth = []

for i in range(1, segments_in_width):
    xCoordInWidth.append(0)  # x固定为0
    yArrayInWidth.append(distance_in_width * i)
    zCoordInWidth.append(0)  # z固定为0

print(yArrayInWidth)  # 打印短边采样点y坐标，教学用途

# 绘制采样点
ax.scatter(xArrayInLength, yCoordInLength, zCoordInLength, c="blue")  # 长边采样点，蓝色
ax.scatter(xCoordInWidth, yArrayInWidth, zCoordInWidth, c="red")      # 短边采样点，红色

# 显示3D图形
plt.show()
```
